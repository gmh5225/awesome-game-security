`README.md`:

```md
# Thats no pipe

Frida-based project to intercept Windows named pipe communication

## Objectives

This project aims at intercepting IPC communications by hooking specific WinAPI functions and forwarding them to an HTTP proxy using a websocket.

## Architecture

Read from and write to a named pipe can happen in different functions. In this section, WriteFile depicts a write to a named pipe and ReadFile depicts a read from a named pipe.

When intercepted, data to be written follows this schema.

```
WriteFile --> frida hook --> python process --> websocket client --> HTTP Proxy --> websocket server --> python process --> frida hook --> WriteFile
```

On the other side, data to be read by the process follows this schema.

```
ReadFile --> frida hook --> python process --> websocket server --> HTTP Proxy --> websocket client --> python process --> frida hook --> ReadFile
```

The HTTP Proxy will receive a message "to the server" when the action is a write to a named pipe, and a message "to the client" when the action is the read from a named pipe.

## Setup

### Using Burp as an HTTP Proxy

Add a proxy on port 8090, listening on all interfaces

### Build the project

Build the frida agent script using `make build`, the agent will be in `agent/dist/`

### Windows setup

1. Copy the `backend/*.py` and `backend/requirements.txt` (backend) and `agent/dist/agent.js` (agent) to the windows host
2. `py -m pip install -r requirements.txt`
3. `py .\main.py <pid_of_target> .\agent.js --ws-port 9090 --ws-host <ip_of_windows> --http-proxy-port 8090 --http-proxy-host <ip_of_burp> --handle-regexes '.*' --log-level INFO`

## Disclamer

Data sent to the HTTP Proxy follows a specific format defined in `docs/format.md`, tools to manipulate this custom format are provieded in `tools/`.
**Modifying metadata can result in unability of the project to recognize and correctly route the message to the correct named pipe.**

The frida script indefinitely waits for a response, therefore the process **can hang indefinitely** if you decide to drop the message.

Injecting new messages is currently not supported by the project.

## Development

### LSP

Frida provides typescript bindings to develop the agent. Once devDependencies have been installed, VSCode should recognize types automagically.

A Makefile target `make dev-env` is provided to easily setup a development environment.
```

```ts
export const randomId = (length = 10) => {
  let result = '';
  const characters = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
  const charactersLength = characters.length;
  let counter = 0;
  while (counter < length) {
    result += characters.charAt(Math.floor(Math.random() * charactersLength));
    counter += 1;
  }
  return result
}

// funcName and message are mandatory
export const sendMsg = ({ funcName, message, id, handleId, handlePath }: { funcName: string, message: number[] | ArrayBuffer, id?: string, handleId?: number, handlePath?: string }) => {
  if (!id) {
    id = "auto_" + randomId();
  }
  if (!handleId) {
    handleId = 0
  }
  if (!handlePath) {
    handlePath = "generic_pipe"
  }
  const payload = JSON.stringify({ funcName, id, handleId, handlePath })
  send(payload, message)
  return id
}
// identifier and callback are mandatory
export const recvMsg = ({ identifier, callback, wait }: { identifier: string, callback: MessageCallback, wait?: boolean }) => {
  const msgRecv = recv(identifier, callback)
  if (!!wait) {
    msgRecv.wait()
  }
}

// Get a read operation from the data
// if dontwait is true, the operation returns "immediately" with NO_DATA if the pipe is empty
export const popReadOperation = ({ identifier, callback, handleId, bufferLength, dontwait }: { identifier?: string, callback: (status: string, payload: number[]) => void, handleId: number, bufferLength: number, dontwait?: boolean }) => {
  if (!identifier) {
    identifier = "auto_" + randomId();
  }
  const payload = JSON.stringify({ funcName: "ThatsNoPipeGetReadData", id: identifier, handleId, bufferLength, dontwait })
  send(payload)
  const msgRecv = recv(identifier, (value) => callback(value.status, value.payload))
  msgRecv.wait()
}

  SUCCESS="SUCCESS",
  FAILED="FAILED",
  BUFFER_TOO_SMALL="BUFFER_TOO_SMALL",
```

```ts
import { popReadOperation, ReadResultStatus, recvMsg, sendMsg } from "./communication.js";
import { log } from "./globals.js";
import { ERROR_INSUFFICIENT_BUFFER, ERROR_OPERATION_ABORTED, ERROR_SUCCESS } from "./hooks.js";
import { logOverlappedOperations, popOverlappedOperationByOverlapped } from "./overlapped_operations.js";

const attachedFunctions: number[] = [];
const isHooked = (addr: NativePointer) => {
  return attachedFunctions.indexOf(addr.toInt32()) !== -1
}

const completionRoutineOnEnter = (args: InvocationArguments) => {
  const dwErrorCode = args[0];
  const dwNumberOfBytesTransfered = args[1];
  const lpOverlapped = args[2];

  log("CustomCompletionRoutine-e:", dwErrorCode, dwNumberOfBytesTransfered, lpOverlapped)

  if (dwErrorCode.toInt32() !== 0) {
    // The I/O completion was not successful, do nothing
    return
  }

  // Otherwise, we need to check if the overlapped is known
  const overlappedOperation = popOverlappedOperationByOverlapped(lpOverlapped);
  if (overlappedOperation === undefined) {
    // Probably because the routine was called using tracked and untracked pipes
    // therefore, not an error
    return
  }

  const handleId = overlappedOperation.handleId;
  const handlePath = overlappedOperation.handlePath;
  const buffer = overlappedOperation.pBuffer;
  const bufferLength = overlappedOperation.bufferLength;
  const identifier = sendMsg({
    id: "to_ReadOperations",
    funcName: "CustomCompletionRoutine",
    message: buffer.readByteArray(dwNumberOfBytesTransfered.toInt32()) ?? new ArrayBuffer(0),
    handlePath,
    handleId
  })

  popReadOperation({
    handleId: handleId,
    bufferLength: bufferLength,
    callback: (status, payload) => {
      log("CustomCompletionRoutine-m:", status, payload)
      if (ReadResultStatus.SUCCESS === status) {
        if (bufferLength >= payload.length) {
          lpOverlapped.add(Process.pointerSize).writeU32(payload.length);
          args[1] = ptr(payload.length);
          args[0] = ptr(ERROR_SUCCESS);
          buffer.writeByteArray(payload);
        } else { 
          console.error("CustomCompletionRoutine: Buffer is shorter than data read!")
        }
      } else if (ReadResultStatus.FAILED === status) {
        lpOverlapped.add(Process.pointerSize).writeU32(0);
        args[1] = ptr(0);
        args[0] = ptr(ERROR_OPERATION_ABORTED); // ERROR_OPERATION_ABORTED
        // result.replace(ptr(ERROR_IO_INCOMPLETE)); // ERROR_IO_INCOMPLETE
        // Replace data so that we completely hide data from the process
        // even if it does not follow the spec
        buffer.writeByteArray(Array<number>(lpOverlapped.add(Process.pointerSize).readU32()).fill(0));
      } else if (ReadResultStatus.BUFFER_TOO_SMALL === status) {
        lpOverlapped.add(Process.pointerSize).writeU32(0);
        args[1] = ptr(0);
        args[0] = ptr(ERROR_INSUFFICIENT_BUFFER); // ERROR_INSUFFICIENT_BUFFER
        // Replace data so that we completely hide data from the process
        // even if it does not follow the spec
        buffer.writeByteArray(Array<number>(lpOverlapped.add(Process.pointerSize).readU32()).fill(0));
      } else if (ReadResultStatus.NO_DATA === status) {
        // Thats weird, we shouldn't be in that case
        lpOverlapped.add(Process.pointerSize).writeU32(0);
        args[1] = ptr(0);
        args[0] = ptr(ERROR_SUCCESS);
        // Replace data so that we completely hide data from the process
        // even if it does not follow the spec
        buffer.writeByteArray(Array<number>(lpOverlapped.add(Process.pointerSize).readU32()).fill(0));
      } else {
        console.error(`CustomCompletionRoutine: Unhandled status code ${status}`)
      }
    }
  })
}

const completionRoutineOnLeave = (result: InvocationReturnValue) => {
  // Do stuff
}

export const hookCompletionRoutine = (routineAddr: NativePointer) => {
  if (!isHooked(routineAddr)) {
    log(`Hooking function ${routineAddr}`)
    Interceptor.attach(routineAddr, {
      onEnter: completionRoutineOnEnter,
      onLeave: completionRoutineOnLeave,
    })
    attachedFunctions.push(routineAddr.toInt32())
  }
};
```

```ts
import { pushHeapAllocatedBuffer } from "../lib/heap.js";
import { log } from '../globals.js';
import { writeFile } from '../hooks.js';

const createFileAddr = Module.getExportByName(null, "CreateFileA");
const createFile = new SystemFunction(createFileAddr, 'pointer', ['pointer', 'uint32', 'uint32', 'pointer', 'uint32', 'uint32', 'pointer']);
const GENERIC_READ = 0x80000000;
const GENERIC_WRITE = 0x40000000;
const OPEN_EXISTING = 3;
const FILE_FLAG_OVERLAPPED = 0x40000000;

const connectToPipe = (pipePath: string) => {
  const finalPath = "\\\\.\\pipe" + pipePath;
  const lpFileName = Memory.alloc(finalPath.length);
  lpFileName.writeUtf8String(finalPath);
  log(lpFileName.readByteArray(finalPath.length));
  const result = createFile(lpFileName, GENERIC_READ | GENERIC_WRITE, 0, ptr(0), OPEN_EXISTING, FILE_FLAG_OVERLAPPED, ptr(0x0)) as WindowsSystemFunctionResult<NativePointer>;
  log(result.value, result.lastError);
  return result.value
}

const sendDataToPipeFromBuffer = (pipeHandle: NativePointer, buffer: NativePointer, size: number) => {
  const bytesWritten = Memory.alloc(0x4);
  pushHeapAllocatedBuffer(bytesWritten);
  log("Calling WriteFile:", pipeHandle, buffer, size, bytesWritten, ptr(0))
  const success = writeFile(pipeHandle, buffer, size, bytesWritten, ptr(0));
  return [success, bytesWritten.readU32()];
}

const sendDataToExistingPipe_internal = (handle: NativePointer, payload: ArrayBuffer) => {
  log(`Sending the following data (size ${payload.byteLength}):\n` + hexdump(payload))
  const buffer = Memory.alloc(payload.byteLength);
  pushHeapAllocatedBuffer(buffer);
  buffer.writeByteArray(payload);
  const [success, nbBytesWritten] = sendDataToPipeFromBuffer(handle, buffer, payload.byteLength);
  log("Sent data to pipe", success === 0 ? "failed" : "success", nbBytesWritten)
}

export const sendDataToExistingPipe = (handle: number, payload: ArrayBuffer) => {
  return sendDataToExistingPipe_internal(ptr(handle), payload)
}

export const sendDataToPipe = (pipeName: string, payload: ArrayBuffer) => {
  const handle = connectToPipe(pipeName);
  log("Connected to named pipe");
  sendDataToExistingPipe_internal(handle, payload);
}
```

```ts
import {

  getQueuedCompletionStatusExAddr,
  getLastError,

import {
  recvMsg,
  sendMsg
} from './communication.js'
import { log } from './globals.js';
import {
  getLpNumberOfBytesTransferred,
  logOverlappedOperations,
  popOverlappedOperationByOverlapped
} from './overlapped_operations.js';

export const hookGetQueuedCompletionStatus = () => {
  Interceptor.attach(getQueuedCompletionStatusAddr, {
    onEnter: function(args) {
      const completionPort = args[0];
      const lpNumberOfBytesTransferred = args[1];
      const lpCompletionKey = args[2];
      const lpOverlapped = args[3];
      const dwMilliseconds = args[4];

      log("GetQueuedCompletionStatus-e:", completionPort, lpNumberOfBytesTransferred, lpCompletionKey, lpOverlapped, dwMilliseconds)
      this.completionPort = completionPort;
      this.lpNumberOfBytesTransferred = lpNumberOfBytesTransferred;
      this.lpCompletionKey = lpCompletionKey;
      this.lpOverlapped = lpOverlapped;
      this.dwMilliseconds = dwMilliseconds;
      this.doIntercept = true;
    },
    onLeave: function(result) {
      if (!this.doIntercept) {
        return
      }
      const completionPort: NativePointer = this.completionPort;
      const lpNumberOfBytesTransferred: NativePointer = this.lpNumberOfBytesTransferred;
      const lpCompletionKey: NativePointer = this.lpCompletionKey;
      const lpOverlapped: NativePointer = this.lpOverlapped;
      const dwMilliseconds: NativePointer = this.dwMilliseconds;

      const returnedLpOverlapped = lpOverlapped.readPointer();
      const overlappedOperation = popOverlappedOperationByOverlapped(returnedLpOverlapped);
      log("GetQueuedCompletionStatus-l:", completionPort, lpNumberOfBytesTransferred, lpCompletionKey, lpOverlapped, dwMilliseconds, result)
      if (result.toInt32() === 0) {
        // The syscall failed, return
        // The overlapped was still triggered, we need to forget about it
        // Following is a SYSCALL
        const lasterr = getLastError();
        log("GetQueuedCompletionStatus-e: failed with", result.toInt32(), lasterr.value)
        return
      }

      log("GetQueuedCompletionStatus-l-metadata:", completionPort, lpNumberOfBytesTransferred, lpCompletionKey, lpOverlapped, dwMilliseconds, "R", result.toInt32(), returnedLpOverlapped, overlappedOperation === undefined ? "undefined": overlappedOperation.pOverlapped);
      logOverlappedOperations()
      if (overlappedOperation === undefined) {
        // Does not correspond to a known lpOverlapped
        return
      }

      // Ths correspond to a known overlapped, we need to intercept the buffer
      const bufferLength = getLpNumberOfBytesTransferred(overlappedOperation).readU32();
      const buffer = overlappedOperation.pBuffer;
      const handlePath = overlappedOperation.handlePath;
      const handleId = overlappedOperation.handleId;
      if (bufferLength === undefined) {
        console.error("GetQueuedCompletionStatus: Could not find the buffer length, cannot intercept")
        return
      }

      const identifier = sendMsg({
        funcName: "GetQueuedCompletionStatus",
        message: buffer.readByteArray(bufferLength) ?? new ArrayBuffer(0),
        handlePath,
        handleId
      })
      recvMsg({
        identifier,
        callback: (value) => {
          const data = value.payload;
          // If length is 0, it will trigger an error, instead, change the return value of ReadFile
          const mBufferLength = data.length;
          // TODO: handle modified data
          // const mBuffer = Memory.alloc(mBufferLength);
          // mBuffer.writeByteArray(data);
          // pushHeapAllocatedBuffer(mBuffer);

          // this.buffer = mBuffer;
          // this.nbBytesRead = new NativePointer(mBufferLength);
        },
        wait: true
      })
    }
  })
}
```

```ts
export const nbArrayEqual = (arr1: Uint8Array, arr2: Uint8Array) => {
  for (let i = arr1.length; -1 < i; i -= 1) {
    if ((arr1[i] !== arr2[i])) return false;
  }
  return true;
```

```ts
import {
  FILE_TYPE_PIPE,
  getFileType,
  getFileInformationByHandleEx,
  FileNameInfo
} from '../hooks.js'

let targetHandlesPath: RegExp[] = [];
const blacklistedHandlePaths: RegExp[] = [/\\frida-.*/];

export const isTargetHandlePath = (handlePath: string) => {
  if (handlePath === "") {
    return false
  }
  // Proactively blacklist frida communication from being intercepted
  if (blacklistedHandlePaths.some((el) => el.test(handlePath))) {
    return false;
  }
  return targetHandlesPath.some((el) => el.test(handlePath));
};
export const setHandleRegexes = (regexes: string[]) => {
  targetHandlesPath = regexes.map((value) => new RegExp(value));
}

export const getPathByHandle = (handle: NativePointer) => {
  const size = 512;
  // Following function is a SYSCALL
  const fileType = getFileType(handle);
  if (fileType !== FILE_TYPE_PIPE) {
    return "";
  }
  const lpBuffer = Memory.alloc(size);
  // Following function is a SYSCALL
  const retval = getFileInformationByHandleEx(handle, FileNameInfo, lpBuffer, size) as WindowsSystemFunctionResult<number>;
  const status = retval.value;
  const lasterror = retval.lastError;
  if (status === 0) {
    console.error(`Error while calling GetFileInformationByHandleEx: ${lasterror}`);
    return "";
  };
  const fileNameLength = lpBuffer.readU32();
  const fileNameInfo = lpBuffer.add(0x04).readUtf16String(fileNameLength);
  return fileNameInfo ?? "";
};
```

```ts
type ApcContext = {
    apcRoutine: NativePointer,
    arg_0: NativePointer,
    arg_1: NativePointer,
    arg_2: NativePointer,
}

const GlobalApcContexts: { [id:number]: ApcContext } = {}

const randU32Sync = () => {
    return Math.floor(Math.random() * 0xffffffff)
}

// The following function will be executed in the context of the target thread
const UserAPCDispatcher3argsCallback = new NativeCallback((idContext: number) => {
    // Retrieve the context
    const apcContext = GlobalApcContexts[idContext]
    log("Calling original routine")
    const routine = new SystemFunction(apcContext.apcRoutine, 'void', ['pointer', 'pointer', 'pointer'])
    const { arg_0, arg_1, arg_2 } = apcContext
    delete GlobalApcContexts[idContext]
    // Finally call the routine
    routine(apcContext.arg_0, apcContext.arg_1, apcContext.arg_2)
}, 'void', ['uint32'])

export const QueueAPC = (apcRoutine: NativePointer, threadId: number, arg_0: NativePointer, arg_1: NativePointer, arg_2: NativePointer) => {
    const openThreadRet = OpenThread(THREAD_GET_CONTEXT | THREAD_SET_CONTEXT, 0, threadId) as WindowsSystemFunctionResult<number>;
    if (openThreadRet.value === 0) {
        log("OpenThread: failed to open handle to thread:", openThreadRet.lastError)
        return
    }
    const threadHandle = openThreadRet.value
    log("QueueUserAPC:", apcRoutine, threadHandle, arg_0, arg_1, arg_2)
    log("QueueUserAPC apcRoutine hexdump:","\n"+hexdump(apcRoutine, {length: 300}))
    // Kernel is allowed to queue APC with 3 arguments, but we are not, we need to create a custom dispatcher that will call the
    // apcRoutine with correct arguments
    const idContext = randU32Sync()
    GlobalApcContexts[idContext] = {
        apcRoutine,
        arg_0,
        arg_1,
        arg_2,
    }
    const queueUserApcRet = QueueUserAPC(UserAPCDispatcher3argsCallback, threadHandle, ptr(idContext)) as WindowsSystemFunctionResult<number>;
    if (queueUserApcRet.value === 0) {
        log("QueueUserAPC: failed queue APC:", queueUserApcRet.lastError)
    }
    CloseThreadHandle(threadHandle)
}
```

```ts
              // even if it does not follow the spec
              this.Buffer.writeByteArray(Array<number>(this.IoStatusBlock_Information.readU32()).fill(0));
            } else {

        // asynchronous operations or call the completion routine
        return

    } as ((this: InvocationContext, args: InvocationReturnValue) => void)
```

```ts
export const hookNtWriteFile = () => {
  Interceptor.attach(NtWriteFileAddr, {
    onEnter: function(this: NtWriteFileInvocationContext, args: InvocationArguments) {
      this.FileHandle = args[0]; // [in] HANDLE
      this.Event = args[1]; // [in, optional] HANDLE
      this.ApcRoutine = args[2]; // [in, optional] PIO_APC_ROUTINE
      this.ApcContext = args[3]; // [in, optional] PVOID
      this.IoStatusBlock = args[4]; // [out] PIO_STATUS_BLOCK
      this.Buffer = args[5]; // [in] PVOID
      this.Length = args[6]; // [in] ULONG
      this.ByteOffset = args[7] // [in, optional] PLARGE_INTEGER
      this.Key = args[8] // [in, optional] PULONG

      // Check if the Handle is a NamedPipe, and if we should intercept it
      const fileHandlePath = getPathByHandle(this.FileHandle)

      if (!isTargetHandlePath(fileHandlePath)) {
        this.doLogSyscall = false
        return
      }
      if (isFridaScriptBuffer(this.Buffer)) {
        // We detected an injected WriteFile operation, prevent frida from intercepting itself
        this.doLogSyscall = false
        log("NtWriteFile-e: detected self buffer")
        return
      }
      this.doLogSyscall = true

      log("NtWriteFile-e:", this.FileHandle, this.Event, this.ApcRoutine, this.ApcContext, this.IoStatusBlock, this.Buffer, this.Length, this.ByteOffset, this.Key)

      const message = this.Buffer.readByteArray(this.Length.toInt32()) ?? new ArrayBuffer(0)
      const identifier = sendMsg({
        funcName: "NtWriteFile",
        message,
        handlePath: fileHandlePath,
        handleId: this.FileHandle.toInt32()
      })
      recvMsg({
        identifier,
        callback: (value) => {
          const data: number[] = value.payload;
          const dataArrayBuffer = new Uint8Array(data);
          if (!arrayBufferEqual(dataArrayBuffer.buffer, message)) {
            if (data.length <= this.Length.toInt32()) {
              // We just need to overwrite the buffer
              this.Buffer.writeByteArray(data)
              // Make sure subsequent data are zeroed
              this.Buffer.add(data.length).writeByteArray(Array.from(new Uint8Array(this.Length.toInt32() - data.length)))
              // Modify the buffer length of the writefile call
              args[6] = ptr(data.length)
            } else {
              // We need to allocate a new buffer
              const mBuffer = Memory.alloc(data.length)
              mBuffer.writeByteArray(data)
              pushHeapAllocatedBuffer(mBuffer);
              args[5] = mBuffer;
              args[6] = ptr(data.length);
            }
          }
        },
        wait: true
      })
    } as ((this: InvocationContext, args: InvocationArguments) => void),
    onLeave: function(this: NtWriteFileInvocationContext, result: InvocationReturnValue) {
      this.ResultNtStatus = result // NTSATUS

      if (!!this.doLogSyscall) {
        log("NtWriteFile-l:", this.FileHandle, this.Event, this.ApcRoutine, this.ApcContext, this.IoStatusBlock, this.Buffer, this.Length, this.ByteOffset, this.Key, this.ResultNtStatus)
      }
    } as ((this: InvocationContext, args: InvocationReturnValue) => void)
  })
}
```

```ts
import { config } from "./globals.js";
import { getOverlappedResult } from "./hooks.js";

type OverlappedOperation = {
  pOverlapped: NativePointer;
  hEvent: number;
  pBuffer: NativePointer;
  bufferLength: number;
  pNbBytesRead?: NativePointer;
  handleId: number;
  handlePath: string;
}

const overlappedOperations: OverlappedOperation[] = []

export const pushOverlappedOperation = (overlappedOperation: OverlappedOperation) => {
  const index = overlappedOperations.findIndex((registeredOverlappedOperation) => registeredOverlappedOperation.pOverlapped.toUInt32() === overlappedOperation.pOverlapped.toUInt32())
  if (index !== -1) {
    // Not found, aborting
    console.warn(`Another overlapped operation is registered with the same address ${overlappedOperation.pOverlapped}`)
  }
  overlappedOperations.push(overlappedOperation)
}

export const logOverlappedOperations = () => {
  const lpOverlappedList = overlappedOperations.map((overlappedOperation) => overlappedOperation.pOverlapped);
  if (config.logSyscalls) {
    console.log("OverlappedOperation[]:", ...lpOverlappedList)
  }
}

export const getLpNumberOfBytesTransferred = (overlappedOperation: OverlappedOperation) => {
  if (overlappedOperation.pNbBytesRead !== undefined && overlappedOperation.pNbBytesRead.toInt32() != 0) {
    return overlappedOperation.pNbBytesRead
  } else {
    const lpNumberOfBytesTransferred = overlappedOperation.pOverlapped.add(Process.pointerSize).readPointer();
    return lpNumberOfBytesTransferred;
  }
}

export const popOverlappedOperationByOverlapped = (pOverlapped: NativePointer) => {
  const index = overlappedOperations.findIndex((overlappedOperation) => overlappedOperation.pOverlapped.toUInt32() === pOverlapped.toUInt32())
  if (index === -1) {
    // Not found, aborting
    return undefined
  }
  const ops = overlappedOperations.splice(index, 1);
  if (ops.length !== 1) {
    // We should never end in this branch
    console.error(`Found index ${index} but no elements removed from list!`)
    return undefined
  }
  return ops[0];
}

export const findOverlappedOperationByEvent = (hEvent: NativePointer) => {
  const index = overlappedOperations.findIndex((overlappedOperation) => overlappedOperation.hEvent === hEvent.toInt32())
  return overlappedOperations[index];
}

export const popOverlappedOperationByEvent = (hEvent: NativePointer) => {
  const index = overlappedOperations.findIndex((overlappedOperation) => overlappedOperation.hEvent === hEvent.toInt32())
  if (index === -1) {
    // Not found, aborting
    return undefined
  }
  const ops = overlappedOperations.splice(index, 1)
  if (ops.length !== 1) {
    // We should never end in this branch
    console.error(`Found index ${index} but no elements removed from list!`)

  }
```

```ts
export const hookGetOverlappedResult = () => {
  Interceptor.attach(getOverlappedResultAddr, {
    onEnter: function(args) {
      // See Microsoft documentation for meaning of all arguments
      const handle = args[0];
      const lpOverlapped = args[1];
      const nbBytesTransferred = args[2];
      const bWait = args[3];

      const handlePath = getPathByHandle(handle);

      if (!isTargetHandlePath(handlePath)) {
        // Not something we monitor, exit
        return
      }
      const overlappedOperation = popOverlappedOperationByOverlapped(lpOverlapped)

      this.lpOverlapped = lpOverlapped
      this.nbBytesTransferred = nbBytesTransferred
      this.handlePath = handlePath
      this.handleId = handle.toInt32()

      log("GetOverlappedResult-e:", handle, lpOverlapped, overlappedOperation !== undefined ? "TRACKED" : "unk");
      if (overlappedOperation === undefined) {
        // No overlapped operation could be found, that is a problem...
        console.warn(`GetOverlappedResult found an untracked overlapped structure at ${lpOverlapped} associated with a target handle ${handle} (${handlePath})`)
        return
      }

      // Save all context data so that we can access them after the syscall
      this.doIntercept = true
      this.lpOverlapped = lpOverlapped
      this.buffer = overlappedOperation.pBuffer
      this.nbBytesTransferred = nbBytesTransferred
      this.handlePath = handlePath
      this.handleId = handle.toInt32()
    },
    onLeave: function(result) {
      if (!this.doIntercept) {
        return;
      }
      if (result.toInt32() === 0) {
        // Function failed
        return;
      }

      // Restore the context
      const lpOverlapped: NativePointer = this.lpOverlapped;
      const buffer: NativePointer = this.buffer;
      const bufferLength = (this.nbBytesTransferred as NativePointer).readU32();
      const handlePath: string = this.handlePath;
      const handleId: number = this.handleId;

      log("GetOverlappedResult-l:", lpOverlapped)
      const identifier = sendMsg({
        funcName: "GetOverlappedResult",
        message: buffer.readByteArray(bufferLength) ?? new ArrayBuffer(0),
        id: 'to_ReadOperations',
        handlePath,
        handleId
      })
      popReadOperation({
        handleId,
        bufferLength,
        callback: (status, payload) => {
          log("GetOverlappedResult", status, payload)
        }
      })
    }
  })
}
```

```ts
import {
  readFileAddr,
  readFileExAddr,
  ERROR_IO_PENDING,
  overlappedEventOffset,
  readFile,
} from './hooks.js';
import {
  popReadOperation,
  recvMsg,
  sendMsg
} from './communication.js'
import { hookCompletionRoutine } from './dynamic_hook.js';
import { log } from './globals.js';
import { logOverlappedOperations, popOverlappedOperationByOverlapped, pushOverlappedOperation } from './overlapped_operations.js';
import {
  isTargetHandlePath,
  getPathByHandle,
} from './lib/handle_filtering.js'

export const hookReadFileEx = () => {
  Interceptor.attach(readFileExAddr, {
    onEnter: function(args) {
      // See Microsoft documentation for meaning of all arguments
      const hFile = args[0];
      const lpBuffer = args[1];
      const nNumberOfBytesToRead = args[2].toInt32();
      const lpOverlapped = args[3];
      const lpCompletionRoutine = args[4];

      const handlePath = getPathByHandle(hFile);

      if (!isTargetHandlePath(handlePath)) {
        this.doIntercept = false;
        return
      }
      if (lpCompletionRoutine.toInt32() !== 0) {
        // The developper probably wanted to handle the result of the read in a completion routine
        // therefore we need to ensure that function is hooked
        hookCompletionRoutine(lpCompletionRoutine)
      }

      // This is an overlapped operation
      // register the overlapped operation for further use
      pushOverlappedOperation({
        pOverlapped: lpOverlapped,
        pBuffer: lpBuffer,
        bufferLength: nNumberOfBytesToRead,
        pNbBytesRead: undefined, // Should be retrieved in the Overlapped structure
        hEvent: lpOverlapped.add(overlappedEventOffset).readU32(),
        handleId: hFile.toInt32(),
        handlePath: handlePath,
      })
      this.isOverlapped = true;
      this.lpOverlapped = lpOverlapped;

      log("ReadFileEx-e:", hFile.toInt32(), lpBuffer, nNumberOfBytesToRead, lpOverlapped, lpCompletionRoutine, "overlapped", lpOverlapped.add(overlappedEventOffset).readU32())
    },
    // TODO: Check return code to avoid waiting for an overlapped that will never be used
  })
}

export const hookReadFile = () => {
  Interceptor.attach(readFileAddr, {
    onEnter: function(args) {
      // See Microsoft documentation for meaning of all arguments
      const hFile = args[0];
      const lpBuffer = args[1];
      const nNumberOfBytesToRead = args[2].toInt32();
      const lpNumberOfBytesRead = args[3];
      const lpOverlapped = args[4];

      const handlePath = getPathByHandle(hFile);

      if (!isTargetHandlePath(handlePath)) {
        this.doIntercept = false;
        return
      }

      // Save all context data so that we can access them after the syscall
      this.doIntercept = true
      this.buffer = lpBuffer
      this.nbBytesRead = lpNumberOfBytesRead
      this.handlePath = handlePath
      this.handleId = hFile.toInt32()

      // Check if there are already data waiting in the pipe
      this.statusCode = "NO_DATA";
      this.readData = [];
      const data = popReadOperation({
        callback: (status, payload) => {
          this.statusCode = status;
          this.readData = payload;
        },
        handleId: hFile.toInt32(),
        bufferLength: nNumberOfBytesToRead,
        dontwait: true,
      });
      if (this.statusCode != 'NO_DATA') {
        // Intercept call and nullify the syscall
        args[0] = ptr(0) // Invalid Handle, should be enough to make the syscall fail
        args[1] = ptr(0)
        args[2] = ptr(0)
        args[3] = ptr(0)
        args[4] = ptr(0)
        log("ReadFile-e: nullifying syscall");
        return
      }

      if (lpOverlapped.toInt32() === 0) {
        this.isOverlapped = false;

        log("ReadFile-e:", hFile, lpBuffer, nNumberOfBytesToRead, lpNumberOfBytesRead, lpOverlapped, "direct")
      } else {

        log("ReadFile-e:", hFile, lpBuffer, nNumberOfBytesToRead, lpNumberOfBytesRead, lpOverlapped, "overlapped", lpOverlapped.add(overlappedEventOffset).readU32())
        // This is an overlapped operation
        // register the overlapped operation for further use in getOverlappedResult
        pushOverlappedOperation({
          pOverlapped: lpOverlapped,
          pBuffer: lpBuffer,
          bufferLength: nNumberOfBytesToRead,
          pNbBytesRead: lpNumberOfBytesRead,
          hEvent: lpOverlapped.add(overlappedEventOffset).readU32(),
          handleId: this.handleId,
          handlePath: handlePath,
        })
        this.isOverlapped = true;
        this.lpOverlapped = lpOverlapped;
      }
    },
    onLeave: function(result) {
      if (!this.doIntercept) {
        return;
      }

      // Restore the context
      const lpBuffer: NativePointer = this.buffer;
      const lpNumberOfBytesRead: NativePointer = this.nbBytesRead;
      const handlePath: string = this.handlePath;
      const handleId: number = this.handleId;
      const isOverlapped: boolean = this.isOverlapped;
      const lpOverlapped: NativePointer = this.lpOverlapped;
      const retCode = result.toInt32();

      const statusCode: string = this.statusCode;
      const readData: number[] = this.readData;

      if (statusCode != "NO_DATA") {
        if (statusCode == "BUFFER_TOO_SMALL") {
          // TODO: handle
        } else if (statusCode == "SUCCESS") {
          if (!isOverlapped) {
            // Case 1, synchronous read
            result.replace(ptr(1)); // Read Success
            lpNumberOfBytesRead.writeInt(readData.length);
            lpBuffer.writeByteArray(readData);
          } else {
            // CASE 2, asynchronous read
            // TODO
          }
        }
      }

      const nbBytesRead = lpNumberOfBytesRead.readU32();
      log("ReadFile-l:", handleId, lpBuffer, "unk", nbBytesRead, lpOverlapped, isOverlapped ? "overlapped" : "direct", retCode)

      // If the return code is non-0 and we read bytes, the operation completed successfully
      if (retCode !== 0 && lpNumberOfBytesRead.readU32() !== 0) {
        const identifier = sendMsg({
          funcName: "ReadFile",
          message: lpBuffer.readByteArray(nbBytesRead) ?? new ArrayBuffer(0),
          handlePath,
          handleId
        })
        recvMsg({
          identifier,
          callback: (value) => {
            const data = value.payload;
            // If length is 0, it will trigger an error, instead, change the return value of ReadFile
            const mBufferLength = data.length;
            // TODO: handle modified data

            // mBuffer.writeByteArray(data);
            // pushHeapAllocatedBuffer(mBuffer);

            // this.buffer = mBuffer;
            // this.nbBytesRead = new NativePointer(mBufferLength);
          },
          wait: true
        })
        return
      }

      // If the operation was async (overlapped), and we got an error that is ERROR_IO_PENDING,
      // we need to keep the event registered to intercept it during other syscalls
      if (isOverlapped && retCode !== ERROR_IO_PENDING) {
        log("ReadFile: expecting other call with following overlapped", lpOverlapped)
        return
      }

      // Otherwise, there is an error in the syscall, we unregister the event
      const overlappedOperation = popOverlappedOperationByOverlapped(lpOverlapped);
      if (isOverlapped && overlappedOperation === undefined) {
        console.warn("ReadFile: Wanted to unregister an overlapped that was not registered", lpOverlapped)
      }
    }
```

```ts
  waitForSingleObjectExAddr,
} from './hooks.js';
import { log } from './globals.js';

export const hookWaitForSingleObject = () => {
  Interceptor.attach(waitForSingleObjectAddr, {
    onEnter: function(args) {
      const hHandle = args[0];
      const dwMilliseconds = args[1];

      // Check if we know that event
      const overlappedOperation = findOverlappedOperationByEvent(hHandle);
      if (!overlappedOperation) {
        // We don't know that event, return
        return
      }
      this.doIntercept = true;
      log("WaitForSingleObject-e:", hHandle.toInt32(), dwMilliseconds.toInt32(), overlappedOperation?.pOverlapped)
    },
    onLeave: function(result) {
      if (!this.doIntercept) {
        return;
      }
      log("WaitForSingleObject-l:", result)
    }
  })
}
```

```py
        """

        the returned instance.
        """
        if cls not in cls._instances:

class PredicateFailed(Exception):
    """Raised when attempting to dequeue an element and the predicate fails."""
    pass


WRITE_FUNCTIONS = ['NtWriteFile', 'WriteFile']
READ_FUNCTIONS = ['NtReadFile', 'ReadFile', 'GetOverlappedResult', 'GetQueuedCompletionStatus', 'NtRemoveIoCompletion', 'CustomCompletionRoutine']

# The following class is a slightly modified version of asyncio.Queue
# which offers the possibility to block the pop of the element if it does not
# satisfy a predicates
class PredicateQueue[T]:
    '''Create a queue object with a given maximum size.

    When getting an element, the "PredicateFailed" exception is raised if the element to be
    poped does not satisfy the "predicate" function.

    If maxsize is <= 0, the queue size is infinite.
    '''

    def __init__(self, maxsize=0):
        self.maxsize = maxsize
        self._init(maxsize)

        # mutex must be held whenever the queue is mutating.  All methods
        # that acquire mutex must release it before returning.  mutex
        # is shared between the three conditions, so acquiring and
        # releasing the conditions also acquires and releases mutex.
        self.mutex = threading.Lock()

        # Notify not_empty whenever an item is added to the queue; a
        # thread waiting to get is notified then.
        self.not_empty = threading.Condition(self.mutex)

        # Notify not_full whenever an item is removed from the queue;
        # a thread waiting to put is notified then.
        self.not_full = threading.Condition(self.mutex)

        # Notify all_tasks_done whenever the number of unfinished tasks
        # drops to zero; thread waiting to join() is notified to resume
        self.all_tasks_done = threading.Condition(self.mutex)
        self.unfinished_tasks = 0

    def task_done(self):
        '''Indicate that a formerly enqueued task is complete.

        Used by Queue consumer threads.  For each get() used to fetch a task,
        a subsequent call to task_done() tells the queue that the processing
        on the task is complete.

        If a join() is currently blocking, it will resume when all items
        have been processed (meaning that a task_done() call was received
        for every item that had been put() into the queue).

        Raises a ValueError if called more times than there were items
        placed in the queue.
        '''
        with self.all_tasks_done:
            unfinished = self.unfinished_tasks - 1
            if unfinished <= 0:
                if unfinished < 0:
                    raise ValueError('task_done() called too many times')
                self.all_tasks_done.notify_all()
            self.unfinished_tasks = unfinished

    def join(self):
        '''Blocks until all items in the Queue have been gotten and processed.

        The count of unfinished tasks goes up whenever an item is added to the
        queue. The count goes down whenever a consumer thread calls task_done()
        to indicate the item was retrieved and all work on it is complete.

        When the count of unfinished tasks drops to zero, join() unblocks.
        '''
        with self.all_tasks_done:
            while self.unfinished_tasks:
                self.all_tasks_done.wait()

    def qsize(self):
        '''Return the approximate size of the queue (not reliable!).'''
        with self.mutex:
            return self._qsize()

    def empty(self):
        '''Return True if the queue is empty, False otherwise (not reliable!).

        This method is likely to be removed at some point.  Use qsize() == 0
        as a direct substitute, but be aware that either approach risks a race
        condition where a queue can grow before the result of empty() or
        qsize() can be used.

        To create code that needs to wait for all queued tasks to be
        completed, the preferred technique is to use the join() method.
        '''
        with self.mutex:
            return not self._qsize()

    def full(self):
        '''Return True if the queue is full, False otherwise (not reliable!).

        This method is likely to be removed at some point.  Use qsize() >= n
        as a direct substitute, but be aware that either approach risks a race
        condition where a queue can shrink before the result of full() or
        qsize() can be used.
        '''
        with self.mutex:
            return 0 < self.maxsize <= self._qsize()

    def put(self, item: T, block=True, timeout=None):
        '''Put an item into the queue.

        If optional args 'block' is true and 'timeout' is None (the default),
        block if necessary until a free slot is available. If 'timeout' is
        a non-negative number, it blocks at most 'timeout' seconds and raises
        the Full exception if no free slot was available within that time.
        Otherwise ('block' is false), put an item on the queue if a free slot
        is immediately available, else raise the Full exception ('timeout'
        is ignored in that case).
        '''
        with self.not_full:
            if self.maxsize > 0:
                if not block:
                    if self._qsize() >= self.maxsize:
                        raise queue.Full
                elif timeout is None:
                    while self._qsize() >= self.maxsize:
                        self.not_full.wait()
                elif timeout < 0:
                    raise ValueError("'timeout' must be a non-negative number")
                else:
                    endtime = time() + timeout
                    while self._qsize() >= self.maxsize:
                        remaining = endtime - time()
                        if remaining <= 0.0:
                            raise queue.Full
                        self.not_full.wait(remaining)
            self._put(item)
            self.unfinished_tasks += 1
            self.not_empty.notify()

    def get(self, block=True, timeout=None, predicate: Callable[[T], bool]=lambda _: True):
        '''Remove and return an item from the queue.

        If optional args 'block' is true and 'timeout' is None (the default),
        block if necessary until an item is available. If 'timeout' is
        a non-negative number, it blocks at most 'timeout' seconds and raises
        the Empty exception if no item was available within that time.
        Otherwise ('block' is false), return an item if one is immediately
        available, else raise the Empty exception ('timeout' is ignored
        in that case).
        '''
        with self.not_empty:
            if not block:
                if not self._qsize():
                    raise queue.Empty
            elif timeout is None:
                while not self._qsize():
                    self.not_empty.wait()
            elif timeout < 0:
                raise ValueError("'timeout' must be a non-negative number")
            else:
                endtime = time() + timeout
                while not self._qsize():
                    remaining = endtime - time()
                    if remaining <= 0.0:
                        raise queue.Empty
                    self.not_empty.wait(remaining)
            item = self._get(predicate)
            self.not_full.notify()
            return item

    def put_nowait(self, item: T):
        '''Put an item into the queue without blocking.

        Only enqueue the item if a free slot is immediately available.
        Otherwise raise the Full exception.
        '''
        return self.put(item, block=False)

    def get_nowait(self, predicate: Callable[[T], bool]):
        '''Remove and return an item from the queue without blocking.

        Only get an item if one is immediately available. Otherwise
        raise the Empty exception.
        '''
        return self.get(block=False, predicate=predicate)

    # Override these methods to implement other queue organizations
    # (e.g. stack or priority queue).
    # These will only be called with appropriate locks held

    # Initialize the queue representation
    def _init(self, maxsize):
        self.queue: collections.deque[T] = collections.deque()

    def _qsize(self):
        return len(self.queue)

    # Put a new item in the queue
    def _put(self, item):
        self.queue.append(item)

    # Get an item from the queue, or raise PredicateFailed
    def _get(self, predicate: Callable[[T], bool]):
        left_el = self.queue[0]
        if not predicate(left_el):
            raise PredicateFailed
        return self.queue.popleft()

class ReadOperationsController(metaclass=SingletonMeta):
    _queues: dict[int, PredicateQueue[ReadOperation]]
    _create_lock: threading.Lock

    def __init__(self) -> None:
        self._queues = {}
        self._create_lock = threading.Lock()

    def ensure_created(self, handle_id: int):
        with self._create_lock:
            if self._queues.get(handle_id) is None:
                self._queues[handle_id] = PredicateQueue()

    def get(self, handle_id: int, max_length: int, wait: bool = True):
        self.ensure_created(handle_id)
        try:
            read_operation = self._queues[handle_id].get(block=wait, predicate=lambda el: len(el.buffer) <= max_length)
        except PredicateFailed:
            return ReadResult(status="BUFFER_TOO_SMALL", data=b"")
        except queue.Empty:
            return ReadResult(status="NO_DATA", data=b"")
        return ReadResult(status="SUCCESS", data=read_operation.buffer)

    def put(self, handle_id: int, item: ReadOperation):
        self.ensure_created(handle_id)
        self._queues[handle_id].put(item)

class WSMessage():
    identifier: str
    function: str
    data: bytes

    def __init__(self) -> None:
        pass

    @staticmethod
    def from_raw(msg: bytes | str):
        if type(msg) == str:
            msg_bytes = msg.encode()
        elif type(msg) == bytes:
            msg_bytes = msg
        else:
            raise RuntimeError(f"Unknown message type: {type(msg)}")

        ws_msg = WSMessage()
        metadata_bytes, data = msg_bytes.split(b"\n", maxsplit=1)
        metadata = json.loads(metadata_bytes)
        ws_msg.identifier = metadata['identifier']
        ws_msg.function = metadata.get('function', 'unknown')
        ws_msg.data = data
        return ws_msg

    def to_bytes(self):
        metadata = json.dumps({"identifier": self.identifier, "function":self.function})
        return metadata.encode() + b'\n' + self.data

class ProxyConnection():
    '''This materialises a connection to a proxy. It is composed of both a websocket client and server.
    Its inner logic ensures both client and server are connected.
    The creation of a ProxyConnection object will create a thread which will handle the client part'''
    path: str
    handle_id: int
    script: Script
    rpc: ScriptExportsSync
    connected: bool
    _connected_condition: threading.Condition
    _ws_client_app: websocket.WebSocketApp
    _ws_client_thread: threading.Thread
    _ws_server_conn: ServerConnection
    _ws_server_thread: threading.Thread

    def __init__(self, ws_server_host: str, ws_server_port: int, path: str, script: Script, http_proxy_host: str, http_proxy_port: int) -> None:
        self.ws_server_host = ws_server_host
        self.path = path
        self.handle_id = int(path.split('/')[-1])
        self.script = script
        self.rpc = script.exports_sync
        self.connected = False
        self._connected_condition = threading.Condition()
        logging.debug(f"Creating connection to 'ws://{ws_server_host}:{ws_server_port}/pipe/{path}' via http://{http_proxy_host}:{http_proxy_port}")
        self._ws_client_app = websocket.WebSocketApp(f"ws://{ws_server_host}:{ws_server_port}/pipe/{path}",
                                                     on_message=lambda ws_client_conn, msg: self._recv_client(ws_client_conn, msg),
                                                     on_open=self._on_open_client,
                                                     )
        self._ws_client_thread = threading.Thread(target=self._run_ws_client, args=(http_proxy_host, http_proxy_port))
        self._ws_client_thread.start()

    def __del__(self):
        logging.debug("Object is about to be deleted, closing connection")
        try:
            self._ws_client_app.close()
            self._ws_server_conn.close()  # type:ignore
        except Exception as e:
            logging.exception(e)

    def send_client(self, msg: bytes) -> None:
        self._ws_client_app.send(msg, websocket.ABNF.OPCODE_BINARY)

    def send_server(self, msg: bytes) -> None:
        asyncio.run(self._ws_server_conn.send(msg))  # type:ignore

    async def _run_recv_server(self):
        try:
            while True:
                msg = await self._ws_server_conn.recv()
                ws_msg = WSMessage.from_raw(msg)
                # This will end up in a WriteFile
                WriteOperationsController().forward_message_from_websocket(handle=self.handle_id, ws_msg=ws_msg, script=self.script, rpc=self.rpc)
        except Exception as e:
            logging.exception(e)
            await self._ws_server_conn.close()  # type:ignore

    def _recv_client(self, ws_client_conn: websocket.WebSocket, message: bytes):
        ws_msg = WSMessage.from_raw(message)
        # This will end up in a ReadFile
        if ws_msg.identifier == 'to_ReadOperations':
            ReadOperationsController().put(self.handle_id, ReadOperation(ws_msg.data))
        # Legacy method: forward without going to ReadOperationsController
        else:
            self.script.post({"type": ws_msg.identifier, "payload": list(ws_msg.data)})

    def _run_ws_client(self, http_proxy_host: str, http_proxy_port: int):
        try:
            logging.debug(f"Connecting using proxy {http_proxy_host}:{http_proxy_port}")
            self._ws_client_app.run_forever(
                    proxy_type='http',
                    http_proxy_host=http_proxy_host,
                    http_proxy_port=http_proxy_port,
                    http_no_proxy=['dummyhost'])
        except Exception as e:
            logging.exception(e)
        self._ws_client_app.close()
        logging.debug("run_ws_client exited")

    def _on_open_client(self, websocket: websocket.WebSocket):
        with self._connected_condition:
            self.connected = True
            self._connected_condition.notify_all()

    def wait_connected(self):
        if not self.connected:
            with self._connected_condition:
                condition_set = self._connected_condition.wait(10)
                if not condition_set:
                    logging.warning("Timeout waiting for client WS to connect")


class WSConnectionRouter(metaclass=SingletonMeta):
    '''This class is a Singleton, it manages all ProxyConnections, ensuring to send messages through the right ProxyConnection.
    It also handles the creation of ProxyConnection on the fly. The first call to the constructor must specify its global variables'''
    script: Script | None = None
    ws_server_host: str = ""
    ws_server_port: int = 0
    http_proxy_host: str = ""
    http_proxy_port: int = 0
    connections: dict[str, ProxyConnection] = {}

    def __init__(self, ws_server_host: str = "", ws_server_port: int = 0, script: Script | None = None, http_proxy_host: str = "", http_proxy_port: int = 0):
        if self.ws_server_host == "":
            if ws_server_host == "":
                raise RuntimeError("You need to initialize 'ws_server_host'")
            self.ws_server_host = ws_server_host
        if self.ws_server_port == 0:
            if ws_server_port == 0:
                raise RuntimeError("You need to initialize 'ws_server_port'")
            self.ws_server_port = ws_server_port
        if self.script is None:
            if script is None:
                raise RuntimeError("You need to initialize 'script'")
            self.script = script
        if self.http_proxy_host == "":
            if http_proxy_host == "":
                raise RuntimeError("You need to initialize 'http_proxy_host'")
            self.http_proxy_host = http_proxy_host
        if self.http_proxy_port == 0:
            if http_proxy_port == 0:
                raise RuntimeError("You need to initialize 'http_proxy_port'")
            self.http_proxy_port = http_proxy_port
        self.connections = {}

    def init_connection(self, path: str):
        connection = self.connections.get(path)
        if connection is not None:
            return connection
        if self.script is None:
            raise RuntimeError("Script is undefined cannot create connection")
        self.connections[path] = ProxyConnection(self.ws_server_host, self.ws_server_port, path, self.script ,self.http_proxy_host, self.http_proxy_port)
        self.connections[path].wait_connected()
        return self.connections[path]

    def send_with_client(self, path: str, msg: bytes):
        connection = self.connections.get(path)
        if connection is None:
            connection = self.init_connection(path)
        connection.send_client(msg)

    def send_with_server(self, path: str, msg: bytes):
        connection = self.connections.get(path)
        if connection is None:
            connection = self.init_connection(path)
        connection.send_server(msg)


def on_message(message: dict[str, str], data: bytes):
    '''What to do when the agent sends a message to the backend.
    It uses WSConnectionRouter to send messages.'''
    message_metadata = message.get('payload')
    message_type = message.get("type")
    if message_type == "error":
        logging.error(message)
        logging.error(message.get("stack"))
        return
    if message_metadata is None:
        logging.error(f"Unknown message: {message}")
        return

    payload = json.loads(message_metadata)

    if payload.get('funcName') == "ThatsNoPipeGetReadData":
        # Trick to "call a python function" from frida
        handle_data_request(payload)
        return

    logging.debug(f"{payload.get('funcName')} {message} => {data}")
    # Read the handleId and handlePath from the JS script
    path = f"{quote_plus(payload.get('handlePath'))}/{payload.get('handleId')}"
    # Create a WSMessage so that we can have metadata
    ws_msg = WSMessage()
    ws_msg.identifier = payload.get('id')
    ws_msg.function = payload.get('funcName', 'unknown')
    ws_msg.data = data
    full_data = ws_msg.to_bytes()

    if payload.get('funcName') in WRITE_FUNCTIONS:
        WriteOperationsController().forward_message_from_pipe(ws_msg=ws_msg, path=path)
    elif payload.get('funcName') in READ_FUNCTIONS:
        WSConnectionRouter().send_with_server(path, full_data)
    else:
        logging.error(f"Unknown function: {payload.get('funcName')}")

async def shutdown(loop: asyncio.AbstractEventLoop):
    tasks = [t for t in asyncio.all_tasks(loop) if t is not asyncio.current_task(loop)]
    for task in tasks:
        task.cancel()
    await asyncio.gather(*tasks, return_exceptions=True)
    loop.stop()

class CLI(cmd.Cmd):

    intro = 'Type help to get list of commands'

    def __init__(self, completekey: str = "tab", stdin: None = None, stdout: None = None, rpc: ScriptExportsSync | None = None) -> None:
        super().__init__(completekey, stdin, stdout)
        self.rpc = rpc
        self._to_snake_case_pattern = re.compile(r'(?<!^)(?=[A-Z])')

    def _to_snake_case(self, name: str) -> str:
        return self._to_snake_case_pattern.sub('_', name).lower()

    def do_lsrpc(self, line: str):

        functions = list(map(self._to_snake_case, js_functions))

    def do_frida(self, line: str):
        """Sends a command using frida"""

        except Exception as error:

            self.do_lsrpc('')

def start_background_loop(loop: asyncio.AbstractEventLoop):

def main():
    args = parse_cmdline()
    logging.basicConfig(level=logging.getLevelName(args.log_level))

    # Read the script
    try:
        with open(args.scriptpath, 'r', encoding="utf-8", newline='\n') as f:
            script_str = f.read()
    except FileNotFoundError:
        logging.error("Script not found, did you correctly build the agent? Did you specify the path to agent.js?")
        return

    # Attack to target process and create the script
    session = frida.attach(args.pid)
    script = session.create_script(script_str)

    # Run the asyncio loop in another python thread
    loop = asyncio.new_event_loop()
    asyncio_loop_thread = threading.Thread(target=start_background_loop, args=(loop,), daemon=True)
    stop_ws = loop.create_future()
    # Start the websocket server untip stop_ws is signaled
    loop.create_task(run_ws_server(stop_ws, args.ws_port))

    # Init ws_server_port, subsequent calls to WSConnectionRouter will use the parameters defined in the first call
    router = WSConnectionRouter(args.ws_host, args.ws_port, script, args.http_proxy_host, args.http_proxy_port)

    # register message handler and load script in session
    script.on('message', on_message)
    script.load()

    # Load RPC from the agent
    rpc = script.exports_sync
    # Initializes the CLI interface
    cli = CLI(rpc=rpc)
    logging.info(f"Filtering on the following regexes: {args.handle_regexes}")
    # Modify the handle regex list on which handles are filtered
    rpc.set_handle_regexes(args.handle_regexes)
    rpc.set_log_syscalls(args.log_syscalls)
    logging.info("[!] Ctrl+C to detach from instrumented program (this may take a while due to hooking of non-busy wait functions).\n\n")
    asyncio_loop_thread.start()
    rpc.install_hooks(not args.high_level_hooking)
    try:
        cli.cmdloop()
    except KeyboardInterrupt:
        logging.info("Received KeyboardInterrupt, exiting")
    session.detach()
    logging.info("Session detached, goodbye...")
    stop_ws.set_result(True)
    loop.stop()
    logging.info(f"Sent stop signal to main loop")
    asyncio_loop_thread.join()
    logging.info("Thread is now stopped")
    loop.run_until_complete(shutdown(loop))
    loop.close()
```

```md
# Format

## Websockets messages

Once a named pipe message is intercepted, a new websocket is open using the format `/pipe/{pipe_name}/{handle_id}`

While the pipe_name is informative, the handle is used internally to route to the correct named pipe handle.

Websocket messages sent by the projet to the HTTP Proxy contains a header and the pipe data.
The header does not contain any `\n` character. Header and data are separated by a single `\n` character.
The header is JSON encoded.

### Header

The keys of the JSON header have the following meaning:

- `identifier`: this is the internal identifier of the operation, it is used by the frida script to know which WinAPI call triggered the message.
- `function`: this is the WinAPI function that was intercepted. This is purely informative.
```

`docs/hooks.md`:

```md
# Hooks

This section describe which cases are handled by the frida script.
We suppose that we always hook the client.

## Synchronous operations

### Client to server

Synchronous communication from client to server is handled by two functions:
- WriteFile
- WriteFileEx

Both functions are hooked

### From server to client

Synchronous communication from server to client can only be handled by two functions:
- ReadFile
- ReadFileEx

Read and modification of the buffer can be applied when the function returns.

Both functions are hooked

## Asynchronous operations

### Client to server

Asynchronous operations from client to server can only be handled by two functions:
- WriteFile
- WriteFileEx

Both functions are hooked, and buffer can be read and modified when the function is called.

### Server to client

Asynchronous communication from server to client begin by a call to one of the two following functions:
- ReadFile
- ReadFileEx
This internally calls NtReadFile.

Functions are called with Overlapped structures which contains an Event, or with a completion routine.
Depending on the client's structure, the notification "data is available" is made by a call to another WinAPI function.
Overlapped and Event objects that are found linked to a target handled are checked in arguments of the following functions. If such an object is found, the buffer is intercepted.
- GetOverlappedResult
- GetOverlappedResultEx
- GetQueuedCompletionStatus
- GetQueuedCompletionStatusEx
- WaitForSingleObject
- WaitForSingleObjectEx

However, calls to ReadFileEx can include an address to a completion routine (which is basically a pointer to a function). These functions are hooked when they are seen.

## Summary

| Function name               | Variant      | Hooking | Interception | Allows modification | Allows injection |
| --------------------------- | ------------ | ------- | ------------ | ------------------- | ---------------- |
| WriteFile                   | -            | Yes     | Yes          | Yes                 | No               |
| WriteFileEx                 | -            | Yes     | Yes          | Yes                 | No               |
| NtWriteFile                 | -            | Yes     | Yes          | Yes                 | Yes              |
| ReadFile                    | Synchronous  | Yes     | Yes          | No                  | No               |
| ReadFileEx                  | Synchronous  | Yes     | Yes          | No                  | No               |
| NtReadFile                  | Synchronous  | Yes     | Yes          | No                  | No               |
| ReadFile                    | Asynchronous | Yes     | Partial      | -                   | No               |
| ReadFileEx                  | Asynchronous | Yes     | Partial      | -                   | No               |
| NtReadFile                  | Asynchronous | Yes     | Yes          | -                   | No               |
```

```go
func ReadWriteCompletionRoutineRoutine(handle windows.Handle, message string, readFirst bool) {
	var readMaxLength = 65536
	readData := make([]byte, readMaxLength)
	var err error
	var event windows.Handle
	var overlapped windows.Overlapped

	completionRoutine := func(dwErrorCode syscall.Errno, dwNumberOfBytesTransfered uint32, lpOverlapped *windows.Overlapped) {
		if dwErrorCode == windows.ERROR_SUCCESS {
			log.Info("Got data (%v bytes): %s", dwNumberOfBytesTransfered, readData[:dwNumberOfBytesTransfered])
		}
	}

	go WriteOverlappedRoutine(handle, message)

	for {
		event, err = windows.CreateEvent(nil, 0, 0, nil)
		if err != nil {
			log.Error("CreateEvent: %s", err)
			return
		}
		overlapped = windows.Overlapped{
			HEvent: event,
		}
		log.Info("About to read")
		err = winlib.ReadFileEx(handle, readData, &overlapped, completionRoutine)
		if err != nil && err != windows.ERROR_IO_PENDING {
			log.Error("ReadFile: %v", err)
			return
		}
		log.Info("Called ReadFile")
		// We need to be in an alertable state, therefore, we need WaitForSingleObjectEx
		// see https://learn.microsoft.com/en-us/windows/win32/fileio/alertable-i-o
		ret, _ := winlib.WaitForSingleObjectEx(overlapped.HEvent, windows.INFINITE, true)
		if ret != windows.WAIT_IO_COMPLETION && ret != 0 {
			log.Error("WaitForSingleObjectEx: %v", ret)
			return
		}
	}
}
```

```go
			log.Error("GetOverlappedResult: %s", err)
			return
		}

	}
```

```go
func CreateAcl() (sa *windows.SecurityAttributes, err error) {

	if err != nil {
		log.Error("SecurityDescriptorFromString")
		return
	}

	var currentProcessToken windows.Token
	err = windows.OpenProcessToken(windows.CurrentProcess(), windows.TOKEN_QUERY, &currentProcessToken)
	if err != nil {
		log.Error("OpenProcessToken")
		return
	}
	defer currentProcessToken.Close()
	tokenUser, err := currentProcessToken.GetTokenUser()
	if err != nil {
		log.Error("GetTokenUser")
		return
	}

	currentUser := windows.TRUSTEE{
		MultipleTrustee:          nil,
		MultipleTrusteeOperation: windows.NO_MULTIPLE_TRUSTEE,
		TrusteeForm:              windows.TRUSTEE_IS_SID,
		TrusteeType:              windows.TRUSTEE_IS_USER,
		TrusteeValue:             windows.TrusteeValueFromSID(tokenUser.User.Sid),
	}

	everyoneSID, err := windows.CreateWellKnownSid(windows.WinWorldSid)
	if err != nil {
		log.Error("CreateWellKnownSid")
		return
	}
	everyoneTrustee := windows.TRUSTEE{
		MultipleTrustee:          nil,
		MultipleTrusteeOperation: windows.NO_MULTIPLE_TRUSTEE,
		TrusteeForm:              windows.TRUSTEE_IS_SID,
		TrusteeType:              windows.TRUSTEE_IS_WELL_KNOWN_GROUP,
		TrusteeValue:             windows.TrusteeValueFromSID(everyoneSID),
	}
	aces := []windows.EXPLICIT_ACCESS{
		{
			AccessPermissions: windows.FILE_GENERIC_READ | windows.FILE_GENERIC_WRITE,
			AccessMode:        windows.GRANT_ACCESS,
			Inheritance:       windows.NO_INHERITANCE,
			Trustee:           currentUser,
		},
		{
			AccessPermissions: windows.FILE_GENERIC_READ | windows.FILE_GENERIC_WRITE,
			AccessMode:        windows.GRANT_ACCESS,
			Inheritance:       windows.NO_INHERITANCE,
			Trustee:           everyoneTrustee,
		},
	}
	originalAcl, _, err := sddl.DACL()
	if err != nil {
		log.Error("GetDACL")
		return
	}

	acl, err := windows.ACLFromEntries(aces, originalAcl)
	if err != nil {
		log.Error("ACLFromEntries")
		return
	}
	newSddl, err := windows.NewSecurityDescriptor()
	if err != nil {
		log.Error("ACLFromEntries")
		return
	}

	err = newSddl.SetDACL(acl, true, false)
	if err != nil {
		log.Error("SetDACL")
		return
	}

	sa = &windows.SecurityAttributes{
		Length:             uint32(unsafe.Sizeof(windows.SecurityAttributes{})),
		SecurityDescriptor: newSddl,
		InheritHandle:      0,
	}
	return
}

func Run(name string, overlapped bool, routine routines.ReadWriteRoutine) {
	pipename := "\\\\.\\pipe\\" + name

	// PIPE_TYPE_MESSAGE and PIPE_READMODE_MESSAGE are non-default
	var pipeMode uint32 = windows.PIPE_TYPE_MESSAGE | windows.PIPE_READMODE_MESSAGE | windows.PIPE_WAIT

	var dwOpenMode uint32 = windows.PIPE_ACCESS_DUPLEX
	if overlapped {
		dwOpenMode = dwOpenMode | windows.FILE_FLAG_OVERLAPPED
	}

	var sa *windows.SecurityAttributes = nil
	var err error
	if config.FirstPipeInstance {
		dwOpenMode = dwOpenMode | windows.FILE_FLAG_FIRST_PIPE_INSTANCE
		sa, err = CreateAcl()
		if err != nil {
			log.Error(err.Error())
			return
		}
	}

	handle, err := winlib.CreateNamedPipe(pipename, dwOpenMode, pipeMode, windows.PIPE_UNLIMITED_INSTANCES, 65536, 65536, 0, sa)
	if err != nil {
		return
	}

	err = windows.ConnectNamedPipe(handle, nil)
	if err != nil {
		log.Error(err.Error())
		windows.CloseHandle(handle)
		return
	}

	routine(handle, "Server says ", true)
	windows.CloseHandle(handle)

}
```

```go
// We define the callback that will be passed to every call to ReadFileEx

static inline void __stdcall c_readfile_callback(uint32_t dwErrorCode, uint32_t dwNumberOfBytesTransfered, uintptr_t lpOverlapped) {

import "C"

type CompletionRoutine func(syscall.Errno, uint32, *windows.Overlapped)

// Copy paste from golang.org/x/sys/windows
var (
	procReadFileEx = modkernel32.NewProc("ReadFileEx")
)

var c_readfile_callback_addr = uintptr(C.get_c_readfile_callback_addr())

// This map is used to known where to route the execution flow once we have the callback
var callbacksByOverlapped = map[uintptr]*CompletionRoutine{}

// Gateway function to link to the correct callback, this is mandatory as CGO creates C stubs to call Go functions from C
//
//export GoReadFileCallback
func GoReadFileCallback(dwErrorCode C.uint32_t, dwNumberOfBytesTransfered C.uint32_t, lpOverlapped C.uintptr_t) {
	// Get the callback from the map, if we fail, then the user did something wrong with its overlapped objects
	routine := *callbacksByOverlapped[uintptr(lpOverlapped)]
	// Deregister it
	delete(callbacksByOverlapped, uintptr(lpOverlapped))
	// Call it with correct parameters
	routine(syscall.Errno(dwErrorCode), uint32(dwNumberOfBytesTransfered), (*windows.Overlapped)(unsafe.Pointer(uintptr(lpOverlapped))))
}

// Copy paste from golang.org/x/sys/windows with small modifications
func readFileEx(handle windows.Handle, buf []byte, overlapped *windows.Overlapped, completionRoutine CompletionRoutine) (err error) {
	var _p0 *byte
	if len(buf) > 0 {
		_p0 = &buf[0]
	}
	var completionRoutineAddr uintptr = 0
	if completionRoutine != nil {
		completionRoutineAddr = c_readfile_callback_addr
		callbacksByOverlapped[uintptr(unsafe.Pointer(overlapped))] = &completionRoutine
	}
	r1, _, e1 := syscall.SyscallN(procReadFileEx.Addr(), uintptr(handle), uintptr(unsafe.Pointer(_p0)), uintptr(len(buf)), uintptr(unsafe.Pointer(overlapped)), completionRoutineAddr)
	if r1 == 0 {
		err = errnoErr(e1)
	}
	return
}
```

```py
#!/usr/bin/env python3
import argparse
import sys
import json

EXAMPLES='''
Examples:
  Extract the data that was sent through the named pipe
  $ ./pipe_parser.py data ./pipe_msg_from_proxy.raw | xxd

  Get all metadata of the named pipe message in JSON format
  $ ./pipe_parser.py header ./pipe_msg_from_proxy.raw | jq .

  Read the message from stdin
  $ cat ./pipe_msg_from_proxy.raw | ./pipe_parser.py data - | xxd
'''


def parse_cmdline():
    parser = argparse.ArgumentParser(epilog=EXAMPLES, formatter_class=argparse.RawDescriptionHelpFormatter)

    parser.add_argument("--output", "-o", type=str, default="-", help="Output file, defaults to stdout if undefined")

    subparsers = parser.add_subparsers(dest="action")

    data_parser = subparsers.add_parser("data", help="Extract data from the pipe message")
    data_parser.add_argument("input", type=str, help="Input filename or '-' for stdout")

    header_parser = subparsers.add_parser("header", help="Extract the metadata in JSON format")
    header_parser.add_argument("input", type=str, help="Input filename or '-' for stdout")

    encode_parser = subparsers.add_parser("encode", help="Encode the message")
    encode_parser.add_argument("header", type=str, help="JSON header to add to the message")
    encode_parser.add_argument("data", type=str, help="Input filename or '-' for stdout")

    return parser.parse_args()


def log_error(msg: str):
    print("[ERROR] " + msg, file=sys.stderr)


def read_data(filename: str) -> bytes:
    """Reads data from filename or stdin if '-' is provided as argument"""
    content = b""
    if filename == "-":
        content = sys.stdin.buffer.read()
    else:
        with open(filename, 'rb') as f:
            content = f.read()
    return content


def write_data(data: bytes, filename: str):
    """Writes data to a filename or stdout if '-' is provided as argument"""
    if filename == "-":
        sys.stdout.buffer.write(data)
    else:
        with open(filename, 'wb') as f:
            f.write(data)


def get_header(content: bytes):
    """Returns the header of the message."""
    return content.split(b"\n", maxsplit=1)[0]


def get_data(content: bytes):
    """Returns the actual data that was sent through the pipe"""
    return content.split(b"\n", maxsplit=1)[1]


def decode(action: str, input_filename: str, output_filename: str):
    """Decodes a websocket message to extract its header or data"""
    full_data = read_data(input_filename)

    if action == "header":
        data = get_header(full_data)
    else:
        data = get_data(full_data)

    write_data(data, output_filename)


def encode(header: str, data_filename: str, output_filename: str):
    """Creates a new websocket message formed with the specified header and content of the specified file"""
    try:
        json_header = json.loads(header)
    except:
        log_error("Header does not seem to be a valid JSON")
        sys.exit(1)
    compressed_header = json.dumps(json_header).encode()
    if b"\n" in compressed_header:
        log_error("Header contains newlines, aborting")
        sys.exit(1)
    
    data = read_data(data_filename)

    write_data(compressed_header + b"\n" + data, output_filename)


def main():
    args = parse_cmdline()
    if args.action == "encode":
        encode(args.header, args.data, args.output)
    else:
        decode(args.action, args.input, args.output)


if __name__ == "__main__":
    main()
```
