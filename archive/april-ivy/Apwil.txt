`README.md`:

```md
# apwil 🐇

windows gamehacking lib in rust, internal & external, by  april

## what's in it

- `process` : open by pid/name, handle hijacking via `NtDuplicateObject`, wow64 check, image path
- `memory` : read/write/slice/put, pointer chains, string + wstring reads, alloc, protect, optional direct syscall path
- `module` : list/find modules (peb walk w/ toolhelp fallback), pe parsing, exports, file version
- `scan` : ida-style sigs, avx2, scan a module or any range
- `hook` : inline (iced-x86), vmt, iat, hardware breakpoints, veh dispatcher
- `render` : d3d11/d3d12/opengl/vulkan probes for swapchain/present hooking
- `math` : Vec2/3/4, Mat4, world-to-screen, fov, lerp etc
- `syscall` : direct syscalls, ssn resolver w/ halo's gate, indirect via syscall/ret gadget
- `unhook` : fresh ntdll over the hooked one
- `input` : key state, mouse/keyboard events, SendInput, cursor
- `window` : find/enum, client rect, foreground, title/class

## features

default has everything turned on. individual features:

```
memory module scan hooks thread-arm math syscall unhook input window
```

## a tiny example

```rust
use std::thread::sleep;
use std::time::Duration;

use apwil::prelude::*;
use apwil::{
    input,
    memory,
    module,
};

const DW_LOCAL_PAWN: usize = 0x2056700;
const DW_ENTLIST: usize = 0x24D0DC0;
const OFF_TEAM: usize = 0x3EB;
const OFF_HEALTH: usize = 0x34C;
const OFF_XHAIR: usize = 0x344C;
const VK_MENU: i32 = 0x12;

fn entity(t: &Target, list: usize, idx: u32) -> Result<usize> {
    let bucket = ((idx & 0x7FFF) >> 9) as isize * 8;
    let slot = (idx & 0x1FF) as isize * 112;
    unsafe { memory::chain(t, list, &[0x10 + bucket, slot, 0]) }
}

fn click() {
    input::mouse_down();
    sleep(Duration::from_millis(10));
    input::mouse_up();
}

fn main() -> Result<()> {
    let t = Target::by_name("cs2.exe")?;
    let client = module::find(&t, "client.dll")?;
    let local_ptr = client.base + DW_LOCAL_PAWN;
    let list_ptr = client.base + DW_ENTLIST;

    loop {
        sleep(Duration::from_millis(2));
        if !input::down(VK_MENU) {
            continue;
        }
        let me: usize = unsafe { memory::read(&t, local_ptr).unwrap_or(0) };
        let list: usize = unsafe { memory::read(&t, list_ptr).unwrap_or(0) };
        if me == 0 || list == 0 {
            continue;
        }
        let my_team: u8 = unsafe { memory::read(&t, me + OFF_TEAM).unwrap_or(0) };
        let id: u32 = unsafe { memory::read(&t, me + OFF_XHAIR).unwrap_or(0) };
        if id == 0 || id == 0xFFFFFFFF {
            continue;
        }
        let target = entity(&t, list, id).unwrap_or(0);
        if target < 0x10000 || target > 0x7FFFFFFFFFFF {
            continue;
        }
        let team: u8 = unsafe { memory::read(&t, target + OFF_TEAM).unwrap_or(0) };
        let hp: i32 = unsafe { memory::read(&t, target + OFF_HEALTH).unwrap_or(0) };
        if (team == 2 || team == 3) && team != my_team && hp > 0 && hp <= 100 {
            click();
            sleep(Duration::from_millis(120));
        }
    }
}
```

use std::sync::{
    Mutex,
    OnceLock,
};

use super::hwbp::{
    Kind,
    Slot,
    set,
};
use super::inline::Inline;
use crate::ffi::*;
use crate::process::Handle;

#[derive(Clone, Copy)]
pub(crate) struct Rec {
    pub addr: usize,
    pub slot: Slot,
    pub kind: Kind,
    pub len: u8,
}

static ARMED: OnceLock<Mutex<Vec<Rec>>> = OnceLock::new();
static HOOK: OnceLock<Mutex<Option<Inline>>> = OnceLock::new();

fn armed() -> &'static Mutex<Vec<Rec>> {
    ARMED.get_or_init(|| Mutex::new(Vec::new()))
}

pub(crate) fn add(addr: usize, slot: Slot, kind: Kind, len: u8) {
    armed().lock().unwrap().push(Rec {
        addr,
        slot,
        kind,
        len,
    });
}

pub(crate) fn remove(addr: usize) {
    let mut g = armed().lock().unwrap();
    if let Some(i) = g.iter().position(|r| r.addr == addr) {
        g.remove(i);
    }
}

pub(crate) fn ensure() {
    let cell = HOOK.get_or_init(|| Mutex::new(None));
    let mut g = cell.lock().unwrap();
    if g.is_some() {
        return;
    }
    unsafe {
        let ntdll: Vec<u16> = "ntdll.dll\0".encode_utf16().collect();
        let m = GetModuleHandleW(ntdll.as_ptr());
        if m.is_null() {
            return;
        }
        let p = match GetProcAddress(m, c"NtCreateThreadEx".as_ptr() as *const u8) {
            Some(f) => f as usize,
            None => return,
        };
        ORIG.set(0).ok();
        if let Ok(h) = Inline::new(p, detour as *const () as usize) {
            ORIG.set(h.orig()).ok();
            *g = Some(h);
        }
    }
}

static ORIG: OnceLock<usize> = OnceLock::new();

type NtCreate = unsafe extern "system" fn(
    *mut HANDLE,
    u32,
    *mut core::ffi::c_void,
    HANDLE,
    *mut core::ffi::c_void,
    *mut core::ffi::c_void,
    u32,
    usize,
    usize,
    usize,
    *mut core::ffi::c_void,
) -> i32;

unsafe extern "system" fn detour(
    handle: *mut HANDLE,
    access: u32,
    attrs: *mut core::ffi::c_void,
    proc: HANDLE,
    start: *mut core::ffi::c_void,
    arg: *mut core::ffi::c_void,
    flags: u32,
    zb: usize,
    ss: usize,
    ms: usize,
    al: *mut core::ffi::c_void,
) -> i32 {
    unsafe {
        let o: NtCreate = core::mem::transmute(*ORIG.get().unwrap_or(&0));
        let status = o(
            handle, access, attrs, proc, start, arg, flags, zb, ss, ms, al,
        );
        if status >= 0 && !handle.is_null() {
            let h = *handle;
            if !h.is_null() && proc == GetCurrentProcess() {
                let g = armed().lock().unwrap();
                if !g.is_empty() {
                    let wrap = Handle::wrap(h);
                    for r in g.iter() {
                        let _ = set(&wrap, r.addr, r.slot, r.kind, r.len, true);
                    }
                    let _ = wrap.into_raw();
                }
            }
        }
        status
    }
}

use super::veh;
use crate::error::{
    Error,
    Result,
};
use crate::ffi::*;

pub type Cb = unsafe extern "system" fn(*mut EXCEPTION_POINTERS) -> i32;

pub struct Guard {
    addr: usize,
    size: usize,
    cb: Cb,
    base: u32,
}

impl Guard {
    pub unsafe fn arm(addr: usize, size: usize, cb: Cb) -> Result<Self> {
        unsafe {
            let mut mbi: MEMORY_BASIC_INFORMATION = core::mem::zeroed();
            if VirtualQuery(
                addr as *const _,
                &mut mbi,
                core::mem::size_of::<MEMORY_BASIC_INFORMATION>(),
            ) == 0
            {
                return Err(Error::last());
            }
            let base = mbi.Protect & !PAGE_GUARD;
            let mut old: u32 = 0;
            if VirtualProtect(addr as *const _, size, base | PAGE_GUARD, &mut old) == 0 {
                return Err(Error::last());
            }
            veh::add(cb);
            Ok(Guard {
                addr,
                size,
                cb,
                base,
            })
        }
    }

    pub fn at(&self) -> usize {
        self.addr
    }

    pub unsafe fn rearm(&self) -> Result<()> {
        unsafe {
            let mut old: u32 = 0;
            if VirtualProtect(
                self.addr as *const _,
                self.size,
                self.base | PAGE_GUARD,
                &mut old,
            ) == 0
            {
                return Err(Error::last());
            }
            Ok(())
        }
    }

    pub fn disarm(&mut self) {
        if self.size == 0 {
            return;
        }
        unsafe {
            veh::remove(self.cb);
            let mut old: u32 = 0;
            VirtualProtect(self.addr as *const _, self.size, self.base, &mut old);
        }
        self.size = 0;
    }
}

impl Drop for Guard {
    fn drop(&mut self) {
        self.disarm();
    }
}

```rs
use std::sync::atomic::{
    AtomicUsize,
    Ordering,
};

use super::veh;
use crate::error::{
    Error,
    Result,
};
use crate::ffi::*;
use crate::process::{
    self,
    Handle,
};

#[derive(Clone, Copy)]
pub enum Slot {
    Dr0 = 0,
    Dr1 = 1,
    Dr2 = 2,
    Dr3 = 3,
}

#[derive(Clone, Copy)]
pub enum Kind {
    Exec,
    Write,
    Read,
    ReadWrite,
}

pub type Cb = unsafe extern "system" fn(*mut EXCEPTION_POINTERS) -> i32;

const FREE: usize = usize::MAX;

static SLOTS: [AtomicUsize; 4] = [
    AtomicUsize::new(FREE),
    AtomicUsize::new(FREE),
    AtomicUsize::new(FREE),
    AtomicUsize::new(FREE),
];

pub struct Hwbp {
    addr: usize,
    slot: Slot,
    cb: Cb,
    armed: bool,
}

impl Hwbp {
    pub fn arm(addr: usize, slot: Slot, kind: Kind, len: u8, cb: Cb) -> Result<Self> {
        let i = slot as usize;
        if SLOTS[i]
            .compare_exchange(FREE, addr, Ordering::AcqRel, Ordering::Acquire)
            .is_err()
        {
            return Err(Error::Bad);
        }
        veh::add(cb);
        #[cfg(feature = "thread-arm")]
        {
            super::auto::ensure();
            super::auto::add(addr, slot, kind, len);
        }
        let pid = unsafe { GetCurrentProcessId() };
        let ts = match process::threads(pid) {
            Ok(v) => v,
            Err(e) => {
                SLOTS[i].store(FREE, Ordering::Release);
                return Err(e);
            }
        };
        for t in ts {
            if t.tid == unsafe { GetCurrentThreadId() } {
                continue;
            }
            unsafe {
                set(
                    &t.open(THREAD_GET_CONTEXT | THREAD_SET_CONTEXT | THREAD_SUSPEND_RESUME)?,
                    addr,
                    slot,
                    kind,
                    len,
                    true,
                )?;
            }
        }
        Ok(Hwbp {
            addr,
            slot,
            cb,
            armed: true,
        })
    }

    pub fn at(&self) -> usize {
        self.addr
    }

    pub fn disarm(&mut self) {
        if !self.armed {
            return;
        }
        self.armed = false;
        unsafe {
            #[cfg(feature = "thread-arm")]
            super::auto::remove(self.addr);
            veh::remove(self.cb);
            let pid = GetCurrentProcessId();
            if let Ok(ts) = crate::process::threads(pid) {
                for t in ts {
                    if t.tid == GetCurrentThreadId() {
                        continue;
                    }
                    if let Ok(h) =
                        t.open(THREAD_GET_CONTEXT | THREAD_SET_CONTEXT | THREAD_SUSPEND_RESUME)
                    {
                        let _ = set(&h, self.addr, self.slot, Kind::Exec, 1, false);
                    }
                }
            }
            SLOTS[self.slot as usize].store(FREE, Ordering::Release);
        }
    }
}

impl Drop for Hwbp {
    fn drop(&mut self) {
        self.disarm();
    }
}

pub(crate) unsafe fn set(
    h: &Handle,
    addr: usize,
    slot: Slot,
    kind: Kind,
    len: u8,
    on: bool,
) -> Result<()> {
    unsafe {
        SuspendThread(h.raw());
        let mut ctx: CONTEXT = core::mem::zeroed();
        ctx.ContextFlags = CONTEXT_DEBUG_REGISTERS_AMD64;
        if GetThreadContext(h.raw(), &mut ctx) == 0 {
            ResumeThread(h.raw());
            return Err(Error::last());
        }
        let i = slot as usize;
        let dr = match slot {
            Slot::Dr0 => &mut ctx.Dr0,
            Slot::Dr1 => &mut ctx.Dr1,
            Slot::Dr2 => &mut ctx.Dr2,
            Slot::Dr3 => &mut ctx.Dr3,
        };
        let mut dr7 = ctx.Dr7;
        if on {
            *dr = addr as u64;
            dr7 |= 1u64 << (i * 2);
            let cond = match kind {
                Kind::Exec => 0u64,
                Kind::Write => 1u64,
                Kind::ReadWrite => 3u64,
                Kind::Read => 3u64,
            };
            let lenc = match len {
                1 => 0u64,
                2 => 1,
                4 => 3,
                8 => 2,
                _ => 0,
            };
            dr7 &= !(0xFu64 << (16 + i * 4));
            dr7 |= (cond | (lenc << 2)) << (16 + i * 4);
        } else {
            *dr = 0;
            dr7 &= !(1u64 << (i * 2));
            dr7 &= !(0xFu64 << (16 + i * 4));
        }
        ctx.Dr7 = dr7;
        if SetThreadContext(h.raw(), &ctx) == 0 {
            ResumeThread(h.raw());
            return Err(Error::last());
        }
        ResumeThread(h.raw());
        Ok(())
    }
}

pub unsafe fn cont(info: *mut EXCEPTION_POINTERS) -> i32 {
    unsafe {
        let c = (*info).ContextRecord;
        (*c).EFlags |= 0x10000;
        (*c).Dr6 = 0;
    }
    -1
}

#[repr(C)]
#[derive(Clone, Copy)]
struct Desc {
    thunk: u32,
    stamp: u32,
    forwarder: u32,
    name: u32,
    iat: u32,

pub struct Iat {
    slot: *mut usize,
    prev: usize,

pub enum Sym<'a> {
    Name(&'a str),
    Ord(u16),
}

impl<'a> From<&'a str> for Sym<'a> {
    fn from(s: &'a str) -> Self {
        Sym::Name(s)
    }
}

impl From<u16> for Sym<'_> {
    fn from(o: u16) -> Self {
        Sym::Ord(o)
    }
}

impl Iat {
    pub unsafe fn swap<'a, S: Into<Sym<'a>>>(
        m: &Module,
        dll: &str,
        sym: S,
        fun: usize,
    ) -> Result<Self> {
        let want = sym.into();
        unsafe {
            let bytes = core::slice::from_raw_parts(m.base as *const u8, m.size);
            let pe = Pe::parse(bytes)?;
            let dir = pe.dirs[1];
            if dir.size == 0 {
                return Err(Error::NotFound);
            }
            let mut p = dir.rva as usize;
            loop {
                if p + core::mem::size_of::<Desc>() > bytes.len() {
                    return Err(Error::NotFound);
                }
                let d: &Desc = &*(bytes.as_ptr().add(p) as *const Desc);
                if d.name == 0 && d.iat == 0 {
                    return Err(Error::NotFound);
                }
                let nameb = &bytes[d.name as usize..];
                let end = nameb.iter().position(|&c| c == 0).unwrap_or(0);
                let dn = core::str::from_utf8(&nameb[..end]).unwrap_or("");
                if dn.eq_ignore_ascii_case(dll) {
                    let mut t = d.thunk as usize;
                    let mut i = d.iat as usize;
                    loop {
                        if t + 8 > bytes.len() {
                            break;
                        }
                        let v = *(bytes.as_ptr().add(t) as *const u64);
                        if v == 0 {
                            break;
                        }
                        let hit = if v & (1u64 << 63) != 0 {
                            match want {
                                Sym::Ord(o) => (v as u16) == o,
                                Sym::Name(_) => false,
                            }
                        } else {
                            match want {
                                Sym::Name(n) => {
                                    let nrva = v as usize + 2;
                                    let nb = &bytes[nrva..];
                                    let e = nb.iter().position(|&c| c == 0).unwrap_or(0);
                                    &nb[..e] == n.as_bytes()
                                }
                                Sym::Ord(_) => false,
                            }
                        };
                        if hit {
                            let slot = (m.base + i) as *mut usize;
                            let mut old: u32 = 0;
                            if VirtualProtect(slot as *const _, 8, PAGE_READWRITE, &mut old) == 0 {
                                return Err(Error::last());
                            }
                            let prev = *slot;
                            *slot = fun;
                            VirtualProtect(slot as *const _, 8, old, &mut old);
                            return Ok(Iat { slot, prev });
                        }
                        t += 8;
                        i += 8;
                    }
                }
                p += core::mem::size_of::<Desc>();
            }
        }
    }

    pub fn at(&self) -> usize {
        self.slot as usize
    }

    pub fn disarm(&mut self) {
        if self.slot.is_null() {
            return;
        }
        unsafe {
            let mut old: u32 = 0;
            if VirtualProtect(self.slot as *const _, 8, PAGE_READWRITE, &mut old) != 0 {
                *self.slot = self.prev;
                VirtualProtect(self.slot as *const _, 8, old, &mut old);
            }
            self.slot = core::ptr::null_mut();
        }
    }
}

impl Drop for Iat {
    fn drop(&mut self) {
        self.disarm();
    }
}

use super::trampoline;
use crate::error::{

use crate::ffi::*;

static INSTALLED: OnceLock<Mutex<HashSet<usize>>> = OnceLock::new();

fn installed() -> &'static Mutex<HashSet<usize>> {
    INSTALLED.get_or_init(|| Mutex::new(HashSet::new()))
}

pub struct Inline {
    target: usize,
    tramp: usize,
    saved: [u8; 32],
    saved_n: usize,
    patch_n: usize,
    abs: bool,
}

impl Inline {
    pub unsafe fn new(target: usize, detour: usize) -> Result<Self> {
        unsafe {
            {
                let mut g = installed().lock().unwrap();
                if !g.insert(target) {
                    return Err(Error::Bad);
                }
            }
            let res = Self::build(target, detour);
            if res.is_err() {
                installed().lock().unwrap().remove(&target);
            }
            res
        }
    }

    unsafe fn build(target: usize, detour: usize) -> Result<Self> {
        unsafe {
            let need = pick(target, detour);
            let mut bytes = [0u8; 32];
            core::ptr::copy_nonoverlapping(target as *const u8, bytes.as_mut_ptr(), 32);
            let mut dec = Decoder::with_ip(64, &bytes, target as u64, DecoderOptions::NONE);
            let mut len = 0usize;
            let mut insns: Vec<Instruction> = Vec::new();
            while len < need.0 && dec.can_decode() {
                let i = dec.decode();
                if i.is_invalid() {
                    return Err(Error::Bad);
                }
                len += i.len();
                insns.push(i);
            }
            if len < need.0 {
                return Err(Error::Bad);
            }

            let lo = target as u64;
            let hi = lo + len as u64;
            for ins in &insns {
                if !safe(ins, lo, hi) {
                    return Err(Error::Bad);
                }
            }

            let tramp_size = insns.len() * 15 + 14;
            let tramp = trampoline::near(target, tramp_size)?;
            let mut enc = Encoder::new(64);
            let mut cur = tramp as u64;
            for ins in &insns {
                let n = enc.encode(ins, cur).map_err(|_| Error::Bad)?;
                cur += n as u64;
            }
            let body = enc.take_buffer();
            if body.len() + 14 > tramp_size {
                trampoline::drop_page(tramp);
                return Err(Error::Bad);
            }
            core::ptr::copy_nonoverlapping(body.as_ptr(), tramp as *mut u8, body.len());
            let back = target + len;
            write_abs_jmp(tramp + body.len(), back);

            let mut saved = [0u8; 32];
            saved[..len].copy_from_slice(&bytes[..len]);

            let mut old: u32 = 0;
            if VirtualProtect(target as *const _, len, PAGE_EXECUTE_READWRITE, &mut old) == 0 {
                trampoline::drop_page(tramp);
                return Err(Error::last());
            }
            if need.1 {
                write_abs_jmp(target, detour);
            } else {
                write_rel_jmp(target, detour);
            }
            let jlen = if need.1 { 14 } else { 5 };
            for i in jlen..len {
                *((target + i) as *mut u8) = 0x90;
            }
            VirtualProtect(target as *const _, len, old, &mut old);
            FlushInstructionCache(GetCurrentProcess(), target as *const _, len);

            Ok(Inline {
                target,
                tramp,
                saved,
                saved_n: len,
                patch_n: need.0,
                abs: need.1,
            })
        }
    }

    pub fn orig(&self) -> usize {
        self.tramp
    }

    pub fn at(&self) -> usize {
        self.target
    }
}

    pub fn disarm(&mut self) {
        if self.saved_n == 0 {
            return;
        }
        unsafe {
            let mut old: u32 = 0;
            if VirtualProtect(
                self.target as *const _,
                self.saved_n,
                PAGE_EXECUTE_READWRITE,
                &mut old,
            ) != 0
            {
                core::ptr::copy_nonoverlapping(
                    self.saved.as_ptr(),
                    self.target as *mut u8,
                    self.saved_n,
                );
                VirtualProtect(self.target as *const _, self.saved_n, old, &mut old);
                FlushInstructionCache(GetCurrentProcess(), self.target as *const _, self.saved_n);
            }
            if self.tramp != 0 {
                trampoline::drop_page(self.tramp);
                self.tramp = 0;
            }
            self.saved_n = 0;
            installed().lock().unwrap().remove(&self.target);
            let _ = self.patch_n;
            let _ = self.abs;
        }
    }
}

impl Drop for Inline {
    fn drop(&mut self) {
        self.disarm();
    }

fn pick(target: usize, detour: usize) -> (usize, bool) {
    let d = (detour as i64) - (target as i64) - 5;
    if d >= i32::MIN as i64 && d <= i32::MAX as i64 {
        (5, false)
    } else {
        (14, true)
    }
}

unsafe fn write_rel_jmp(at: usize, to: usize) {
    unsafe {
        let rel = (to as i64) - (at as i64) - 5;
        *(at as *mut u8) = 0xE9;
        *((at + 1) as *mut i32) = rel as i32;
    }
}

unsafe fn write_abs_jmp(at: usize, to: usize) {
    unsafe {
        let bytes: [u8; 14] = [0xFF, 0x25, 0x00, 0x00, 0x00, 0x00, 0, 0, 0, 0, 0, 0, 0, 0];
        core::ptr::copy_nonoverlapping(bytes.as_ptr(), at as *mut u8, 6);
        *((at + 6) as *mut u64) = to as u64;
    }
}

fn safe(ins: &Instruction, lo: u64, hi: u64) -> bool {
    match ins.flow_control() {
        FlowControl::Next | FlowControl::Return => {}
        FlowControl::UnconditionalBranch | FlowControl::ConditionalBranch | FlowControl::Call => {
            for k in 0..ins.op_count() {
                match ins.op_kind(k) {
                    OpKind::NearBranch16 | OpKind::NearBranch32 | OpKind::NearBranch64 => {
                        let t = ins.near_branch_target();
                        if t >= lo && t < hi {
                            return false;
                        }
                    }
                    OpKind::FarBranch16 | OpKind::FarBranch32 => return false,
                    _ => {}
                }
            }
        }
        FlowControl::IndirectBranch | FlowControl::IndirectCall => {
            if ins.is_ip_rel_memory_operand() {
                let t = ins.ip_rel_memory_address();
                if t >= lo && t < hi {
                    return false;
                }
            }
        }
        FlowControl::Interrupt | FlowControl::XbeginXabortXend | FlowControl::Exception => {
            return false;
        }
    }
    if ins.is_ip_rel_memory_operand() {
        let t = ins.ip_rel_memory_address();
        if t >= lo && t < hi {
            return false;
        }
    }
    true
}

use std::collections::HashMap;
use std::sync::{
    Mutex,
    OnceLock,
};

use crate::error::{
    Error,
    Result,
};
use crate::ffi::*;

const PAGE: usize = 0x1000;
const SLAB: usize = 16;
const COUNT: usize = PAGE / SLAB;

struct Block {
    base: usize,
    used: [bool; COUNT],
}

struct Pool {
    blocks: Vec<Block>,
    sizes: HashMap<usize, usize>,
}

static POOL: OnceLock<Mutex<Pool>> = OnceLock::new();

fn pool() -> &'static Mutex<Pool> {
    POOL.get_or_init(|| {
        Mutex::new(Pool {
            blocks: Vec::new(),
            sizes: HashMap::new(),
        })
    })
}

pub fn near(target: usize, size: usize) -> Result<usize> {
    let slabs = size.div_ceil(SLAB);
    if slabs == 0 || slabs > COUNT {
        return Err(Error::Bad);
    }
    let mut g = pool().lock().unwrap();
    for b in g.blocks.iter_mut() {
        if !within(target, b.base) || !within(target, b.base + PAGE - 1) {
            continue;
        }
        if let Some(off) = take(&mut b.used, slabs) {
            let addr = b.base + off * SLAB;
            g.sizes.insert(addr, slabs);
            return Ok(addr);
        }
    }
    let base = alloc(target)?;
    let mut used = [false; COUNT];
    for k in 0..slabs {
        used[k] = true;
    }
    g.blocks.push(Block { base, used });
    g.sizes.insert(base, slabs);
    Ok(base)
}

pub unsafe fn drop_page(addr: usize) {
    let mut g = pool().lock().unwrap();
    let n = match g.sizes.remove(&addr) {
        Some(v) => v,
        None => return,
    };
    for b in g.blocks.iter_mut() {
        if addr >= b.base && addr < b.base + PAGE {
            let off = (addr - b.base) / SLAB;
            for k in off..off + n {
                if k < COUNT {
                    b.used[k] = false;
                }
            }
            return;
        }
    }
}

fn within(target: usize, addr: usize) -> bool {
    let d = (addr as i128) - (target as i128);
    d >= i32::MIN as i128 && d <= i32::MAX as i128
}

fn take(used: &mut [bool; COUNT], n: usize) -> Option<usize> {
    let mut run = 0usize;
    let mut start = 0usize;
    for i in 0..COUNT {
        if used[i] {
            run = 0;
            start = i + 1;
        } else {
            run += 1;
            if run >= n {
                for k in start..start + n {
                    used[k] = true;
                }
                return Some(start);
            }
        }
    }
    None
}

fn alloc(target: usize) -> Result<usize> {
    unsafe {
        let mut info: SYSTEM_INFO = core::mem::zeroed();
        windows_sys::Win32::System::SystemInformation::GetSystemInfo(&mut info);
        let gran = info.dwAllocationGranularity as usize;
        let lo = target.saturating_sub(0x7FFF_0000);
        let hi = target.saturating_add(0x7FFF_0000);
        let mut a = (target & !(gran - 1)).saturating_sub(gran);
        while a >= lo {
            let mut mbi: MEMORY_BASIC_INFORMATION = core::mem::zeroed();
            if VirtualQuery(
                a as *const _,
                &mut mbi,
                core::mem::size_of::<MEMORY_BASIC_INFORMATION>(),
            ) == 0
            {
                break;
            }
            if mbi.State == MEM_FREE && mbi.RegionSize as usize >= PAGE {
                let p = VirtualAlloc(
                    a as *mut _,
                    PAGE,
                    MEM_COMMIT | MEM_RESERVE,
                    PAGE_EXECUTE_READWRITE,
                );
                if !p.is_null() {
                    return Ok(p as usize);
                }
            }
            if (mbi.BaseAddress as usize) < gran {
                break;
            }
            a = (mbi.BaseAddress as usize).saturating_sub(gran);
        }
        let mut a = (target | (gran - 1)).wrapping_add(1);
        while a <= hi {
            let mut mbi: MEMORY_BASIC_INFORMATION = core::mem::zeroed();
            if VirtualQuery(
                a as *const _,
                &mut mbi,
                core::mem::size_of::<MEMORY_BASIC_INFORMATION>(),
            ) == 0
            {
                break;
            }
            if mbi.State == MEM_FREE && mbi.RegionSize as usize >= PAGE {
                let p = VirtualAlloc(
                    a as *mut _,
                    PAGE,
                    MEM_COMMIT | MEM_RESERVE,
                    PAGE_EXECUTE_READWRITE,
                );
                if !p.is_null() {
                    return Ok(p as usize);
                }
            }
            a = (mbi.BaseAddress as usize).wrapping_add(mbi.RegionSize);
        }
        Err(Error::NotFound)
    }
}

use std::sync::{

use crate::ffi::*;

type Fn = unsafe extern "system" fn(*mut EXCEPTION_POINTERS) -> i32;

static CHAIN: OnceLock<Mutex<Arc<Vec<Fn>>>> = OnceLock::new();
static REG: AtomicBool = AtomicBool::new(false);

fn slot() -> &'static Mutex<Arc<Vec<Fn>>> {
    CHAIN.get_or_init(|| Mutex::new(Arc::new(Vec::new())))
}

pub fn add(f: Fn) {
    let m = slot();
    let mut g = m.lock().unwrap();
    let mut next = (**g).clone();
    next.push(f);
    *g = Arc::new(next);
    drop(g);
    init();
}

pub fn remove(f: Fn) {
    let m = slot();
    let mut g = m.lock().unwrap();
    let mut next = (**g).clone();
    if let Some(i) = next.iter().position(|x| (*x as usize) == (f as usize)) {
        next.remove(i);
        *g = Arc::new(next);
    }
}

fn init() {
    if REG
        .compare_exchange(false, true, Ordering::AcqRel, Ordering::Acquire)
        .is_ok()
    {
        unsafe {
            AddVectoredExceptionHandler(1, Some(dispatch));
        }
    }
}

const EXCEPTION_CONTINUE_SEARCH: i32 = 0;

unsafe extern "system" fn dispatch(info: *mut EXCEPTION_POINTERS) -> i32 {
    let snap = {
        let m = slot();
        let g = m.lock().unwrap();
        g.clone()
    };
    for f in snap.iter() {
        let r = unsafe { f(info) };
        if r != EXCEPTION_CONTINUE_SEARCH {
            return r;
        }
    }
    EXCEPTION_CONTINUE_SEARCH
}

    saved: HashMap<usize, usize>,

impl Vmt {
    pub unsafe fn new(obj: *mut core::ffi::c_void) -> Self {
        unsafe {
            Vmt {
                vtbl: *(obj as *mut *mut usize),
                saved: HashMap::new(),
            }
        }

    pub unsafe fn raw(vtbl: *mut usize) -> Self {
        Vmt {
            vtbl,
            saved: HashMap::new(),
        }
    }

    pub fn at(&self) -> usize {
        self.vtbl as usize
    }

    pub unsafe fn swap(&mut self, idx: usize, fun: usize) -> Result<usize> {
        unsafe {
            let slot = self.vtbl.add(idx);
            let mut old: u32 = 0;
            if VirtualProtect(
                slot as *const _,
                core::mem::size_of::<usize>(),
                PAGE_READWRITE,
                &mut old,
            ) == 0
            {
                return Err(Error::last());
            }
            let ap = &*(slot as *const AtomicPtr<()>);
            let prev = ap.swap(fun as *mut (), Ordering::AcqRel) as usize;
            VirtualProtect(
                slot as *const _,
                core::mem::size_of::<usize>(),
                old,
                &mut old,
            );
            self.saved.entry(idx).or_insert(prev);
            Ok(prev)
        }
    }

    pub unsafe fn restore(&mut self, idx: usize) -> Result<()> {
        unsafe {
            if let Some(prev) = self.saved.remove(&idx) {
                let slot = self.vtbl.add(idx);
                let mut old: u32 = 0;
                if VirtualProtect(
                    slot as *const _,
                    core::mem::size_of::<usize>(),
                    PAGE_READWRITE,
                    &mut old,
                ) == 0
                {
                    return Err(Error::last());
                }
                let ap = &*(slot as *const AtomicPtr<()>);
                ap.store(prev as *mut (), Ordering::Release);
                VirtualProtect(
                    slot as *const _,
                    core::mem::size_of::<usize>(),
                    old,
                    &mut old,
                );
            }
            Ok(())
        }
    }
}

        unsafe {
            let keys: Vec<usize> = self.saved.keys().copied().collect();
            for k in keys {
                let _ = self.restore(k);
            }
        }
    }
}

#[repr(C)]
#[derive(Clone, Copy, Default, PartialEq)]
pub struct Mat4 {
    pub m: [[f32; 4]; 4],
}

impl Mat4 {
    #[inline]
    pub const fn zero() -> Self {
        Self { m: [[0.0; 4]; 4] }
    }

    #[inline]
    pub const fn identity() -> Self {
        let mut m = [[0.0f32; 4]; 4];
        m[0][0] = 1.0;
        m[1][1] = 1.0;
        m[2][2] = 1.0;
        m[3][3] = 1.0;
        Self { m }
    }

    #[inline]
    pub fn from_rows(r: [[f32; 4]; 4]) -> Self {
        Self { m: r }
    }

    #[inline]
    pub fn from_cols(c: [[f32; 4]; 4]) -> Self {
        let mut m = [[0.0f32; 4]; 4];
        for i in 0..4 {
            for j in 0..4 {
                m[i][j] = c[j][i];
            }
        }
        Self { m }
    }

    #[inline]
    pub fn row(&self, i: usize) -> Vec4 {
        Vec4::new(self.m[i][0], self.m[i][1], self.m[i][2], self.m[i][3])
    }

    #[inline]
    pub fn col(&self, j: usize) -> Vec4 {
        Vec4::new(self.m[0][j], self.m[1][j], self.m[2][j], self.m[3][j])
    }

    #[inline]
    pub fn mul_vec4(&self, v: Vec4) -> Vec4 {
        let r = &self.m;
        Vec4::new(
            r[0][0] * v.x + r[0][1] * v.y + r[0][2] * v.z + r[0][3] * v.w,
            r[1][0] * v.x + r[1][1] * v.y + r[1][2] * v.z + r[1][3] * v.w,
            r[2][0] * v.x + r[2][1] * v.y + r[2][2] * v.z + r[2][3] * v.w,
            r[3][0] * v.x + r[3][1] * v.y + r[3][2] * v.z + r[3][3] * v.w,
        )
    }

    #[inline]
    pub fn mul_point(&self, v: Vec3) -> Vec4 {
        self.mul_vec4(Vec4::new(v.x, v.y, v.z, 1.0))
    }

    #[inline]
    pub fn mul(&self, o: &Mat4) -> Mat4 {
        let a = &self.m;
        let b = &o.m;
        let mut r = [[0.0f32; 4]; 4];
        for i in 0..4 {
            for j in 0..4 {
                r[i][j] =
                    a[i][0] * b[0][j] + a[i][1] * b[1][j] + a[i][2] * b[2][j] + a[i][3] * b[3][j];
            }
        }
        Mat4 { m: r }
    }

    #[inline]
    pub fn transpose(&self) -> Mat4 {
        let a = &self.m;
        let mut r = [[0.0f32; 4]; 4];
        for i in 0..4 {
            for j in 0..4 {
                r[i][j] = a[j][i];
            }
        }
        Mat4 { m: r }
    }

#[inline]
pub fn w2s(vp: &Mat4, world: Vec3, w: f32, h: f32) -> Option<Vec2> {
    let c = vp.mul_point(world);
    if c.w <= 0.001 {
        return None;
    }
    let inv = 1.0 / c.w;
    let nx = c.x * inv;
    let ny = c.y * inv;
    Some(Vec2::new(
        (nx * 0.5 + 0.5) * w,
        (1.0 - (ny * 0.5 + 0.5)) * h,
    ))
}

#[inline]
pub fn w2s_fast(vp: &Mat4, world: Vec3, w: f32, h: f32) -> Option<Vec2> {
    let m = &vp.m;
    let cw = m[3][0] * world.x + m[3][1] * world.y + m[3][2] * world.z + m[3][3];
    if cw <= 0.001 {
        return None;
    }
    let cx = m[0][0] * world.x + m[0][1] * world.y + m[0][2] * world.z + m[0][3];
    let cy = m[1][0] * world.x + m[1][1] * world.y + m[1][2] * world.z + m[1][3];
    let inv = 1.0 / cw;
    Some(Vec2::new(
        (cx * inv * 0.5 + 0.5) * w,
        (1.0 - (cy * inv * 0.5 + 0.5)) * h,
    ))
}

#[inline]
pub fn angles(from: Vec3, to: Vec3) -> Vec2 {
    let d = to - from;
    let hyp = (d.x * d.x + d.y * d.y).sqrt();
    let pitch = (-d.z).atan2(hyp).to_degrees();
    let yaw = d.y.atan2(d.x).to_degrees();
    Vec2::new(pitch, yaw)
}

#[inline]
pub fn fwd(pitch: f32, yaw: f32) -> Vec3 {
    let p = pitch.to_radians();
    let y = yaw.to_radians();
    let cp = p.cos();
    Vec3::new(cp * y.cos(), cp * y.sin(), -p.sin())
}

#[inline]
pub fn fov(view: Vec2, target: Vec2) -> f32 {
    let mut d = target - view;
    d.x = ((d.x + 180.0).rem_euclid(360.0)) - 180.0;
    d.y = ((d.y + 180.0).rem_euclid(360.0)) - 180.0;
    (d.x * d.x + d.y * d.y).sqrt()
}

#[inline]
pub fn lerp(a: f32, b: f32, t: f32) -> f32 {
    a + (b - a) * t
}

#[inline]
pub fn smooth(cur: Vec2, target: Vec2, t: f32) -> Vec2 {
    cur + (target - cur) * t

    pub fn new(t: &'a Target, size: usize, prot: u32) -> Result<Self> {
        let addr = unsafe { alloc(t, 0, size, MEM_COMMIT | MEM_RESERVE, prot)? };
        Ok(Self { t, addr, size })
    }

    pub fn at(t: &'a Target, addr: usize, size: usize, prot: u32) -> Result<Self> {
        let a = unsafe { alloc(t, addr, size, MEM_COMMIT | MEM_RESERVE, prot)? };
        Ok(Self { t, addr: a, size })
    }
}

impl Drop for Alloc<'_> {
    fn drop(&mut self) {
        unsafe {
            let _ = free(self.t, self.addr, 0);
        }
    }
}

pub unsafe fn alloc(t: &Target, addr: usize, size: usize, kind: u32, prot: u32) -> Result<usize> {
    unsafe {
        let p = match t {
            Target::Local => VirtualAlloc(addr as *mut _, size, kind, prot),
            Target::Remote(_) => {
                #[cfg(feature = "syscall")]
                if let Some((st, a)) = super::sys::alloc(t.raw(), addr, size, kind, prot) {
                    return if st < 0 { Err(Error::Nt(st)) } else { Ok(a) };
                }
                VirtualAllocEx(t.raw(), addr as *mut _, size, kind, prot)
            }
        };
        if p.is_null() {
            Err(Error::last())
        } else {
            Ok(p as usize)
        }
    }
}

pub unsafe fn free(t: &Target, addr: usize, size: usize) -> Result<()> {
    unsafe {
        let r = match t {
            Target::Local => VirtualFree(addr as *mut _, size, MEM_RELEASE),

                    return if st < 0 { Err(Error::Nt(st)) } else { Ok(()) };
                }
                VirtualFreeEx(t.raw(), addr as *mut _, size, MEM_RELEASE)
            }
        };
        if r == 0 { Err(Error::last()) } else { Ok(()) }
    }

pub unsafe fn slice(t: &Target, addr: usize, buf: &mut [u8]) -> Result<()> {
    match t {
        Target::Local => {
            unsafe {
                core::ptr::copy_nonoverlapping(addr as *const u8, buf.as_mut_ptr(), buf.len());
            }
            Ok(())
        }
        Target::Remote(_) => unsafe {
            #[cfg(feature = "syscall")]
            if let Some(st) = sys::read(t.raw(), addr, buf.as_mut_ptr(), buf.len()) {
                return if st < 0 { Err(Error::Nt(st)) } else { Ok(()) };
            }
            let mut got: usize = 0;
            let r = windows_sys::Win32::System::Diagnostics::Debug::ReadProcessMemory(
                t.raw(),
                addr as *const _,
                buf.as_mut_ptr() as *mut _,
                buf.len(),
                &mut got,
            );
            if r == 0 {
                Err(Error::last())
            } else if got != buf.len() {
                Err(Error::Bounds)
            } else {
                Ok(())
            }
        },
    }

pub unsafe fn chain(t: &Target, base: usize, offs: &[isize]) -> Result<usize> {
    unsafe {
        if offs.is_empty() {
            return Ok(base);
        }
        let mut addr = base;
        let last = offs.len() - 1;
        for (i, o) in offs.iter().enumerate() {
            if i == last {
                return Ok((addr as isize + *o) as usize);
            }
            addr = read::<usize>(t, (addr as isize + *o) as usize)?;
            if addr == 0 {
                return Err(Error::Bounds);
            }
        }
        Ok(addr)
    }
}

pub unsafe fn read_str(t: &Target, addr: usize, cap: usize) -> Result<String> {

        let mut buf = vec![0u8; cap];
        slice(t, addr, &mut buf)?;

        Ok(String::from_utf8_lossy(&buf[..n]).into_owned())

unsafe fn set(t: &Target, addr: usize, size: usize, prot: u32) -> Result<u32> {
    unsafe {
        let mut old: u32 = 0;
        let r = match t {
            Target::Local => VirtualProtect(addr as *const _, size, prot, &mut old),
            Target::Remote(_) => {
                #[cfg(feature = "syscall")]
                if let Some((st, o)) = super::sys::protect(t.raw(), addr, size, prot) {
                    return if st < 0 { Err(Error::Nt(st)) } else { Ok(o) };
                }
                VirtualProtectEx(t.raw(), addr as *const _, size, prot, &mut old)
            }
        };
        if r == 0 { Err(Error::last()) } else { Ok(old) }
    }
}

unsafe extern "system" {
    fn GetFileVersionInfoSizeW(file: *const u16, handle: *mut u32) -> u32;
    fn GetFileVersionInfoW(file: *const u16, handle: u32, len: u32, data: *mut u8) -> i32;
    fn VerQueryValueW(block: *const u8, sub: *const u16, out: *mut *mut u8, len: *mut u32) -> i32;
}

pub fn version_of(path: &str) -> Result<(u16, u16, u16, u16)> {
    unsafe {
        let w: Vec<u16> = path.encode_utf16().chain(core::iter::once(0)).collect();
        let mut h = 0u32;
        let n = GetFileVersionInfoSizeW(w.as_ptr(), &mut h);
        if n == 0 {
            return Err(Error::last());
        }
        let mut buf = vec![0u8; n as usize];
        if GetFileVersionInfoW(w.as_ptr(), 0, n, buf.as_mut_ptr()) == 0 {
            return Err(Error::last());
        }
        let sub: Vec<u16> = "\\\0".encode_utf16().collect();
        let mut out: *mut u8 = core::ptr::null_mut();
        let mut len = 0u32;
        if VerQueryValueW(buf.as_ptr(), sub.as_ptr(), &mut out, &mut len) == 0 || out.is_null() {
            return Err(Error::NotFound);
        }
        let ms = *(out.add(8) as *const u32);
        let ls = *(out.add(12) as *const u32);
        Ok((
            (ms >> 16) as u16,
            (ms & 0xFFFF) as u16,
            (ls >> 16) as u16,
            (ls & 0xFFFF) as u16,
        ))
    }
}

pub fn list(t: &Target) -> Result<Vec<Module>> {
    if matches!(t, Target::Local) {
        return Ok(peb::list());
    }
    if let Ok(v) = peb::remote(t)
        && !v.is_empty()
    {
        return Ok(v);
    }
    unsafe {
        let pid = t.pid()?;
        let snap = CreateToolhelp32Snapshot(TH32CS_SNAPMODULE | TH32CS_SNAPMODULE32, pid);
        if snap == INVALID_HANDLE_VALUE {
            return Err(Error::last());
        }
        let _g = Handle::wrap(snap);
        let mut e: MODULEENTRY32W = core::mem::zeroed();
        e.dwSize = core::mem::size_of::<MODULEENTRY32W>() as u32;
        let mut out = Vec::new();
        if Module32FirstW(snap, &mut e) == 0 {
            return Ok(out);
        }
        loop {
            out.push(Module {
                name: wide(&e.szModule),
                base: e.modBaseAddr as usize,
                size: e.modBaseSize as usize,
            });
            if Module32NextW(snap, &mut e) == 0 {
                break;
            }
        }
        Ok(out)
    }
}

pub fn find(t: &Target, name: &str) -> Result<Module> {
    list(t)?
        .into_iter()
        .find(|m| m.name.eq_ignore_ascii_case(name))
        .ok_or(Error::NotFound)
}

pub fn export(t: &Target, m: &Module, sym: &str) -> Result<usize> {
    let mut buf = vec![0u8; m.size];
    unsafe {
        crate::memory::slice(t, m.base, &mut buf)?;
    }
    let p = pe::Pe::parse(&buf)?;
    let rva = p.export(sym).ok_or(Error::NotFound)?;
    Ok(m.base + rva as usize)
}

impl<'a> Pe<'a> {
    pub fn parse(b: &'a [u8]) -> Result<Self> {
        if b.len() < core::mem::size_of::<DosHdr>() {
            return Err(Error::Parse);
        }
        let dos: &DosHdr = unsafe { &*(b.as_ptr() as *const DosHdr) };
        if dos.magic != DOS_MAGIC {
            return Err(Error::Parse);
        }
        let off = dos.lfanew as usize;
        if off + 4 + core::mem::size_of::<FileHdr>() + core::mem::size_of::<OptHdr64>() > b.len() {
            return Err(Error::Parse);
        }
        let sig = unsafe { *(b.as_ptr().add(off) as *const u32) };
        if sig != NT_SIG {
            return Err(Error::Parse);
        }
        let fh: &FileHdr = unsafe { &*(b.as_ptr().add(off + 4) as *const FileHdr) };
        if fh.opt_size as usize != core::mem::size_of::<OptHdr64>() {
            return Err(Error::Parse);
        }
        let opt_off = off + 4 + core::mem::size_of::<FileHdr>();
        let opt: &OptHdr64 = unsafe { &*(b.as_ptr().add(opt_off) as *const OptHdr64) };
        if opt.magic != 0x20B {
            return Err(Error::Parse);
        }
        let dirs = opt.dirs;
        let sect_off = opt_off + fh.opt_size as usize;
        let sect_n = fh.sections as usize;
        if sect_off + sect_n * core::mem::size_of::<Section>() > b.len() {
            return Err(Error::Parse);
        }
        let sections = unsafe {
            core::slice::from_raw_parts(b.as_ptr().add(sect_off) as *const Section, sect_n)
        };
        let opt_ref: &OptHdr64Ref = unsafe { &*(opt as *const OptHdr64 as *const OptHdr64Ref) };
        Ok(Pe {
            bytes: b,
            opt: opt_ref,
            sections,
            dirs,
        })
    }

    pub fn rva_to_off(&self, rva: u32) -> Option<usize> {
        for s in self.sections {
            let lo = s.virt_addr;
            let hi = lo + s.virt_size.max(s.raw_size);
            if rva >= lo && rva < hi {
                return Some((rva - lo + s.raw_ptr) as usize);
            }
        }
        None
    }

    pub fn at_rva(&self, rva: u32) -> &[u8] {
        &self.bytes[rva as usize..]
    }

    pub fn export(&self, sym: &str) -> Option<u32> {
        let dir = self.dirs[0];
        if dir.size == 0 {
            return None;
        }
        let base = dir.rva as usize;
        if base + core::mem::size_of::<ExportDir>() > self.bytes.len() {
            return None;
        }
        let ed: &ExportDir = unsafe { &*(self.bytes.as_ptr().add(base) as *const ExportDir) };
        let names = ed.name_rva_arr as usize;
        let funcs = ed.func_rva as usize;
        let ords = ed.ord_rva as usize;
        for i in 0..ed.name_n as usize {
            if names + (i + 1) * 4 > self.bytes.len() {
                break;
            }
            let nrva = unsafe { *(self.bytes.as_ptr().add(names + i * 4) as *const u32) } as usize;
            let nbytes = &self.bytes[nrva..];
            let end = nbytes.iter().position(|&c| c == 0).unwrap_or(0);
            if &nbytes[..end] == sym.as_bytes() {
                let ord =
                    unsafe { *(self.bytes.as_ptr().add(ords + i * 2) as *const u16) } as usize;
                if funcs + (ord + 1) * 4 > self.bytes.len() {
                    return None;
                }
                let f = unsafe { *(self.bytes.as_ptr().add(funcs + ord * 4) as *const u32) };
                return Some(f);
            }
        }
        None
    }

    base_pri: usize,

const DLL_BASE: usize = 0x30;
const DLL_SIZE: usize = 0x40;
const NAME_LEN: usize = 0x58;

type NtQuery = unsafe extern "system" fn(HANDLE, u32, *mut core::ffi::c_void, u32, *mut u32) -> i32;

pub fn remote(t: &Target) -> Result<Vec<Module>> {
    unsafe {
        let ntdll: Vec<u16> = "ntdll.dll\0".encode_utf16().collect();
        let m = GetModuleHandleW(ntdll.as_ptr());
        if m.is_null() {
            return Err(Error::NotFound);
        }
        let p = match GetProcAddress(m, c"NtQueryInformationProcess".as_ptr() as *const u8) {
            Some(f) => f,
            None => return Err(Error::NotFound),
        };
        let nq: NtQuery = core::mem::transmute(p);
        let mut pb: ProcBasic = core::mem::zeroed();
        let mut got: u32 = 0;
        let s = nq(
            t.raw(),
            0,
            &mut pb as *mut _ as *mut _,
            core::mem::size_of::<ProcBasic>() as u32,
            &mut got,
        );
        if s < 0 || pb.peb == 0 {
            return Err(Error::NotFound);
        }
        let ldr_ptr: usize = crate::memory::read(t, pb.peb + PEB_LDR)?;
        if ldr_ptr == 0 {
            return Err(Error::NotFound);
        }
        let head = ldr_ptr + LDR_HEAD;
        let first: usize = crate::memory::read(t, head)?;
        let mut cur = first;
        let mut out = Vec::new();
        let mut n = 0;
        while cur != 0 && cur != head && n < 4096 {
            let base: usize = crate::memory::read(t, cur + DLL_BASE)?;
            let size: u32 = crate::memory::read(t, cur + DLL_SIZE)?;
            let name_len: u16 = crate::memory::read(t, cur + NAME_LEN)?;
            let name_buf: usize = crate::memory::read(t, cur + NAME_BUF)?;
            let mut name = String::new();
            if name_buf != 0 && name_len > 0 {
                let mut bytes = vec![0u8; name_len as usize];
                if crate::memory::slice(t, name_buf, &mut bytes).is_ok() {
                    let u: Vec<u16> = bytes
                        .chunks_exact(2)
                        .map(|c| u16::from_le_bytes([c[0], c[1]]))
                        .collect();
                    name = String::from_utf16_lossy(&u);
                }
            }
            if base != 0 {
                out.push(Module {
                    name,
                    base,
                    size: size as usize,
                });
            }
            let next: usize = crate::memory::read(t, cur)?;
            if next == cur {
                break;
            }
            cur = next;
            n += 1;
        }
        Ok(out)
    }

#[repr(C)]
#[derive(Clone, Copy)]
struct Info {
    object: *mut core::ffi::c_void,
    pid: usize,
    handle: usize,
    granted: u32,
    creator: u16,
    attrs: u16,
    kind: u32,

#[link(name = "ntdll")]
unsafe extern "system" {
    fn NtQuerySystemInformation(class: u32, info: *mut u8, len: u32, ret: *mut u32) -> i32;
    fn NtDuplicateObject(
        src_p: *mut core::ffi::c_void,
        src_h: *mut core::ffi::c_void,
        dst_p: *mut core::ffi::c_void,
        dst_h: *mut *mut core::ffi::c_void,
        access: u32,
        attrs: u32,
        opts: u32,
    ) -> i32;
}

const SYSTEM_HANDLE_INFORMATION: u32 = 0x10;
const STATUS_INFO_LENGTH_MISMATCH: i32 = 0xC0000004u32 as i32;
const DUPLICATE_SAME_ACCESS: u32 = 0x2;

pub fn hijack(pid: u32, access: u32) -> Result<Target> {
    unsafe {
        let mut size: u32 = 1 << 20;
        let mut buf: Vec<u8>;
        loop {
            buf = vec![0u8; size as usize];
            let mut ret = 0u32;
            let st = NtQuerySystemInformation(
                SYSTEM_HANDLE_INFORMATION,
                buf.as_mut_ptr(),
                size,
                &mut ret,
            );
            if st == 0 {
                break;
            }
            if st == STATUS_INFO_LENGTH_MISMATCH {
                size = size.saturating_mul(2);
                if size > (1 << 28) {
                    return Err(Error::Nt(st));
                }
                continue;
            }
            return Err(Error::Nt(st));
        }
        let count = *(buf.as_ptr() as *const usize);
        let entries = buf.as_ptr().add(core::mem::size_of::<usize>()) as *const Info;
        let me = windows_sys::Win32::System::Threading::GetCurrentProcess();
        let my_pid = windows_sys::Win32::System::Threading::GetCurrentProcessId() as usize;
        let mut sources: std::collections::HashMap<u32, Handle> = std::collections::HashMap::new();
        let mut dead: std::collections::HashSet<u32> = std::collections::HashSet::new();
        for i in 0..count {
            let e = *entries.add(i);
            let pid_src = e.pid as u32;
            if pid_src as usize == my_pid {
                continue;
            }
            if (e.granted & access) != access {
                continue;
            }
            if dead.contains(&pid_src) {
                continue;
            }
            let src = match sources.get(&pid_src) {
                Some(h) => h.raw(),
                None => {
                    let h = windows_sys::Win32::System::Threading::OpenProcess(
                        0x40,
                        crate::ffi::FALSE,
                        pid_src,
                    );
                    if h.is_null() {
                        dead.insert(pid_src);
                        continue;
                    }
                    sources.insert(pid_src, Handle::wrap(h));
                    h
                }
            };
            let mut dup: *mut core::ffi::c_void = core::ptr::null_mut();
            let st = NtDuplicateObject(
                src,
                e.handle as *mut _,
                me,
                &mut dup,
                0,
                0,
                DUPLICATE_SAME_ACCESS,
            );
            if st != 0 || dup.is_null() {
                continue;
            }
            let dh = Handle::wrap(dup);
            let got = windows_sys::Win32::System::Threading::GetProcessId(dup);
            if got == pid {
                return Ok(Target::Remote(dh));
            }
        }
        let _ = me;
        Err(Error::NotFound)
    }
}

impl Target {
    pub fn hijack(pid: u32) -> Result<Self> {
        hijack(pid, crate::ffi::PROCESS_ALL_ACCESS)
    }

    pub fn hijack_with(pid: u32, access: u32) -> Result<Self> {
        hijack(pid, access)
    }
}

        Self::open(pid)
    }

    pub fn find(name: &str) -> Result<u32> {
        unsafe {
            let snap = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);
            if snap == INVALID_HANDLE_VALUE {
                return Err(Error::last());
            }
            let _g = Handle::wrap(snap);
            let mut e: PROCESSENTRY32W = core::mem::zeroed();
            e.dwSize = core::mem::size_of::<PROCESSENTRY32W>() as u32;
            if Process32FirstW(snap, &mut e) == 0 {
                return Err(Error::last());
            }
            loop {
                let n = wide(&e.szExeFile);
                if n.eq_ignore_ascii_case(name) {
                    return Ok(e.th32ProcessID);
                }
                if Process32NextW(snap, &mut e) == 0 {
                    break;
                }
            }
            Err(Error::NotFound)
        }
    }

    pub fn pid(&self) -> Result<u32> {
        unsafe {
            match self {
                Target::Local => Ok(windows_sys::Win32::System::Threading::GetCurrentProcessId()),
                Target::Remote(h) => {
                    let v = windows_sys::Win32::System::Threading::GetProcessId(h.raw());
                    if v == 0 { Err(Error::last()) } else { Ok(v) }
                }
            }
        }
    }

    pub fn raw(&self) -> HANDLE {
        match self {
            Target::Local => unsafe { windows_sys::Win32::System::Threading::GetCurrentProcess() },
            Target::Remote(h) => h.raw(),
        }
    }

    pub fn local() -> Self {
        Target::Local
    }

    pub fn is_wow64(&self) -> Result<bool> {
        unsafe {
            let mut w: i32 = 0;
            let r = windows_sys::Win32::System::Threading::IsWow64Process(self.raw(), &mut w);
            if r == 0 {
                Err(Error::last())
            } else {
                Ok(w != 0)
            }
        }
    }

    pub fn path(&self) -> Result<String> {
        unsafe {
            let mut buf = [0u16; 1024];
            let mut n = buf.len() as u32;

                0,
                buf.as_mut_ptr(),

            }

        }

}

pub(crate) fn wide(buf: &[u16]) -> String {
    let n = buf.iter().position(|&c| c == 0).unwrap_or(buf.len());

        unsafe {
            let h = self.open(THREAD_SUSPEND_RESUME)?;
            let r = SuspendThread(h.raw());
            if r == u32::MAX {
                Err(Error::last())
            } else {
                Ok(r)
            }
        }
    }

    pub fn resume(&self) -> Result<u32> {
        unsafe {
            let h = self.open(THREAD_SUSPEND_RESUME)?;
            let r = ResumeThread(h.raw());
            if r == u32::MAX {
                Err(Error::last())
            } else {
                Ok(r)
            }
        }
    }

    pub unsafe fn hijack(&self, addr: usize) -> Result<u64> {
        unsafe {
            let h = self.open(THREAD_GET_CONTEXT | THREAD_SET_CONTEXT | THREAD_SUSPEND_RESUME)?;
            SuspendThread(h.raw());
            let mut ctx: CONTEXT = core::mem::zeroed();
            ctx.ContextFlags = CONTEXT_CONTROL_AMD64;
            if GetThreadContext(h.raw(), &mut ctx) == 0 {
                ResumeThread(h.raw());
                return Err(Error::last());
            }
            let prev = ctx.Rip;
            ctx.Rip = addr as u64;
            if SetThreadContext(h.raw(), &ctx) == 0 {
                ResumeThread(h.raw());
                return Err(Error::last());
            }
            ResumeThread(h.raw());
            Ok(prev)
        }
    }

    pub unsafe fn restore(&self, rip: u64) -> Result<()> {
        unsafe {
            let h = self.open(THREAD_GET_CONTEXT | THREAD_SET_CONTEXT | THREAD_SUSPEND_RESUME)?;
            SuspendThread(h.raw());
            let mut ctx: CONTEXT = core::mem::zeroed();
            ctx.ContextFlags = CONTEXT_CONTROL_AMD64;
            if GetThreadContext(h.raw(), &mut ctx) == 0 {
                ResumeThread(h.raw());
                return Err(Error::last());
            }
            ctx.Rip = rip;
            if SetThreadContext(h.raw(), &ctx) == 0 {
                ResumeThread(h.raw());
                return Err(Error::last());
            }
            ResumeThread(h.raw());
            Ok(())
        }
    }

    unsafe {
        let snap = CreateToolhelp32Snapshot(TH32CS_SNAPTHREAD, 0);
        if snap == INVALID_HANDLE_VALUE {
            return Err(Error::last());
        }
        let _g = Handle::wrap(snap);
        let mut e: THREADENTRY32 = core::mem::zeroed();
        e.dwSize = core::mem::size_of::<THREADENTRY32>() as u32;
        let mut out = Vec::new();

            return Ok(out);
        }
        loop {
            if e.th32OwnerProcessID == pid {
                out.push(Thread {
                    tid: e.th32ThreadID,
                    pid,
                });
            }
            if Thread32Next(snap, &mut e) == 0 {
                break;
            }
        }

pub unsafe fn vtable(obj: *mut core::ffi::c_void) -> Vec<usize> {
    unsafe {
        if obj.is_null() {
            return Vec::new();
        }
        let vt = *(obj as *mut *mut usize);
        let mut out = Vec::new();
        let mut i = 0isize;
        while i < 1024 {
            let p = *vt.offset(i);
            if p == 0 {
                break;
            }
            out.push(p);
            i += 1;
        }
        out
    }

    unsafe {
        if obj.is_null() {
            return;
        }
        type Rel = unsafe extern "system" fn(*mut core::ffi::c_void) -> u32;
        let vt = *(obj as *mut *mut usize);
        let rel: Rel = core::mem::transmute(*vt.offset(2));

    pub ptr: *mut usize,
    pub entries: Vec<usize>,
}

    pub unsafe fn of(obj: *mut core::ffi::c_void) -> Self {
        unsafe {
            let ptr = if obj.is_null() {
                core::ptr::null_mut()
            } else {
                *(obj as *mut *mut usize)
            };
            Vt {
                ptr,
                entries: vtable(obj),
            }
        }
    }

    pub fn at(&self, idx: usize) -> Option<usize> {
        self.entries.get(idx).copied()
    }

    pub fn len(&self) -> usize {
        self.entries.len()
    }

    pub unsafe fn hook(&self, idx: usize, fun: usize) -> Result<Vmt> {
        unsafe {
            if self.ptr.is_null() || idx >= self.entries.len() {
                return Err(Error::Bounds);
            }
            let mut v = Vmt::raw(self.ptr);
            v.swap(idx, fun)?;
            Ok(v)
        }
    }
}

use windows_sys::Win32::Foundation::HWND;
use windows_sys::Win32::System::LibraryLoader::GetModuleHandleW;
use windows_sys::Win32::System::Threading::GetCurrentProcessId;

use crate::error::{
    Error,
    Result,

pub struct Win {
    pub hwnd: HWND,
    class: Vec<u16>,
}

impl Win {
    pub fn new() -> Result<Self> {
        unsafe {
            let pid = GetCurrentProcessId();
            let tick = std::time::SystemTime::now()
                .duration_since(std::time::UNIX_EPOCH)
                .map(|d| d.subsec_nanos())
                .unwrap_or(0);
            let class: Vec<u16> = format!("c{:08x}{:08x}\0", pid, tick)
                .encode_utf16()
                .collect();
            let name: Vec<u16> = format!("w{:08x}{:08x}\0", pid, tick)
                .encode_utf16()
                .collect();
            let h = GetModuleHandleW(core::ptr::null());
            let wc = WNDCLASSEXW {
                cbSize: core::mem::size_of::<WNDCLASSEXW>() as u32,
                style: 0,
                lpfnWndProc: Some(DefWindowProcW),
                cbClsExtra: 0,
                cbWndExtra: 0,
                hInstance: h as _,
                hIcon: core::ptr::null_mut(),
                hCursor: core::ptr::null_mut(),
                hbrBackground: core::ptr::null_mut(),
                lpszMenuName: core::ptr::null(),
                lpszClassName: class.as_ptr(),
                hIconSm: core::ptr::null_mut(),
            };
            RegisterClassExW(&wc);
            let hwnd = CreateWindowExW(
                0,
                class.as_ptr(),
                name.as_ptr(),
                WS_OVERLAPPEDWINDOW,
                0,
                0,
                100,
                100,
                core::ptr::null_mut(),
                core::ptr::null_mut(),
                h as _,
                core::ptr::null(),
            );
            if hwnd.is_null() {
                UnregisterClassW(class.as_ptr(), h as _);
                return Err(Error::last());
            }
            Ok(Win { hwnd, class })
        }
    }
}

            DestroyWindow(self.hwnd);
            let h = GetModuleHandleW(core::ptr::null());

        }

mod pattern;

pub use pattern::Pattern;
pub use search::{
    all,
    find as scan,
};

use crate::error::Result;
use crate::module::Module;
use crate::process::Target;

pub fn module(t: &Target, m: &Module, p: &Pattern) -> Result<Option<usize>> {
    if matches!(t, Target::Local) {
        let buf = unsafe { core::slice::from_raw_parts(m.base as *const u8, m.size) };
        return Ok(search::find(buf, p).map(|o| m.base + o));
    }
    let mut buf = vec![0u8; m.size];
    unsafe {
        crate::memory::slice(t, m.base, &mut buf)?;
    }
    Ok(search::find(&buf, p).map(|o| m.base + o))
}

pub fn module_all(t: &Target, m: &Module, p: &Pattern) -> Result<Vec<usize>> {
    if matches!(t, Target::Local) {
        let buf = unsafe { core::slice::from_raw_parts(m.base as *const u8, m.size) };
        return Ok(search::all(buf, p)
            .into_iter()
            .map(|o| m.base + o)
            .collect());
    }
    let mut buf = vec![0u8; m.size];
    unsafe {
        crate::memory::slice(t, m.base, &mut buf)?;
    }
    Ok(search::all(&buf, p)
        .into_iter()
        .map(|o| m.base + o)
        .collect())
}

pub fn range(t: &Target, base: usize, size: usize, p: &Pattern) -> Result<Option<usize>> {
    if matches!(t, Target::Local) {
        let buf = unsafe { core::slice::from_raw_parts(base as *const u8, size) };
        return Ok(search::find(buf, p).map(|o| base + o));
    }
    let mut buf = vec![0u8; size];
    unsafe {
        crate::memory::slice(t, base, &mut buf)?;
    }
    Ok(search::find(&buf, p).map(|o| base + o))
}

pub fn range_all(t: &Target, base: usize, size: usize, p: &Pattern) -> Result<Vec<usize>> {
    if matches!(t, Target::Local) {
        let buf = unsafe { core::slice::from_raw_parts(base as *const u8, size) };
        return Ok(search::all(buf, p).into_iter().map(|o| base + o).collect());
    }
    let mut buf = vec![0u8; size];
    unsafe {
        crate::memory::slice(t, base, &mut buf)?;
    }
    Ok(search::all(&buf, p).into_iter().map(|o| base + o).collect())
}

    pub bytes: Vec<u8>,
    pub mask: Vec<bool>,

impl Pattern {
    pub fn ida(s: &str) -> Result<Self> {
        let mut bytes = Vec::new();
        let mut mask = Vec::new();
        for tok in s.split_ascii_whitespace() {
            if tok == "?" || tok == "??" {
                bytes.push(0);
                mask.push(false);
            } else if tok.len() == 2 {
                let b = u8::from_str_radix(tok, 16).map_err(|_| Error::Parse)?;
                bytes.push(b);
                mask.push(true);
            } else {
                return Err(Error::Parse);
            }
        }
        if bytes.is_empty() {
            return Err(Error::Parse);
        }
        Ok(Self { bytes, mask })
    }

    pub fn len(&self) -> usize {
        self.bytes.len()
    }
}

use super::Pattern;

pub fn find(hay: &[u8], p: &Pattern) -> Option<usize> {
    let plen = p.bytes.len();
    if plen == 0 || plen > hay.len() {
        return None;
    }
    let anchor = anchor_of(p)?;
    let byte = p.bytes[anchor];
    let end = hay.len() - plen;
    let stop = end + anchor;
    let mut i = anchor;
    while i <= stop {
        let d = scan(&hay[i..=stop], byte)?;
        let pos = i + d;
        let start = pos - anchor;
        if matches(&hay[start..start + plen], p) {
            return Some(start);
        }
        i = pos + 1;
    }
    None
}

pub fn all(hay: &[u8], p: &Pattern) -> Vec<usize> {
    let mut out = Vec::new();
    let plen = p.bytes.len();
    if plen == 0 || plen > hay.len() {
        return out;
    }
    let anchor = match anchor_of(p) {
        Some(a) => a,
        None => return out,
    };
    let byte = p.bytes[anchor];
    let end = hay.len() - plen;
    let stop = end + anchor;
    let mut i = anchor;
    while i <= stop {
        match scan(&hay[i..=stop], byte) {
            Some(d) => {
                let pos = i + d;
                let start = pos - anchor;
                if matches(&hay[start..start + plen], p) {
                    out.push(start);
                }
                i = pos + 1;
            }
            None => break,
        }
    }
    out
}

fn anchor_of(p: &Pattern) -> Option<usize> {
    let mut best: Option<(usize, u16)> = None;
    for (i, &m) in p.mask.iter().enumerate() {
        if !m {
            continue;
        }
        let f = FREQ[p.bytes[i] as usize];
        match best {
            Some((_, bf)) if bf <= f => {}
            _ => best = Some((i, f)),
        }
    }
    best.map(|(i, _)| i)
}

fn matches(window: &[u8], p: &Pattern) -> bool {
    for j in 0..p.bytes.len() {
        if p.mask[j] && window[j] != p.bytes[j] {
            return false;
        }
    }
    true
}

fn scan(buf: &[u8], b: u8) -> Option<usize> {
    if std::is_x86_feature_detected!("avx2") {
        unsafe { scan_avx2(buf, b) }
    } else {
        unsafe { scan_sse2(buf, b) }
    }
}

#[cfg(target_arch = "x86_64")]
#[target_feature(enable = "avx2")]
unsafe fn scan_avx2(buf: &[u8], b: u8) -> Option<usize> {
    use core::arch::x86_64::*;
    unsafe {
        let needle = _mm256_set1_epi8(b as i8);
        let mut i = 0usize;
        while i + 32 <= buf.len() {
            let chunk = _mm256_loadu_si256(buf.as_ptr().add(i) as *const __m256i);
            let cmp = _mm256_cmpeq_epi8(chunk, needle);
            let m = _mm256_movemask_epi8(cmp) as u32;
            if m != 0 {
                return Some(i + m.trailing_zeros() as usize);
            }
            i += 32;
        }
        let nx = _mm_set1_epi8(b as i8);
        while i + 16 <= buf.len() {
            let chunk = _mm_loadu_si128(buf.as_ptr().add(i) as *const __m128i);
            let cmp = _mm_cmpeq_epi8(chunk, nx);
            let m = _mm_movemask_epi8(cmp) as u32;
            if m != 0 {
                return Some(i + m.trailing_zeros() as usize);
            }
            i += 16;
        }
        while i < buf.len() {
            if buf[i] == b {
                return Some(i);
            }
            i += 1;
        }
    }
    None
}

pub fn ntdll() -> Result<(usize, usize)> {
    unsafe {
        let w: Vec<u16> = "ntdll.dll\0".encode_utf16().collect();
        let m = GetModuleHandleW(w.as_ptr());
        if m.is_null() {
            return Err(Error::NotFound);
        }
        let base = m as usize;
        let bytes = core::slice::from_raw_parts(base as *const u8, 0x1000);
        let dos = bytes.as_ptr() as *const u16;
        if *dos != 0x5A4D {
            return Err(Error::Parse);
        }
        let lfanew = *(bytes.as_ptr().add(0x3C) as *const u32) as usize;
        let size = *(bytes.as_ptr().add(lfanew + 0x50) as *const u32) as usize;
        Ok((base, size))
    }
}

pub fn ssn(name: &str) -> Option<u16> {
    unsafe {
        let (base, size) = ntdll().ok()?;
        let img = core::slice::from_raw_parts(base as *const u8, size);
        let pe = Pe::parse(img).ok()?;
        let rva = pe.export(name)?;
        let stub = (base + rva as usize) as *const u8;
        if *stub == 0x4C && *stub.add(1) == 0x8B && *stub.add(2) == 0xD1 && *stub.add(3) == 0xB8 {
            return Some(*(stub.add(4) as *const u32) as u16);
        }
        halo(base, &pe, rva)
    }
}

unsafe fn halo(base: usize, pe: &Pe, rva: u32) -> Option<u16> {
    unsafe {
        for d in 1..=64i32 {
            for sign in [-1i32, 1] {
                let off = rva as i64 + (d * sign * 32) as i64;
                if off < 0 {
                    continue;
                }
                let p = (base as i64 + off) as *const u8;
                if *p == 0x4C && *p.add(1) == 0x8B && *p.add(2) == 0xD1 && *p.add(3) == 0xB8 {
                    let neighbor = *(p.add(4) as *const u32) as i32;
                    let me = neighbor - (d * sign);
                    if me >= 0 {
                        return Some(me as u16);
                    }
                }
            }
        }
        let _ = pe;
        None
    }
}

pub fn gadget() -> Option<usize> {
    unsafe {
        let (base, size) = ntdll().ok()?;
        let bytes = core::slice::from_raw_parts(base as *const u8, size);
        let mut i = 0;
        while i + 3 < bytes.len() {
            if bytes[i] == 0x0F && bytes[i + 1] == 0x05 && bytes[i + 2] == 0xC3 {
                return Some(base + i);
            }
            i += 1;
        }
        None
    }
}

use crate::ffi::*;

pub fn ntdll() -> Result<()> {
    unsafe {
        let w: Vec<u16> = "ntdll.dll\0".encode_utf16().collect();
        let m = GetModuleHandleW(w.as_ptr());
        if m.is_null() {
            return Err(Error::NotFound);
        }
        let base = m as usize;
        let img = core::slice::from_raw_parts(base as *const u8, 0x1000);
        let lfanew = *(img.as_ptr().add(0x3C) as *const u32) as usize;
        let img_size = *(img.as_ptr().add(lfanew + 0x50) as *const u32) as usize;
        let live = core::slice::from_raw_parts(base as *const u8, img_size);
        let pe_live = Pe::parse(live)?;

        let path: Vec<u16> = r"\\.\GLOBALROOT\KnownDlls\ntdll.dll"
            .encode_utf16()
            .chain(core::iter::once(0))
            .collect();
        let h = open_file(&path)?;
        let _g = crate::process::Handle::wrap(h);
        let mut clean = Vec::new();
        let mut buf = vec![0u8; 0x10000];
        loop {
            let mut got: u32 = 0;
            let ok = ReadFile(
                h,
                buf.as_mut_ptr(),
                buf.len() as u32,
                &mut got,
                core::ptr::null_mut(),
            );
            if ok == 0 || got == 0 {
                break;
            }
            clean.extend_from_slice(&buf[..got as usize]);
        }
        if clean.is_empty() {
            return Err(Error::NotFound);
        }
        let pe_clean = Pe::parse(&clean)?;

        for sec in pe_live.sections {
            let name = &sec.name;
            if name != b".text\0\0\0" {
                continue;
            }
            let dst = base + sec.virt_addr as usize;
            let len = sec.virt_size as usize;
            let src_off = pe_clean.rva_to_off(sec.virt_addr).ok_or(Error::Parse)?;
            if src_off + len > clean.len() {
                return Err(Error::Bounds);
            }
            let mut old: u32 = 0;
            if VirtualProtect(dst as *const _, len, PAGE_EXECUTE_READWRITE, &mut old) == 0 {
                return Err(Error::last());
            }
            core::ptr::copy_nonoverlapping(clean.as_ptr().add(src_off), dst as *mut u8, len);
            VirtualProtect(dst as *const _, len, old, &mut old);
        }
        Ok(())
    }
}

            3,
            0,
            core::ptr::null_mut(),
        );
        if h.is_null() || h == INVALID_HANDLE_VALUE {
            return Err(Error::last());
        }
        Ok(h)
    }
}
