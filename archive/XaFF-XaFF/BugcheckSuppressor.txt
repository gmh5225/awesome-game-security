`BugcheckSuppressor/cet_stubs.asm`:

```asm
; ============================================================================
; cet_restore.asm
;
; Asm stubs for HVCI/kCET-aware bugcheck suppressor.
;
; Provides:
;   - HvciReadSsp:        read shadow stack pointer (rdsspq)
;   - HvciIncSspN:        pop N entries from shadow stack (incsspq rcx)
;   - HvciKcetJmpRestoreFixed: kCET-aware register restore + indirect JMP
;                              (caller-provided shadow-stack pop count)
;   - HvciJmpRestore:          non-kCET register restore + indirect JMP
;
; All restore functions:
;   - Restore non-volatile XMM6-XMM15 from CONTEXT
;   - Restore non-volatile GPRs (RBX, RBP, RSI, RDI, R12-R15)
;   - Restore volatile registers (RDX, R8, R9, R10, R11)   <-- R10/R11 added
;   - Set RAX to override value (e.g., STATUS_UNSUCCESSFUL)
;   - Switch RSP to target stack
;   - STI (re-enable interrupts)
;   - Indirect JMP to TargetRip via [rsp-8] memory operand
;
; Why memory-indirect JMP:
;   The previous version used "jmp r10" with TargetRip preloaded into r10,
;   which meant R10 (and R11 used for the stack switch) leaked the kernel
;   target/RSP value into the resumed bad-driver context. Volatile or not,
;   in the skip-instruction path the bad driver continues mid-function and
;   may have live data in R10/R11 that the trap captured into CONTEXT.
;   By parking TargetRip at [rsp-8] *after* the stack switch we free both
;   R10 and R11 to be restored from CONTEXT before the JMP.
;
; Why [rsp-8] is safe:
;   - We are inside an interrupts-disabled window (`_disable()` in C, then
;     STI shadow on the JMP itself), so no maskable interrupt or scheduler
;     event can use that slot before the JMP fetches it.
;   - NMI on Windows kernel is IST-routed: NMI delivery switches RSP to a
;     dedicated per-CPU IST stack, so it does not write to our [rsp-8].
;   - After the JMP completes, RSP has not moved, so the bad-driver's view
;     of its own stack is unchanged. The slot at [rsp-8] returns to being
;     "below RSP scratch" which is normal kernel-stack convention - the
;     next interrupt/CALL on this CPU may overwrite it freely.
;
; The kCET version additionally:
;   - Pops the dispatcher chain + supervisor token (+ optionally the bad
;     driver's RA) from the shadow stack via INCSSPQ in <=255 chunks.
; ============================================================================

.CODE

; ----------------------------------------------------------------------------
; ULONG64 HvciReadSsp(VOID);
;
; Returns: current shadow stack pointer (0 if kCET not active)
; ----------------------------------------------------------------------------
HvciReadSsp PROC
    xor rax, rax
    db  0F3h, 048h, 00Fh, 01Eh, 0C8h    ; rdsspq rax
    ret
HvciReadSsp ENDP


; ----------------------------------------------------------------------------
; VOID HvciIncSspN(ULONG count);
;
; Pops 'count' entries from the shadow stack via incsspq.
; Count must be 1-255 per call (INCSSPQ encoding limit).
; ----------------------------------------------------------------------------
HvciIncSspN PROC
    db  0F3h, 048h, 00Fh, 0AEh, 0E9h    ; incsspq rcx
    ret
HvciIncSspN ENDP


; ----------------------------------------------------------------------------
; VOID HvciKcetJmpRestoreFixed(
;     PCONTEXT Context,       ; RCX
;     ULONG64  TargetRip,     ; RDX
;     ULONG64  TargetRsp,     ; R8
;     ULONG64  TargetRax,     ; R9
;     ULONG    SspPopCount)   ; [rsp+28h]
;
; Same restore semantics as the older variant but with the SSP pop count
; pre-computed by the caller.
;
; Never returns.
; ----------------------------------------------------------------------------
HvciKcetJmpRestoreFixed PROC
    push    rdx                             ; [rsp+18h] = TargetRip
    push    r8                              ; [rsp+10h] = TargetRsp
    push    r9                              ; [rsp+08h] = TargetRax
    push    rcx                             ; [rsp+00h] = Context

    mov     r8, [rsp + 48h]                 ; r8 = SspPopCount
    test    r8, r8
    jz      fixed_pop_done
fixed_pop_loop:
    mov     rcx, r8
    cmp     rcx, 255
    jbe     fixed_do_pop
    mov     rcx, 255
fixed_do_pop:
    db      0F3h, 048h, 00Fh, 0AEh, 0E9h    ; incsspq rcx
    sub     r8, rcx
    jnz     fixed_pop_loop
fixed_pop_done:

    mov     rcx, [rsp + 00h]                ; rcx = Context

    movdqu  xmm6,  [rcx + 200h]
    movdqu  xmm7,  [rcx + 210h]
    movdqu  xmm8,  [rcx + 220h]
    movdqu  xmm9,  [rcx + 230h]
    movdqu  xmm10, [rcx + 240h]
    movdqu  xmm11, [rcx + 250h]
    movdqu  xmm12, [rcx + 260h]
    movdqu  xmm13, [rcx + 270h]
    movdqu  xmm14, [rcx + 280h]
    movdqu  xmm15, [rcx + 290h]
    ldmxcsr dword ptr [rcx + 034h]

    mov     rbx, [rcx + 090h]
    mov     rbp, [rcx + 0A0h]
    mov     rsi, [rcx + 0A8h]
    mov     rdi, [rcx + 0B0h]
    mov     r12, [rcx + 0D8h]
    mov     r13, [rcx + 0E0h]
    mov     r14, [rcx + 0E8h]
    mov     r15, [rcx + 0F0h]
    mov     rdx, [rcx + 088h]
    mov     r8,  [rcx + 0B8h]
    mov     r9,  [rcx + 0C0h]

    mov     r10, [rsp + 18h]                ; TargetRip
    mov     r11, [rsp + 10h]                ; TargetRsp
    mov     rax, [rsp + 08h]                ; TargetRax

    mov     rsp, r11
    mov     qword ptr [rsp - 8], r10        ; park TargetRip

    mov     r10, [rcx + 0C8h]               ; restore R10 from CONTEXT
    mov     r11, [rcx + 0D0h]               ; restore R11 from CONTEXT
    mov     rcx, [rcx + 080h]               ; restore RCX from CONTEXT

    sti
    jmp     qword ptr [rsp - 8]
HvciKcetJmpRestoreFixed ENDP


; ----------------------------------------------------------------------------
; VOID HvciJmpRestore(
;     PCONTEXT Context,       ; RCX
;     ULONG64  TargetRip,     ; RDX
;     ULONG64  TargetRsp,     ; R8
;     ULONG64  TargetRax)     ; R9
;
; Non-kCET version. Same as kCET but without shadow stack walk.
; Never returns.
; ----------------------------------------------------------------------------
HvciJmpRestore PROC
    push    rdx                             ; [rsp+18h] = TargetRip
    push    r8                              ; [rsp+10h] = TargetRsp
    push    r9                              ; [rsp+08h] = TargetRax

    movdqu  xmm7,  [rcx + 210h]
    movdqu  xmm8,  [rcx + 220h]
    movdqu  xmm9,  [rcx + 230h]
    movdqu  xmm10, [rcx + 240h]
    movdqu  xmm11, [rcx + 250h]
    movdqu  xmm12, [rcx + 260h]
    movdqu  xmm13, [rcx + 270h]
    movdqu  xmm14, [rcx + 280h]
    movdqu  xmm15, [rcx + 290h]
    ldmxcsr dword ptr [rcx + 034h]

    mov     rbx, [rcx + 090h]
    mov     rbp, [rcx + 0A0h]
    mov     rsi, [rcx + 0A8h]
    mov     rdi, [rcx + 0B0h]
    mov     r12, [rcx + 0D8h]
    mov     r13, [rcx + 0E0h]
    mov     r14, [rcx + 0E8h]
    mov     r15, [rcx + 0F0h]
    mov     rdx, [rcx + 088h]
```

```h
typedef PVOID PKPRCB;

typedef struct _PROCESSOR_CONTROL_STATE {

    ULONG            ProcessorNumber;
    BOOLEAN          IsValid;
    volatile PULONG  IpiFrozen;
} PROCESSOR_CONTROL_STATE, * PPROCESSOR_CONTROL_STATE;

extern "C" {

    extern PVOID HalPrivateDispatchTable;

    NTSTATUS      NTAPI RtlGetVersion(_Out_ PRTL_OSVERSIONINFOW VersionInformation);
    VOID          NTAPI RtlCaptureContext(_Out_ PCONTEXT ContextRecord);
    PKTHREAD      NTAPI KeGetCurrentThread(VOID);

    VOID          NTAPI KeStallExecutionProcessor(ULONG MicroSeconds);

    // ------------------------------------------------------------------------
    // Asm stubs (cet_stubs.asm)

    // Read shadow stack pointer (rdsspq).
    ULONG64 HvciReadSsp(VOID);

    // Pop N entries from shadow stack (incsspq rcx). Count must be 1-255.
    VOID HvciIncSspN(ULONG count);

    // pops SspPopCount shadow-stack entries, restores
    // registers from Context, switches RSP, STIs, indirect JMP to TargetRip
    // with RAX = TargetRax. Never returns.
    DECLSPEC_NORETURN VOID HvciKcetJmpRestoreFixed(
        PCONTEXT Context,
        ULONG64  TargetRip,
        ULONG64  TargetRsp,
        ULONG64  TargetRax,
        ULONG    SspPopCount);

    // Non-kCET equivalent of the above. Never returns.
    DECLSPEC_NORETURN VOID HvciJmpRestore(
        PCONTEXT Context,
        ULONG64  TargetRip,
        ULONG64  TargetRsp,
        ULONG64  TargetRax);

    // ----------------------------------------------------------------------
    // SEH-driven recovery

    // Bad-driver-origin recovery. Walks frames from the hook to the bad
```

```cpp
// ============================================================================
// hvci_seh_recovery.cpp - HVCI/kCET-safe SEH-driven bugcheck recovery.
//
// Drives RtlUnwindEx from our hook frame back to the bad driver's caller, running every intermediate 
// __finally handler on the way. That cleans
// up MM/IPI state the kernel had set up before KeBugCheckEx then resumes
// at the caller with RAX = STATUS_ACCESS_VIOLATION - exactly as if SEH
// had naturally caught the violation.
//
// HVCI-safe:  uses only exported Rtl* routines, reads PE .pdata only.
// kCET-safe:  RtlUnwindEx is shadow-stack-aware on Win11 24H2.
// PG-safe:    no mutation of any PatchGuard-watched structure.
// ============================================================================

#include <ntifs.h>
#include <ntddk.h>
#include <intrin.h>
#include "Header.h"

// ----------------------------------------------------------------------------
// Types and macros
// ----------------------------------------------------------------------------

typedef struct _IMAGE_RUNTIME_FUNCTION_ENTRY* PRUNTIME_FUNCTION;

#ifndef UNW_FLAG_NHANDLER
#define UNW_FLAG_NHANDLER 0x0
#endif
#ifndef UNW_FLAG_EHANDLER
#define UNW_FLAG_EHANDLER 0x1
#endif
#ifndef UNW_FLAG_UHANDLER
#define UNW_FLAG_UHANDLER 0x2
#endif

extern PVOID g_NtosBase;
extern ULONG g_NtosSize;
extern PVOID g_OurBase;
extern ULONG g_OurSize;

// ----------------------------------------------------------------------------
// Exports from ntoskrnl. Plain extern "C" decls (no NTSYSAPI) to match
// the project's existing Header.h style - prevents "inconsistent dll
// linkage" warnings against any WDK header that already saw them.
//
// HistoryTable and ContextPointers are passed as PVOID/NULL throughout
// this module; the underlying types (PUNWIND_HISTORY_TABLE,
// PKNONVOLATILE_CONTEXT_POINTERS) are kernel-private structures we
// have no need to introspect.
// ----------------------------------------------------------------------------
extern "C" {

    PRUNTIME_FUNCTION NTAPI RtlLookupFunctionEntry(
        _In_ ULONG64 ControlPc,
        _Out_ PULONG64 ImageBase,
        _Inout_opt_ PVOID HistoryTable);

    PEXCEPTION_ROUTINE NTAPI RtlVirtualUnwind(
        _In_ ULONG HandlerType,
        _In_ ULONG64 ImageBase,
        _In_ ULONG64 ControlPc,
        _In_ PRUNTIME_FUNCTION FunctionEntry,
        _Inout_ PCONTEXT ContextRecord,
        _Out_ PVOID* HandlerData,
        _Out_ PULONG64 EstablisherFrame,
        _Inout_opt_ PVOID ContextPointers);

    VOID NTAPI RtlUnwindEx(
        _In_opt_ PVOID TargetFrame,
        _In_opt_ PVOID TargetIp,
        _In_opt_ PEXCEPTION_RECORD ExceptionRecord,
        _In_ PVOID ReturnValue,
        _In_ PCONTEXT ContextRecord,
        _In_opt_ PVOID HistoryTable);
}

// Target is acceptable if it has a valid RUNTIME_FUNCTION (so the
// resumed frame has consistent unwind metadata for the caller chain).
static BOOLEAN HvciSehTargetIsAcceptable(_In_ ULONG64 rip)
{
    if (rip < 0xFFFF000000000000ULL) return FALSE;
    if (!MmIsAddressValid((PVOID)rip)) return FALSE;

    ULONG64 imageBase = 0;
    PRUNTIME_FUNCTION fe = RtlLookupFunctionEntry(rip, &imageBase, NULL);
    return (fe != NULL) ? TRUE : FALSE;
}

// Classify the frame's RIP. Used as the trap-frame-crossing detector:
// known-image -> known-image transitions are normal walking; the first
// known -> unknown transition means we just unwound through KiPageFault
// and landed in the bad driver.
static BOOLEAN HvciRipIsKnownImage(_In_ ULONG64 rip)
{
    if (g_NtosBase && g_NtosSize) {
        ULONG64 base = (ULONG64)g_NtosBase;
        if (rip >= base && rip < base + g_NtosSize) return TRUE;
    }
    if (g_OurBase && g_OurSize) {
        ULONG64 base = (ULONG64)g_OurBase;
        if (rip >= base && rip < base + g_OurSize) return TRUE;
    }
    return FALSE;
}

// ----------------------------------------------------------------------------
// Drives a structured exception unwind from our hook frame up to the
// bad driver's caller. Returns only on failure; on success the unwinder
// transfers control non-locally and never returns.
#define UNWIND_MAX_DEPTH    64

extern "C"
NTSTATUS HvciSehUnwindRecovery(
    _In_ PCONTEXT savedTrapContext,
    _In_ ULONG64  targetRax)
{
    UNREFERENCED_PARAMETER(targetRax);

    if (!savedTrapContext || !MmIsAddressValid(savedTrapContext)) {
        return STATUS_INVALID_PARAMETER;
    }

    DbgPrint("[Suppressor/SEH] begin unwind: trapRip=%p trapRsp=%p "
        "ntos=%p+%lx self=%p+%lx\n",
        (PVOID)savedTrapContext->Rip, (PVOID)savedTrapContext->Rsp,
        g_NtosBase, g_NtosSize, g_OurBase, g_OurSize);

    if (!g_NtosBase || !g_NtosSize) {
        DbgPrint("[Suppressor/SEH] ntoskrnl range not resolved - cannot "
            "classify frames\n");
        return STATUS_NOT_SUPPORTED;
    }

    // Allocate large CONTEXTs on the heap rather than the stack - we
    // are at HIGH_LEVEL with peers pinned, and CONTEXT is ~1.2KB which
    // is enough to matter on a deeply-nested kernel stack. Pool tag
    // 'hScE' = "Hvci Seh Context".
    //
    // Pool failure here is treated as a fatal recovery failure and
    // surfaced to the caller - the asm fallback can still run.
    PCONTEXT origCtx = (PCONTEXT)ExAllocatePool2(
        POOL_FLAG_NON_PAGED, sizeof(CONTEXT), 'hScE');
    PCONTEXT walkCtx = (PCONTEXT)ExAllocatePool2(
        POOL_FLAG_NON_PAGED, sizeof(CONTEXT), 'hScE');
    if (!origCtx || !walkCtx) {
        if (origCtx) ExFreePoolWithTag(origCtx, 'hScE');
        if (walkCtx) ExFreePoolWithTag(walkCtx, 'hScE');
        DbgPrint("[Suppressor/SEH] context alloc failed\n");
        return STATUS_INSUFFICIENT_RESOURCES;
    }

    origCtx->ContextFlags = CONTEXT_FULL;
    RtlCaptureContext(origCtx);

    // Walk up using a copy of origCtx so we leave origCtx pristine for
    // RtlUnwindEx to consume.
    RtlCopyMemory(walkCtx, origCtx, sizeof(CONTEXT));

    ULONG64 targetFrame = 0;
    ULONG64 targetIp = 0;
    NTSTATUS resultStatus = STATUS_NOT_FOUND;

    __try {
        // The walk has two phases:
        //   Phase A: walking through known images (our hook + ntoskrnl)
        //            on the way down to the trap. Continue unwinding.
        //   Phase B: we've unwound through KiPageFault's machine frame
        //            and walkCtx->Rip now points into the bad driver.
        //            We unwind ONE more frameto reach the bad driver's
        //            caller (back in ntoskrnl - typically IopLoadDriver),
        //            and that's our target.
        BOOLEAN inBadDriver = FALSE;

        for (ULONG depth = 0; depth < UNWIND_MAX_DEPTH; depth++) {
            ULONG64 imageBase = 0;
            PRUNTIME_FUNCTION fe = RtlLookupFunctionEntry(
                walkCtx->Rip, &imageBase, NULL);

            if (!fe) {
                // No .pdata for the current frame. If we're already in
                // the bad driver and this is the bad driver's frame,
                // we cannot proceed the bad driver lacks unwind info
                // If we're in ntoskrnl/self, this is a leaf function we
                // shouldn't have hit either way, bail.
                DbgPrint("[Suppressor/SEH] no .pdata at rip=%p depth=%lu "
                    "inBadDriver=%d - fallback to asm\n",
                    (PVOID)walkCtx->Rip, depth, inBadDriver ? 1 : 0);
                resultStatus = STATUS_NOT_SUPPORTED;
                __leave;
            }

            DbgPrint("[Suppressor/SEH] frame[%lu] rip=%p rsp=%p "
                "imageBase=%p inBadDriver=%d\n",
                depth, (PVOID)walkCtx->Rip, (PVOID)walkCtx->Rsp,
                (PVOID)imageBase, inBadDriver ? 1 : 0);

            // Unwind one frame. After this call walkCtx->Rip is the
            // CALLER's RIP (the return address of the function we
            // just unwound).
            PVOID    handlerData = NULL;
            ULONG64  establisherFrame = 0;
            (VOID)RtlVirtualUnwind(
                UNW_FLAG_NHANDLER,
                imageBase,
                walkCtx->Rip,
                fe,
                walkCtx,
                &handlerData,
                &establisherFrame,
                NULL);

            if (walkCtx->Rip == 0) {
                DbgPrint("[Suppressor/SEH] hit end-of-stack at depth=%lu\n",
                    depth);
                resultStatus = STATUS_NOT_FOUND;
                __leave;
            }

            BOOLEAN nowInKnown = HvciRipIsKnownImage(walkCtx->Rip);

            if (!inBadDriver) {
                if (!nowInKnown) {
                    // First transition out of ntoskrnl/self - we just
                    // unwound through KiPageFault's machine frame and
                    // landed on the bad driver's faulting RIP.
                    DbgPrint("[Suppressor/SEH] entered bad-driver frame "
                        "at depth=%lu rip=%p rsp=%p\n",
                        depth, (PVOID)walkCtx->Rip, (PVOID)walkCtx->Rsp);
                    inBadDriver = TRUE;
                    continue;   // walk ONE more frame to reach caller
                }
                // Still in known image - keep walking.
            }
            else {
                // We were in the bad driver; check if we've now reached
                // back into ntoskrnl/self (the bad driver's caller -
                // typically IopLoadDriver invoking DriverEntry).
                if (nowInKnown && HvciSehTargetIsAcceptable(walkCtx->Rip)) {
                    targetFrame = walkCtx->Rsp;
                    targetIp = walkCtx->Rip;
                    DbgPrint("[Suppressor/SEH] target found: ip=%p frame=%p "
                        "(bad-driver caller, depth=%lu)\n",
                        (PVOID)targetIp, (PVOID)targetFrame, depth);
                    break;
                }
                // Still in bad driver (chained internal call) - keep
                // unwinding until we exit it.
            }
        }

        if (targetFrame == 0 || targetIp == 0) {
            DbgPrint("[Suppressor/SEH] depth limit reached without target\n");
            resultStatus = STATUS_NOT_FOUND;
            __leave;
        }

        // Mirror what SEH would deliver if it had caught the violation
        // naturally - __finally handlers along the unwind path copy this
        // record to inspect the violation details.
        EXCEPTION_RECORD exception;
        RtlZeroMemory(&exception, sizeof(exception));
        exception.ExceptionCode = STATUS_ACCESS_VIOLATION;
        exception.ExceptionFlags = EXCEPTION_NONCONTINUABLE;
        exception.ExceptionAddress = (PVOID)savedTrapContext->Rip;
        exception.NumberParameters = 2;
        exception.ExceptionInformation[0] = 1;                 // write
        exception.ExceptionInformation[1] = (ULONG_PTR)__readcr2();

        DbgPrint("[Suppressor/SEH] dispatching unwind to ip=%p frame=%p\n",
            (PVOID)targetIp, (PVOID)targetFrame);

        // Pool buffers intentionally leak on the success path: control
        // transfers via the unwinder and never returns, so any post-call
        // ExFreePoolWithTag is unreachable. ~2.5 KB per suppression is
        // acceptable for a research PoC.
        RtlUnwindEx(
            (PVOID)targetFrame,
            (PVOID)targetIp,
            &exception,
            (PVOID)(ULONG_PTR)STATUS_ACCESS_VIOLATION,
            origCtx,
            NULL);

        DbgPrint("[Suppressor/SEH] RtlUnwindEx returned - chain invalidated\n");
        resultStatus = STATUS_UNSUCCESSFUL;
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {
        DbgPrint("[Suppressor/SEH] unwind raised exception 0x%08X\n",
            GetExceptionCode());
        resultStatus = STATUS_UNSUCCESSFUL;
    }

    // Failure path: we still own both buffers. Free them and return.
    if (origCtx) ExFreePoolWithTag(origCtx, 'hScE');
    if (walkCtx) ExFreePoolWithTag(walkCtx, 'hScE');

    return resultStatus;
}

// ============================================================================
// HvciKernelOriginUnwindRecovery
//
// SEH unwind for kernel-origin bugchecks (faults whose RIP lies inside
// ntoskrnl - PspProcessDelete null-deref, list-entry validation int 0x29,
// PG self-checks, etc.). Distinct from HvciSehUnwindRecovery because
// kernel-origin has no bad-driver-frame anchor for the image-classifier
// to pivot on.
//
// The caller has ALREADY resolved a target (koCallerRip, koBadRsp2 + offset)
// via cross-stack matching. We accept those as authoritative and call
// RtlUnwindEx directly with that frame as the destination, letting the
// unwinder walk each intermediate ntoskrnl frame and run its __finally
// handlers. That releases the locks/critical-regions/APC counts that the
// asm-restore-only path leaks - which is what causes the cascading freeze
// after "Bypass!" on kernel-origin paths today.
//
// Preconditions (must be satisfied by caller):
//   * IRQL has been lowered to savedIrql (not HIGH_LEVEL)
//   * Bugcheck globals released (slaves out of KiFreezeTargetExecution)
//   * Interrupts enabled (RtlUnwindEx requires this)
//   * targetRip is in canonical kernel space and has valid .pdata
//   * targetRsp is the RSP of the resume frame (caller's RSP, post-call)
//
// Returns STATUS_SUCCESS NEVER (transfers control non-locally on success).
// On failure returns a status; caller should fall through to the asm restore as a last resort
// ============================================================================
extern "C"
NTSTATUS HvciKernelOriginUnwindRecovery(
    _In_ PCONTEXT savedTrapContext,
    _In_ ULONG64  targetRip,
    _In_ ULONG64  targetRsp,
    _In_ ULONG64  targetRax)
{
    if (targetRip == 0 || targetRsp == 0) {
        DbgPrint("[Suppressor/SEH-KO] invalid target ip=%p frame=%p\n",
            (PVOID)targetRip, (PVOID)targetRsp);
        return STATUS_INVALID_PARAMETER;
    }

    if (!HvciSehTargetIsAcceptable(targetRip)) {
        DbgPrint("[Suppressor/SEH-KO] target unacceptable: ip=%p (no .pdata "
            "or non-canonical)\n", (PVOID)targetRip);
        return STATUS_NOT_SUPPORTED;
    }

    DbgPrint("[Suppressor/SEH-KO] begin: caller-target ip=%p frame=%p rax=%p "
        "trapRip=%p\n",
        (PVOID)targetRip, (PVOID)targetRsp, (PVOID)targetRax,
        savedTrapContext ? (PVOID)savedTrapContext->Rip : NULL);

    // CONTEXT is ~1.2KB - heap-allocate to avoid bloating the kernel
    // stack (we are several frames deep already). Pool tag 'hSKO' =
    // "Hvci Seh Kernel-Origin".
    PCONTEXT origCtx = (PCONTEXT)ExAllocatePool2(
        POOL_FLAG_NON_PAGED, sizeof(CONTEXT), 'hSKO');
    PCONTEXT walkCtx = (PCONTEXT)ExAllocatePool2(
        POOL_FLAG_NON_PAGED, sizeof(CONTEXT), 'hSKO');
    if (!origCtx || !walkCtx) {
        if (origCtx) ExFreePoolWithTag(origCtx, 'hSKO');
        if (walkCtx) ExFreePoolWithTag(walkCtx, 'hSKO');
        DbgPrint("[Suppressor/SEH-KO] context alloc failed\n");
        return STATUS_INSUFFICIENT_RESOURCES;
    }

    origCtx->ContextFlags = CONTEXT_FULL;
    RtlCaptureContext(origCtx);
    // origCtx now describes THIS frame.

    // Walk up the stack as far as RtlVirtualUnwind can reach, looking
    // for the DEEPEST valid kernel frame with usable .pdata. The reason:
    // the caller-provided target (koCallerRip) is the immediate caller
    // of the faulting function. Resuming there leaves intact every
    // ntoskrnl frame between the caller and the thread-start anchor -
    // and most of those frames hold locks not protected by __finally,
    // so the asm-restore-only path leaks them and we deadlock seconds
    // later. By unwinding much further up (ideally to a thread-start
    // anchor: ExpWorkerThread /KiSystemServiceCopyEnd /
    // PspSystemThreadStartup) we run __finally on EVERY intermediate
    // frame and resume in code that's structurally able to handle
    // "this thread is done."
    //
    // We don't have symbols for the anchor functions, so we use a
    // proxy: the deepest frame whose RIP still resolves to a valid
    // kernel function entry. Functions with no .pdata are the natural
    // bottom (thread-start trampolines and KiStartSystemThread-class
    // routines often have minimal/no unwind info).
    RtlCopyMemory(walkCtx, origCtx, sizeof(CONTEXT));

    ULONG64 deepestRip = targetRip;
    ULONG64 deepestRsp = targetRsp;
    ULONG   walkedFrames = 0;
    ULONG   acceptable = 0;

    __try {
        for (ULONG depth = 0; depth < 64; depth++) {
            ULONG64 imageBase = 0;
            PRUNTIME_FUNCTION fe = RtlLookupFunctionEntry(
                walkCtx->Rip, &imageBase, NULL);
            if (!fe) break;

            PVOID    handlerData = NULL;
            ULONG64  establisherFrame = 0;
            (VOID)RtlVirtualUnwind(
                UNW_FLAG_NHANDLER,
                imageBase,
                walkCtx->Rip,
                fe,
                walkCtx,
                &handlerData,
                &establisherFrame,
                NULL);

            walkedFrames++;

            if (walkCtx->Rip == 0) break;
            if (!HvciRipIsKnownImage(walkCtx->Rip)) break;
            if (walkCtx->Rsp == 0 || !MmIsAddressValid((PVOID)walkCtx->Rsp)) break;

            // Only accept this frame as a resume target if RIP itself
            // has a function entry too - else RtlUnwindEx will fail on
            // the final dispatch.
            ULONG64 nextBase = 0;
            PRUNTIME_FUNCTION nextFe = RtlLookupFunctionEntry(
                walkCtx->Rip, &nextBase, NULL);
            if (nextFe && walkCtx->Rsp > deepestRsp) {
                deepestRip = walkCtx->Rip;
                deepestRsp = walkCtx->Rsp;
                acceptable++;
            }
        }
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {
        DbgPrint("[Suppressor/SEH-KO] walk raised 0x%08X - using last good "
            "target\n", GetExceptionCode());
    }

    DbgPrint("[Suppressor/SEH-KO] walked %lu frames, %lu acceptable; "
        "deepest target ip=%p frame=%p\n",
        walkedFrames, acceptable, (PVOID)deepestRip, (PVOID)deepestRsp);

    NTSTATUS resultStatus = STATUS_UNSUCCESSFUL;

    __try {
        EXCEPTION_RECORD exception;
        RtlZeroMemory(&exception, sizeof(exception));
        exception.ExceptionCode = STATUS_ACCESS_VIOLATION;
        exception.ExceptionFlags = EXCEPTION_NONCONTINUABLE;
        exception.ExceptionAddress = savedTrapContext
            ? (PVOID)savedTrapContext->Rip
            : (PVOID)deepestRip;
        // Synthesize the AV operand the way KiPageFault would have, so
        // any __finally that introspects ExceptionInformation behaves
        // as in a real AV.
        exception.NumberParameters = 2;
        exception.ExceptionInformation[0] = 0;        // 0 = read; we don't
        // know the actual op
        exception.ExceptionInformation[1] = 0;

        DbgPrint("[Suppressor/SEH-KO] dispatching unwind to deepest target\n");

        // Pool buffers intentionally leak on success (control transfers
        // via the unwinder and never returns). ~2.5 KB per kernel-origin
        // suppression is acceptable for a research PoC.
        RtlUnwindEx(
            (PVOID)deepestRsp,                                  // TargetFrame
            (PVOID)deepestRip,                                  // TargetIp
            &exception,
            (PVOID)(ULONG_PTR)targetRax,                        // RAX at resume
            origCtx,
            NULL);

        // Reaching here means RtlUnwindEx couldn't deliver - typically
        // because a frame between us and target lacks .pdata, or the
        // target frame's RSP doesn't actually appear during the walk.
        DbgPrint("[Suppressor/SEH-KO] RtlUnwindEx returned without "
            "transferring - failure\n");
        resultStatus = STATUS_UNSUCCESSFUL;
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {
        // Common: STATUS_BAD_FUNCTION_TABLE if .pdata gap, or
        // STATUS_INVALID_UNWIND_TARGET if RSP mismatch.
        DbgPrint("[Suppressor/SEH-KO] unwind raised 0x%08X\n",
            GetExceptionCode());
        resultStatus = STATUS_UNSUCCESSFUL;
    }

    // Failure path: free both contexts. Success path leaked them.
    if (origCtx) ExFreePoolWithTag(origCtx, 'hSKO');
    if (walkCtx) ExFreePoolWithTag(walkCtx, 'hSKO');
    return resultStatus;
}
```

```cpp
// ============================================================================
// HIGH-IRQL RING-BUFFER LOG
//
// DbgPrint cannot be called from inside a HalpPrepareForBugcheck hook.
// KeBugCheck2 has already broadcast its freeze IPI before our hook runs;
// DbgPrint emits int 3 -> KdpTrap -> KdEnterDebugger -> KeFreezeExecution
// which tries to freeze peers that are pinned in KiIpiWaitForRequestBarrier
// waiting for an ACK only this CPU can deliver. Result: mutual deadlock.
// (MSDN wdm.h: "calling DbgPrint at IRQL>DIRQL can cause deadlocks.")
//
// We log raw values (no formatting - sprintf is not safe at HIGH_LEVEL)
// into a fixed-size ring; the cleanup DPC, which runs at DISPATCH_LEVEL
// well after the bugcheck context has unwound, flushes the ring to
// DbgPrint where the kernel debugger can show it.
// ============================================================================

#define HVCI_LOG_MAX_ENTRIES 256

#define HVCI_EV_HOOK_ENTRY              1   // cr8, nmi, hookCount, savedRip, savedRsp, ctxFlags
#define HVCI_EV_BAD_PRCB                2
#define HVCI_EV_BAD_CONTEXT             3
#define HVCI_EV_BAD_CTX_FLAGS           4   // ctxFlags
#define HVCI_EV_BAD_RSP                 5   // rsp
#define HVCI_EV_TOKEN_FOUND             6   // sspIdx, savedSsp, ra, popCount
#define HVCI_EV_TOKEN_TARGET_NOT_ON_RSP 7   // ra
#define HVCI_EV_NO_TOKEN                8   // searched depth
#define HVCI_EV_RESOLVE_EXC             9
#define HVCI_EV_CROSS_MATCH            10   // ra, rspOff, sspIdx
#define HVCI_EV_CROSS_FAIL             11
#define HVCI_EV_CROSS_EXC              12
#define HVCI_EV_INVALID_TARGET         13   // ra
#define HVCI_EV_NO_RSP_OFF             14
#define HVCI_EV_TARGET_RESOLVED        15   // targetRip, savedRip, savedRsp, irql, rspOff, sspPop
#define HVCI_EV_TRANSFER_KCET          16
#define HVCI_EV_TRANSFER_NONKCET       17
#define HVCI_EV_HOOK_EXCEPTION         18   // exception code
#define HVCI_EV_FALLBACK               19
#define HVCI_EV_RSP_DUMP               20   // index, slotAddr, slotValue
#define HVCI_EV_SSP_DUMP               21   // index, slotAddr, slotValue
#define HVCI_EV_RSP_HEAD               22   // rspStart
#define HVCI_EV_SSP_HEAD               23   // ssp
#define HVCI_EV_SSP_INVALID_ENTRY      24   // index, slotAddr
#define HVCI_EV_SSP_NULL               25
#define HVCI_EV_SUPPRESS_CLEARED       26
#define HVCI_EV_FIND_SSP_WALK          27   // ssp, target
#define HVCI_EV_FIND_SSP_SLOT          28   // i, slotAddr, value
#define HVCI_EV_FIND_SSP_HIT           29
#define HVCI_EV_TRAP_FRAME             30   // trapOff, badRip, badRsp
#define HVCI_EV_NON_IST_TRAP           31   // savedRsp, badRsp, delta
#define HVCI_EV_SKIP_INSTRUCTION       32   // badRip, instLen, targetRip, dispDepth
#define HVCI_EV_INST_DECODE_FAIL       33   // badRip
#define HVCI_EV_NO_DISPATCHER_TOKEN    34   // searched depth   // i
#define HVCI_EV_FREEZE_HOOK            35   // Freeze, cpuIdx, suppressedCnt, action(0=pass,1=eat,2=unbalanced)
#define HVCI_EV_DPC_WD_RESET           36   // cpuIdx, period, oldCount, action(0=ok,1=skipped)
#define HVCI_EV_HAL_WD_STUB            37   // first byte
#define HVCI_EV_PRE_RESUME             38   // savedIrql, kcet, skip, sspPop, targetRip, targetRsp
#define HVCI_EV_IST_DELTA              39   // savedRsp, badRsp, delta, isIst
#define HVCI_EV_TRAP_FRAME_DETAIL      40   // cs, rflags, rsp, ss, rip, rspBase

typedef struct _HVCI_LOG_ENTRY {
    ULONG  Code;
    ULONG  Cpu;
    ULONG64 Args[6];
} HVCI_LOG_ENTRY;

#pragma data_seg(".data")
HVCI_LOG_ENTRY g_HvciLog[HVCI_LOG_MAX_ENTRIES] = { 0 };
volatile LONG g_HvciLogIdx = 0;
#pragma data_seg()

__forceinline VOID HvciLog(
    ULONG code,
    ULONG64 a1, ULONG64 a2, ULONG64 a3,
    ULONG64 a4, ULONG64 a5, ULONG64 a6)
{
    LONG idx = (InterlockedIncrement(&g_HvciLogIdx) - 1) & (HVCI_LOG_MAX_ENTRIES - 1);
    HVCI_LOG_ENTRY* e = &g_HvciLog[idx];
    e->Cpu = KeGetCurrentProcessorIndex();
    e->Args[0] = a1;
    e->Args[1] = a2;
    e->Args[2] = a3;
    e->Args[3] = a4;
    e->Args[4] = a5;
    e->Args[5] = a6;
    _mm_sfence();
    e->Code = code;     // last write - readers can poll on this
}

// Flush the ring to DbgPrint. Caller MUST be at IRQL <= DISPATCH_LEVEL
// (i.e. the suppression-clear DPC, NOT the bugcheck hook itself).
VOID HvciFlushLogToDbgPrint(VOID)
{
    LONG total = g_HvciLogIdx;
    LONG count = total > HVCI_LOG_MAX_ENTRIES ? HVCI_LOG_MAX_ENTRIES : total;
    LONG start = total > HVCI_LOG_MAX_ENTRIES ? total - HVCI_LOG_MAX_ENTRIES : 0;

    for (LONG i = 0; i < count; i++) {
        LONG slot = (start + i) & (HVCI_LOG_MAX_ENTRIES - 1);
        HVCI_LOG_ENTRY* e = &g_HvciLog[slot];
        if (e->Code == 0) continue;
        DbgPrint("[Suppressor] LOG[%ld] cpu=%lu code=%lu "
            "%p %p %p %p %p %p\n",
            start + i, e->Cpu, e->Code,
            (PVOID)e->Args[0], (PVOID)e->Args[1], (PVOID)e->Args[2],
            (PVOID)e->Args[3], (PVOID)e->Args[4], (PVOID)e->Args[5]);
    }
}

ULONG GetWindowsBuildNumber()
{
    RTL_OSVERSIONINFOW VersionInfo = { 0 };
    VersionInfo.dwOSVersionInfoSize = sizeof(VersionInfo);

    __try {
        if (NT_SUCCESS(RtlGetVersion(&VersionInfo))) {
            return VersionInfo.dwBuildNumber;
        }
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {}

    return 26200;
}

BOOLEAN HvciSafeReadMemory(PVOID Source, PVOID Destination, SIZE_T Size)
{
    if (!Source || !Destination || Size == 0) return FALSE;

    for (ULONG attempt = 0; attempt < HVCI_SAFE_READ_ATTEMPTS; attempt++) {
        __try {
            if (!MmIsAddressValid(Source)) {
                KeStallExecutionProcessor(10);
                continue;
            }
            _mm_mfence();
            RtlCopyMemory(Destination, Source, Size);
            _mm_mfence();
            return TRUE;
        }
        __except (EXCEPTION_EXECUTE_HANDLER) {
            if (attempt < HVCI_SAFE_READ_ATTEMPTS - 1) {
                KeStallExecutionProcessor(50 * (attempt + 1));
            }
        }
    }
    return FALSE;
}

PKPRCB HvciGetCurrentPrcb()
{
    __try {
        PKPRCB prcb = (PKPRCB)__readgsqword(0x20);
        if (MmIsAddressValid(prcb)) return prcb;
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {}
    return nullptr;
}

PKPRCB HvciGetProcessorPrcb(ULONG ProcessorNumber)
{
    if (ProcessorNumber >= MAX_PROCESSORS) return nullptr;

    __try {
        PKPRCB prcb = KeQueryPrcbAddress(ProcessorNumber);
        if (prcb && MmIsAddressValid(prcb)) {
            ULONG numberValue;
            if (HvciSafeReadMemory((PUCHAR)prcb + KPRCB_NUMBER_OFFSET,
                &numberValue, sizeof(ULONG))) {
                if (numberValue == ProcessorNumber) {
                    return prcb;
                }
            }
        }
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {}

    if (ProcessorNumber == KeGetCurrentProcessorIndex()) {
        return HvciGetCurrentPrcb();
    }
    return nullptr;
}

// ============================================================================
// Find ntoskrnl base and text section bounds for pattern scanning
// ============================================================================

PVOID g_NtosBase = nullptr;
ULONG g_NtosSize = 0;

// Our own driver's image range, resolved at init. Used to reject
// resolver candidates whose RA points into our hook code - those are
// never the bad driver's pending RA and would cause kCET shadow-stack
// mismatch (#CP) on the resumed function's RET.
PVOID g_OurBase = nullptr;
ULONG g_OurSize = 0;

__forceinline BOOLEAN HvciIsOurAddress(_In_ ULONG64 ra)
{
    if (g_OurBase == nullptr || g_OurSize == 0) return FALSE;
    ULONG64 base = (ULONG64)g_OurBase;
    return ra >= base && ra < base + g_OurSize;
}

__forceinline BOOLEAN HvciIsSystemDispatchAddress(_In_ ULONG64 ra)
{
    // We deliberately do NOT exclude all of ntoskrnl here - the bad
    // driver's pending RA usually IS in ntoskrnl (e.g. inside
    // PspSystemThreadStartup, which is the caller of any system-thread
    // routine). We only filter out our own module.
    return HvciIsOurAddress(ra);
}

// ============================================================================
// PATTERN SCANNING
// ============================================================================

PUCHAR HvciFindPattern(PUCHAR start, ULONG length,
    const UCHAR* pattern, const char* mask)
{
    ULONG patternLen = (ULONG)strlen(mask);
    if (length < patternLen) return nullptr;

    for (ULONG i = 0; i <= length - patternLen; i++) {
        BOOLEAN match = TRUE;
        for (ULONG j = 0; j < patternLen; j++) {
            if (mask[j] == 'x' && start[i + j] != pattern[j]) {
                match = FALSE;
                break;
            }
        }
        if (match) return start + i;
    }
    return nullptr;
}


// Authoritative resolution path: ZwQuerySystemInformation with class
// SystemModuleInformation returns the loaded-driver list with base+size.
// Find the entry whose [ImageBase, ImageBase+ImageSize) contains
// KeBugCheckEx and use those values verbatim. Robust against false-
// positive MZ patterns inside ntoskrnl's own image (the previous scan
// could land on random "MZ" + valid e_lfanew + "PE\0\0" byte sequences
// embedded as data, then read a bogus SizeOfImage that misclassified
// real kernel frames as bad-driver during unwind).
//
// Caller-time: must be PASSIVE_LEVEL (Zw* APIs require it). We call
// this from DriverEntry, never from the bugcheck hook.
static BOOLEAN HvciFindNtosBaseViaModuleList(_In_ PVOID kbeAddr)
{
    typedef NTSTATUS(NTAPI* PFN_ZwQuerySystemInformation)(
        ULONG, PVOID, ULONG, PULONG);

    typedef struct _HVCI_RTL_PROCESS_MODULE_INFO {
        HANDLE Section;
        PVOID  MappedBase;
        PVOID  ImageBase;
        ULONG  ImageSize;
        ULONG  Flags;
        USHORT LoadOrderIndex;
        USHORT InitOrderIndex;
        USHORT LoadCount;
        USHORT OffsetToFileName;
        UCHAR  FullPathName[256];
    } HVCI_RTL_PROCESS_MODULE_INFO;

    typedef struct _HVCI_RTL_PROCESS_MODULES {
        ULONG NumberOfModules;
        HVCI_RTL_PROCESS_MODULE_INFO Modules[1];
    } HVCI_RTL_PROCESS_MODULES;

    UNICODE_STRING zwName;
    RtlInitUnicodeString(&zwName, L"ZwQuerySystemInformation");
    PFN_ZwQuerySystemInformation pZwQSI =
        (PFN_ZwQuerySystemInformation)MmGetSystemRoutineAddress(&zwName);
    if (!pZwQSI) {
        DbgPrint("[Suppressor] ZwQuerySystemInformation unresolved\n");
        return FALSE;
    }

    ULONG needed = 0;
    NTSTATUS s = pZwQSI(11 /*SystemModuleInformation*/, NULL, 0, &needed);
    if (s != STATUS_INFO_LENGTH_MISMATCH || needed == 0) {
        DbgPrint("[Suppressor] SystemModuleInformation size probe: 0x%08X\n", s);
        return FALSE;
    }

    // Slack for races against driver load/unload between probe and query.
    ULONG bufSize = needed + 0x4000;
    PVOID buf = ExAllocatePool2(POOL_FLAG_NON_PAGED, bufSize, 'NtoS');
    if (!buf) return FALSE;

    BOOLEAN found = FALSE;
    s = pZwQSI(11, buf, bufSize, &needed);
    if (NT_SUCCESS(s)) {
        HVCI_RTL_PROCESS_MODULES* mods = (HVCI_RTL_PROCESS_MODULES*)buf;
        ULONG_PTR kbeVa = (ULONG_PTR)kbeAddr;
        for (ULONG i = 0; i < mods->NumberOfModules; i++) {
            ULONG_PTR base = (ULONG_PTR)mods->Modules[i].ImageBase;
            ULONG sz = mods->Modules[i].ImageSize;
            if (sz == 0) continue;
            if (kbeVa < base || kbeVa >= base + sz) continue;
            g_NtosBase = (PVOID)base;
            g_NtosSize = sz;
            DbgPrint("[Suppressor] ntoskrnl resolved via module list: "
                "%p size=0x%lx (KeBugCheckEx@%p)\n",
                g_NtosBase, g_NtosSize, kbeAddr);
            found = TRUE;
            break;
        }
        if (!found) {
            DbgPrint("[Suppressor] No module contains KeBugCheckEx@%p "
                "(scanned %lu modules)\n", kbeAddr, mods->NumberOfModules);
        }
    }
    else {
        DbgPrint("[Suppressor] SystemModuleInformation query: 0x%08X\n", s);
    }

    ExFreePoolWithTag(buf, 'NtoS');
    return found;
}

BOOLEAN HvciFindNtosBase()
{
    if (g_NtosBase && g_NtosSize) return TRUE;

    UNICODE_STRING name;
    RtlInitUnicodeString(&name, L"KeBugCheckEx");
    PVOID kbeAddr = MmGetSystemRoutineAddress(&name);
    if (!kbeAddr) return FALSE;

    if (HvciFindNtosBaseViaModuleList(kbeAddr)) return TRUE;

    DbgPrint("[Suppressor] Module-list resolution failed - falling back "
        "to validated MZ scan\n");

    // Fallback: walk back from KeBugCheckEx for an MZ stub, but VALIDATE
    // each candidate by requiring KeBugCheckEx to lie within
    // [base, base+SizeOfImage). False MZ patterns inside the real kernel
    // image fail this because their bogus SizeOfImage doesn't reach back
    // to where KeBugCheckEx actually lives.
    PUCHAR pageStart = (PUCHAR)((ULONG_PTR)kbeAddr & ~0xFFFULL);
    ULONG_PTR kbeVa = (ULONG_PTR)kbeAddr;

    for (ULONG i = 0; i < 0x2000; i++) {  // up to 32 MB back
        PUCHAR probe = pageStart - (i * 0x1000);
        if (!MmIsAddressValid(probe)) break;

        __try {
            if (*(PUSHORT)probe != 0x5A4D) continue;            // "MZ"

            LONG e_lfanew = *(LONG*)(probe + 0x3C);
            if (e_lfanew <= 0 || e_lfanew >= 0x1000) continue;
            if (*(PULONG)(probe + e_lfanew) != 0x4550) continue; // "PE\0\0"

            ULONG candidateSize = *(PULONG)(probe + e_lfanew + 0x50);
            if (candidateSize < 0x10000) continue;               // sanity floor

            ULONG_PTR baseVa = (ULONG_PTR)probe;
            if (kbeVa < baseVa || kbeVa >= baseVa + candidateSize) {
                DbgPrint("[Suppressor] MZ@%p size=0x%lx does not contain "
                    "KeBugCheckEx@%p - skipping false positive\n",
                    probe, candidateSize, kbeAddr);
                continue;
            }

            g_NtosBase = probe;
            g_NtosSize = candidateSize;
            DbgPrint("[Suppressor] ntoskrnl base: %p size: 0x%lx (validated)\n",
                probe, candidateSize);
            return TRUE;
        }
        __except (EXCEPTION_EXECUTE_HANDLER) {
            return FALSE;
        }
    }
    return FALSE;
}

// ============================================================================
// Scan ntoskrnl for KiFreezeTargetExecution by its unique prolog signature
// ============================================================================

PVOID HvciScanForKiFreezeTargetExecution()
{
    if (!g_NtosBase || !g_NtosSize) {
        if (!HvciFindNtosBase()) {
            DbgPrint("[Suppressor] Could not find ntoskrnl base\n");
            return nullptr;
        }
    }

    // KiFreezeTargetExecution prolog from disassembly:
    //   48 89 5C 24 18    mov [rsp+18h], rbx
    //   55                push rbp
    //   56                push rsi
    //   57                push rdi
    //   41 54             push r12
    //   41 56             push r14
    //   48 81 EC D0 00 00 00   sub rsp, 0D0h
    UCHAR pat[] = {
        0x48, 0x89, 0x5C, 0x24, 0x18,
        0x55, 0x56, 0x57,
        0x41, 0x54, 0x41, 0x56,
        0x48, 0x81, 0xEC, 0xD0, 0x00, 0x00, 0x00
    };
    const char* mask = "xxxxxxxxxxxxxxxxxxx";

    DbgPrint("[Suppressor] Scanning ntoskrnl for KiFreezeTargetExecution...\n");

    PUCHAR result = HvciFindPattern((PUCHAR)g_NtosBase, g_NtosSize, pat, mask);
    if (result) {
        DbgPrint("[Suppressor] Found KiFreezeTargetExecution at: %p\n", result);
        return result;
    }

    DbgPrint("[Suppressor] KiFreezeTargetExecution prolog not found\n");
    return nullptr;
}

// ============================================================================
// kCET DETECTION
// ============================================================================

// IA32_S_CET MSR - supervisor-mode CET enable. This is the
// authoritative system-wide check; rdsspq returns per-thread SSP and
// is 0 for any thread that has not passed through an ENDBR-protected
// entry yet, which is common for DriverEntry-context threads even on
// systems where kCET is fully active.
//
// IA32_S_CET bits:
//   [0]  SH_STK_EN    - enable supervisor shadow stacks
//   [2]  ENDBR_EN     - enable supervisor IBT
//   [3]  LEG_IW_EN    - legacy code page tracking
//   [4]  NO_TRACK_EN  - allow NOTRACK prefix
#define IA32_S_CET_MSR              0x6A2
#define IA32_S_CET_SH_STK_EN        (1ULL << 0)
#define IA32_S_CET_ENDBR_EN         (1ULL << 2)

BOOLEAN HvciDetectKcet()
{
    DbgPrint("[Suppressor] === kCET DETECTION ===\n");

    BOOLEAN cetBitSet = FALSE;

    __try {
        ULONG_PTR cr4 = __readcr4();
        DbgPrint("[Suppressor] CR4 = 0x%llx, CET bit (0x800000): %s\n",
            (ULONG64)cr4, (cr4 & CR4_CET_BIT) ? "SET" : "NOT SET");

        if (cr4 & CR4_CET_BIT) {
            cetBitSet = TRUE;
        }
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {
        DbgPrint("[Suppressor] Exception reading CR4\n");
        return FALSE;
    }

    if (!cetBitSet) {
        DbgPrint("[Suppressor] CR4.CET not set - kCET is OFF\n");
        return FALSE;
    }

    // Authoritative check: IA32_S_CET.SH_STK_EN. HVCI / Secure Kernel
    // keeps this MSR coherent across all logical processors, so reading
    // from one CPU's context tells us the system-wide state.
    ULONG64 sCet = 0;
    BOOLEAN sCetReadOk = FALSE;
    __try {
        sCet = __readmsr(IA32_S_CET_MSR);
        sCetReadOk = TRUE;
        DbgPrint("[Suppressor] IA32_S_CET = 0x%llx (SH_STK_EN: %s, "
            "ENDBR_EN: %s)\n",
            sCet,
            (sCet & IA32_S_CET_SH_STK_EN) ? "ON" : "OFF",
            (sCet & IA32_S_CET_ENDBR_EN) ? "ON" : "OFF");
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {
        DbgPrint("[Suppressor] IA32_S_CET MSR read faulted "
            "(virtual CPU may not expose CET MSRs)\n");
    }

    // Diagnostic only: rdsspq is per-thread, do NOT use it to gate the
    // detection. A 0 here just means THIS thread has not yet had its
    // kernel shadow stack initialized; other threads on the system may
    // still have full kCET enforcement.
    __try {
        ULONG64 ssp = HvciReadSsp();
        DbgPrint("[Suppressor] Current SSP via rdsspq: %p "
            "(per-thread, informational)\n", (PVOID)ssp);
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {
        DbgPrint("[Suppressor] rdsspq exception\n");
    }

    if (sCetReadOk && (sCet & IA32_S_CET_SH_STK_EN)) {
        DbgPrint("[Suppressor] *** kCET IS ENABLED system-wide ***\n");
        return TRUE;
    }

    DbgPrint("[Suppressor] kCET feature available but disabled "
        "(CR4.CET=1, IA32_S_CET.SH_STK_EN=0)\n");
    return FALSE;
}

// ============================================================================
// OFFSET VERIFICATION
// ============================================================================

VOID HvciVerifyKprcbOffsets()
{
    DbgPrint("[Suppressor] === KPRCB OFFSET VERIFICATION ===\n");

    PKPRCB prcb = HvciGetCurrentPrcb();
    if (!prcb) {
        DbgPrint("[Suppressor] Could not get PRCB\n");
        return;
    }

    DbgPrint("[Suppressor] Current PRCB: %p\n", prcb);

    __try {
        ULONG ipiFrozen = *(PULONG)((PUCHAR)prcb + KPRCB_IPI_FROZEN_OFFSET);
        DbgPrint("[Suppressor] IpiFrozen @ 0x%x: 0x%08X\n",
            KPRCB_IPI_FROZEN_OFFSET, ipiFrozen);
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {
        DbgPrint("[Suppressor] IpiFrozen read FAILED\n");
    }

    __try {
        PVOID ctxPtr = *(PVOID*)((PUCHAR)prcb + KPRCB_CONTEXT_OFFSET);
        BOOLEAN isKernelPtr = ((ULONG_PTR)ctxPtr >= 0xFFFF000000000000ULL);
        BOOLEAN isValid = ctxPtr && MmIsAddressValid(ctxPtr);

        if (isKernelPtr && isValid) {
            ULONG contextFlags = *(PULONG)ctxPtr;
            BOOLEAN looksLikeContext = (contextFlags & CONTEXT_AMD64_BIT) == CONTEXT_AMD64_BIT;
            DbgPrint("[Suppressor] Context @ 0x%x: %p ContextFlags=0x%08X %s\n",
                KPRCB_CONTEXT_OFFSET, ctxPtr, contextFlags,
                looksLikeContext ? "*** GOOD ***" : "(unexpected)");
        }
        else {
            DbgPrint("[Suppressor] Context @ 0x%x: %p (invalid)\n",
                KPRCB_CONTEXT_OFFSET, ctxPtr);
        }
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {
        DbgPrint("[Suppressor] Context read FAILED\n");
    }

    DbgPrint("[Suppressor] === END VERIFICATION ===\n");
}

// PROCESSOR STATE BUILDING
// ============================================================================

BOOLEAN HvciBuildProcessorControlStates()
{
    DbgPrint("[Suppressor] Building processor control states...\n");

    RtlZeroMemory(g_CompleteState.ProcessorStates, sizeof(g_CompleteState.ProcessorStates));
    g_CompleteState.ValidProcessorCount = 0;

    ULONG maxProcessors = KeQueryActiveProcessorCountEx(ALL_PROCESSOR_GROUPS);
    if (maxProcessors > MAX_PROCESSORS) maxProcessors = MAX_PROCESSORS;

    DbgPrint("[Suppressor] System has %lu processors\n", maxProcessors);

    for (ULONG i = 0; i < maxProcessors; i++) {
        __try {
            PKPRCB prcb = HvciGetProcessorPrcb(i);
            if (!prcb) continue;

            PPROCESSOR_CONTROL_STATE state =
                &g_CompleteState.ProcessorStates[g_CompleteState.ValidProcessorCount];

            state->Prcb = prcb;
            state->ProcessorNumber = i;
            state->IpiFrozen = (volatile PULONG)((PUCHAR)prcb + KPRCB_IPI_FROZEN_OFFSET);

            if (MmIsAddressValid((PVOID)state->IpiFrozen)) {
                state->IsValid = TRUE;
                g_CompleteState.ValidProcessorCount++;
                DbgPrint("[Suppressor] CPU[%lu]: PRCB=%p\n", i, prcb);
            }
        }
        __except (EXCEPTION_EXECUTE_HANDLER) {
            continue;
        }
    }

    DbgPrint("[Suppressor] Built control for %lu processors\n",
        g_CompleteState.ValidProcessorCount);
    return (g_CompleteState.ValidProcessorCount > 0);
}

// ============================================================================
// IPIFROZEN pre-stamp
//
// KiFreezeTargetExecution's slave loop exit condition is keyed on
// IpiFrozen != 5, not == 0. We pre-stamp every PRCB->IpiFrozen to 5
// at init via KeIpiGenericCall, so when KeBugCheck2 broadcasts its
// freeze IPI the slaves see "already frozen with our marker" and fall
// through the loop instead of busy-waiting on a barrier we don't fully
// control. Re-stamped on each successful suppression to handle Windows
// code paths that reset IpiFrozen to 0.
// ============================================================================

#define HVCI_IPI_FROZEN_MARKER  5

ULONG_PTR HvciStampIpiFrozenIpiCallback(_In_ ULONG_PTR Context)
{
    UNREFERENCED_PARAMETER(Context);

    PKPRCB prcb = (PKPRCB)__readgsqword(0x20);
    if (prcb && MmIsAddressValid(prcb)) {
        __try {
            volatile PULONG ipiFrozen =
                (volatile PULONG)((PUCHAR)prcb + KPRCB_IPI_FROZEN_OFFSET);
            InterlockedExchange((volatile LONG*)ipiFrozen,
                HVCI_IPI_FROZEN_MARKER);
        }
        __except (EXCEPTION_EXECUTE_HANDLER) {}
    }
    return 0;
}

VOID HvciPrestampIpiFrozenAllCpus(VOID)
{
    DbgPrint("[Suppressor] Pre-stamping IpiFrozen=%d on all CPUs\n",
        HVCI_IPI_FROZEN_MARKER);
    KeIpiGenericCall(HvciStampIpiFrozenIpiCallback, 0);
}

PVOID HvciResolveRipRel32(PVOID disp32Addr, PVOID instructionEnd)
{
    if (!MmIsAddressValid(disp32Addr)) return nullptr;
    INT32 disp = *(INT32*)disp32Addr;
    return (PUCHAR)instructionEnd + disp;
}

PVOID HvciFindFunctionStart(PVOID someAddressInFunction)
{
    PUCHAR addr = (PUCHAR)someAddressInFunction;

    // KiFreezeTargetExecution prolog: 48 89 5C 24 18 55 56 57
    // (mov [rsp+18h], rbx; push rbp; push rsi; push rdi)
    UCHAR pattern[] = { 0x48, 0x89, 0x5C, 0x24, 0x18, 0x55, 0x56, 0x57 };

    for (ULONG i = 0; i < 0x1000; i++) {
        if (!MmIsAddressValid(addr - i)) break;
        if (RtlCompareMemory(addr - i, pattern, sizeof(pattern)) == sizeof(pattern)) {
            return addr - i;
        }
    }
    return nullptr;
}

// ============================================================================
// STATE VARIABLE RESOLUTION
// 
// Resolves KiBugCheckActive, KiHardwareTrigger, KiFreezeExecutionLock,
// KiFreezeLockBackup by disassembling KeFreezeExecution (exported) and
// KeBugCheckEx (exported).
// ============================================================================

BOOLEAN HvciResolveBugCheckStateFromFreezeFunc(PVOID freezeFuncBase)
{
    if (!freezeFuncBase || !MmIsAddressValid(freezeFuncBase)) {
        DbgPrint("[Suppressor] ResolveState: freeze func base invalid\n");
        return FALSE;
    }

    g_KiFreezeTargetExecutionBase = freezeFuncBase;
    DbgPrint("[Suppressor] Freeze func base: %p\n", freezeFuncBase);

    PUCHAR funcStart = (PUCHAR)freezeFuncBase;
    const ULONG SCAN_LEN = 0x300;

    // Pattern: cmp qword ptr [rip+disp32], 0
    // Encoding: 48 83 3D ?? ?? ?? ?? 00
    UCHAR pat1[] = { 0x48, 0x83, 0x3D, 0x00, 0x00, 0x00, 0x00, 0x00 };
    const char* mask1 = "xxx????x";

    PUCHAR p1 = HvciFindPattern(funcStart, SCAN_LEN, pat1, mask1);
    if (!p1) {
        DbgPrint("[Suppressor] Could not find KiFreezeExecutionLock pattern\n");
        return FALSE;
    }

    g_KiFreezeExecutionLock = (volatile LONG*)HvciResolveRipRel32(p1 + 3, p1 + 8);
    DbgPrint("[Suppressor] KiFreezeExecutionLock: %p (offset +0x%lx)\n",
        (PVOID)g_KiFreezeExecutionLock, (ULONG)(p1 - funcStart));

    // Find next occurrence - KiFreezeLockBackup
    ULONG remaining = SCAN_LEN - (ULONG)(p1 + 8 - funcStart);
    PUCHAR p2 = HvciFindPattern(p1 + 8, remaining, pat1, mask1);
    if (p2) {
        g_KiFreezeLockBackup = (volatile LONG*)HvciResolveRipRel32(p2 + 3, p2 + 8);
        DbgPrint("[Suppressor] KiFreezeLockBackup: %p (offset +0x%lx)\n",
            (PVOID)g_KiFreezeLockBackup, (ULONG)(p2 - funcStart));
    }
    else {
        DbgPrint("[Suppressor] KiFreezeLockBackup pattern not found (may not be referenced here)\n");
        g_KiFreezeLockBackup = nullptr;
    }

    // Pattern: mov eax, dword ptr [rip+disp32]
    // Encoding: 8B 05 ?? ?? ?? ??
    UCHAR pat3[] = { 0x8B, 0x05, 0x00, 0x00, 0x00, 0x00 };
    const char* mask3 = "xx????";

    PUCHAR searchStart = p2 ? (p2 + 8) : (p1 + 8);
    remaining = SCAN_LEN - (ULONG)(searchStart - funcStart);
    PUCHAR p3 = HvciFindPattern(searchStart, remaining, pat3, mask3);
    if (p3) {
        g_KiBugCheckActive = (volatile LONG*)HvciResolveRipRel32(p3 + 2, p3 + 6);
        DbgPrint("[Suppressor] KiBugCheckActive: %p (offset +0x%lx)\n",
            (PVOID)g_KiBugCheckActive, (ULONG)(p3 - funcStart));
    }
    else {
        DbgPrint("[Suppressor] KiBugCheckActive pattern not found\n");
    }

    // Validate addresses
    BOOLEAN valid = TRUE;

    if (!g_KiBugCheckActive ||
        (ULONG_PTR)g_KiBugCheckActive < 0xFFFF000000000000ULL ||
        !MmIsAddressValid((PVOID)g_KiBugCheckActive)) {
        DbgPrint("[Suppressor] KiBugCheckActive address invalid\n");
        valid = FALSE;
    }
    if (!g_KiFreezeExecutionLock ||
        (ULONG_PTR)g_KiFreezeExecutionLock < 0xFFFF000000000000ULL ||
        !MmIsAddressValid((PVOID)g_KiFreezeExecutionLock)) {
        DbgPrint("[Suppressor] KiFreezeExecutionLock address invalid\n");
        valid = FALSE;
    }
    if (g_KiFreezeLockBackup &&
        ((ULONG_PTR)g_KiFreezeLockBackup < 0xFFFF000000000000ULL ||
            !MmIsAddressValid((PVOID)g_KiFreezeLockBackup))) {
        DbgPrint("[Suppressor] KiFreezeLockBackup address invalid, ignoring\n");
        g_KiFreezeLockBackup = nullptr;
    }

    if (valid) {
        __try {
            DbgPrint("[Suppressor] Current values:\n");
            if (g_KiBugCheckActive)
                DbgPrint("[Suppressor]   KiBugCheckActive      = %ld\n",
                    *g_KiBugCheckActive);
            if (g_KiFreezeExecutionLock)
                DbgPrint("[Suppressor]   KiFreezeExecutionLock = %ld\n",
                    *g_KiFreezeExecutionLock);
            if (g_KiFreezeLockBackup)
                DbgPrint("[Suppressor]   KiFreezeLockBackup    = %ld\n",
                    *g_KiFreezeLockBackup);
        }
        __except (EXCEPTION_EXECUTE_HANDLER) {
            DbgPrint("[Suppressor] Exception reading state vars\n");
            valid = FALSE;
        }
    }

    return valid;
}

BOOLEAN HvciResolveKiHardwareTrigger()
{
    UNICODE_STRING name;
    RtlInitUnicodeString(&name, L"KeBugCheckEx");
    PVOID keBugCheckEx = MmGetSystemRoutineAddress(&name);

    if (!keBugCheckEx) {
        DbgPrint("[Suppressor] Could not resolve KeBugCheckEx\n");
        return FALSE;
    }

    DbgPrint("[Suppressor] KeBugCheckEx: %p\n", keBugCheckEx);

    // Pattern A: F0 FF 05 ?? ?? ?? ?? (lock inc dword ptr [rip+disp32])
    UCHAR patA[] = { 0xF0, 0xFF, 0x05, 0x00, 0x00, 0x00, 0x00 };
    const char* maskA = "xxx????";

    PUCHAR result = HvciFindPattern((PUCHAR)keBugCheckEx, 0x200, patA, maskA);
    if (result) {
        g_KiHardwareTrigger = (volatile LONG*)HvciResolveRipRel32(result + 3, result + 7);
    }
    else {
        // Pattern B: F0 48 83 05 ?? ?? ?? ?? 01 (lock add cs:KiHardwareTrigger, 1)
        UCHAR patB[] = { 0xF0, 0x48, 0x83, 0x05, 0x00, 0x00, 0x00, 0x00, 0x01 };
        const char* maskB = "xxxx????x";
        result = HvciFindPattern((PUCHAR)keBugCheckEx, 0x200, patB, maskB);
        if (result) {
            g_KiHardwareTrigger = (volatile LONG*)HvciResolveRipRel32(result + 4, result + 9);
        }
    }

    if (g_KiHardwareTrigger &&
        (ULONG_PTR)g_KiHardwareTrigger >= 0xFFFF000000000000ULL &&
        MmIsAddressValid((PVOID)g_KiHardwareTrigger)) {
        DbgPrint("[Suppressor] KiHardwareTrigger: %p (current value: %ld)\n",
            (PVOID)g_KiHardwareTrigger, *g_KiHardwareTrigger);
        return TRUE;
    }

    DbgPrint("[Suppressor] Could not resolve KiHardwareTrigger\n");
    g_KiHardwareTrigger = nullptr;
    return FALSE;
}

BOOLEAN HvciResolveBugCheckStateVariables()
{
    DbgPrint("[Suppressor] === RESOLVING STATE VARIABLES ===\n");

    BOOLEAN ok1 = HvciResolveKiHardwareTrigger();

    BOOLEAN ok2 = FALSE;

    // Try export first
    UNICODE_STRING name;
    RtlInitUnicodeString(&name, L"KeFreezeExecution");
    PVOID keFreezeExecution = MmGetSystemRoutineAddress(&name);

    if (keFreezeExecution) {
        DbgPrint("[Suppressor] KeFreezeExecution: %p\n", keFreezeExecution);
        ok2 = HvciResolveBugCheckStateFromFreezeFunc(keFreezeExecution);
    }
    else {
        DbgPrint("[Suppressor] KeFreezeExecution not exported, scanning for KiFreezeTargetExecution\n");
        PVOID kiFreezeTarget = HvciScanForKiFreezeTargetExecution();
        if (kiFreezeTarget) {
            ok2 = HvciResolveBugCheckStateFromFreezeFunc(kiFreezeTarget);
        }
    }

    if (ok1 || ok2) {
        InterlockedExchange(&g_StateVarsResolved, 1);
        DbgPrint("[Suppressor] State variables resolved (KiHardwareTrigger=%s, KiBugCheckActive=%s)\n",
            ok1 ? "YES" : "NO", ok2 ? "YES" : "NO");
        return TRUE;
    }

    DbgPrint("[Suppressor] State variable resolution FAILED\n");
    return FALSE;
}

// ============================================================================
// HALPTIMERWATCHDOG RESOLUTION
//
// KeBugCheck2 calls HalTimerWatchdogStop early in its preamble. Once
// stopped, the timer-tick watchdog stays disarmed for the rest of the
// system's life - clock tick servicing degrades, and within 10-20s the
// DPC watchdog (0x133) escalates because its profile accumulator was
// also bumped during the long bugcheck preparation.
//
// There is no exported HalTimerWatchdogStart. We invert the operation
// by snapshotting HalpTimerWatchdog's bytes at driver init (when it is
// in known-good state) and bit-blitting the snapshot back over it after
// a successful suppression. This avoids hardcoding field offsets, which
// drift across LCUs within the same major build.
//
// Discovery: HalPrivateDispatchTable[+0x338] points at
// HalTimerWatchdogStop on dispatch-table v23+. Disassemble its prologue
// for the first RIP-relative .data reference - that is HalpTimerWatchdog.
// ============================================================================

// Walk back from any address inside a PE module to its MZ header.
// Returns image base via outBase and size via outSize.
BOOLEAN HvciFindModuleRangeFromAddress(
    _In_  PVOID  anyAddr,
    _Out_ PVOID* outBase,
    _Out_ PULONG outSize)
{
    *outBase = nullptr;
    *outSize = 0;

    if (!anyAddr || !MmIsAddressValid(anyAddr)) return FALSE;

    PUCHAR addr = (PUCHAR)((ULONG_PTR)anyAddr & ~0xFFFULL);
    for (ULONG i = 0; i < 0x4000; i++) {
        PUCHAR test = addr - (i * 0x1000);
        if (!MmIsAddressValid(test)) return FALSE;

        __try {
            if (*(PUSHORT)test == 0x5A4D) {
                LONG e_lfanew = *(LONG*)(test + 0x3C);
                if (e_lfanew > 0 && e_lfanew < 0x1000 &&
                    *(PULONG)(test + e_lfanew) == 0x4550) {
                    *outBase = test;
                    *outSize = *(PULONG)(test + e_lfanew + 0x50);
                    return TRUE;
                }
            }
        }
        __except (EXCEPTION_EXECUTE_HANDLER) {
            return FALSE;
        }
    }
    return FALSE;
}

// Test whether an address falls inside any writable section of a PE
// image. Used to filter brute-force RIP-relative scan candidates so we
// only accept references that point at modifiable HAL data, not code.
//
// Raw byte-offset PE walking (matches the style of HvciFindNtosBase) to
// avoid depending on IMAGE_* macros, which may not be pulled in by the
// WDK header set in use here.
//
// PE32+ section header layout (40 bytes each):
//   +0x00  Name[8]
//   +0x08  Misc.VirtualSize        (ULONG)
//   +0x0C  VirtualAddress          (ULONG)
//   +0x10  SizeOfRawData
//   +0x14  PointerToRawData
//   +0x18  PointerToRelocations
//   +0x1C  PointerToLinenumbers
//   +0x20  NumberOfRelocations
//   +0x22  NumberOfLinenumbers
//   +0x24  Characteristics         (ULONG)  - IMAGE_SCN_MEM_WRITE = 0x80000000
//
// Optional header layout (offsets from NT header start):
//   +0x06  NumberOfSections        (USHORT)
//   +0x14  SizeOfOptionalHeader    (USHORT)
// Section headers begin immediately after the optional header.
#define HVCI_SCN_MEM_WRITE          0x80000000UL
BOOLEAN HvciIsAddressInWritableSection(
    _In_ PVOID  modBase,
    _In_ ULONG  modSize,
    _In_ PVOID  target)
{
    if (!modBase || !target) return FALSE;

    PUCHAR base = (PUCHAR)modBase;
    PUCHAR tgt = (PUCHAR)target;
    if (tgt < base || tgt >= base + modSize) return FALSE;

    __try {
        if (*(PUSHORT)base != 0x5A4D) return FALSE;
        LONG e_lfanew = *(LONG*)(base + 0x3C);
        if (e_lfanew <= 0 || e_lfanew >= 0x1000) return FALSE;

        PUCHAR ntHdr = base + e_lfanew;
        if (*(PULONG)ntHdr != 0x4550) return FALSE;

        // FileHeader sits at ntHdr+4 (NT signature is 4 bytes).
        // FileHeader.NumberOfSections at +0x06 within FileHeader (so +0x06
        // from ntHdr+4 = ntHdr+0x06+4 = ntHdr+0x06... actually IMAGE_FILE_HEADER
        // starts after the 4-byte Signature. NumberOfSections is field 2,
        // at offset 0x02 within IMAGE_FILE_HEADER, so ntHdr + 4 + 2 = ntHdr+0x06.
        USHORT numSections = *(PUSHORT)(ntHdr + 0x06);
        // SizeOfOptionalHeader at offset 0x10 within IMAGE_FILE_HEADER
        // (ntHdr + 4 + 0x10 = ntHdr + 0x14).
        USHORT optHdrSize = *(PUSHORT)(ntHdr + 0x14);

        // First section header = ntHdr + sizeof(Signature) +
        // sizeof(FileHeader) + SizeOfOptionalHeader = ntHdr + 4 + 20 + opt.
        PUCHAR sect = ntHdr + 4 + 20 + optHdrSize;

        for (USHORT i = 0; i < numSections; i++) {
            PUCHAR shdr = sect + (i * 40);
            ULONG vSize = *(PULONG)(shdr + 0x08);
            ULONG vAddr = *(PULONG)(shdr + 0x0C);
            ULONG chars = *(PULONG)(shdr + 0x24);

            PUCHAR sStart = base + vAddr;
            PUCHAR sEnd = sStart + vSize;
            if (tgt >= sStart && tgt < sEnd) {
                return (chars & HVCI_SCN_MEM_WRITE) != 0;
            }
        }
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {
        return FALSE;
    }
    return FALSE;
}

// Brute-force scan a function prologue for the first RIP-relative
// memory reference whose target lands in a writable section of the
// containing module. We don't try to decode opcodes - we just locate
// any plausible (ModRM mode=00, rm=101) byte and try the three common
// instruction trailers (no-imm, imm8, imm32). The writable-section
// filter rejects refs that resolve into .text padding or read-only
// constants, which eliminates the false positives that a pure pattern
// match would otherwise generate.
PVOID HvciScanFirstRipRelInModule(
    _In_ PUCHAR funcStart,
    _In_ ULONG  scanLen,
    _In_ PVOID  modBase,
    _In_ ULONG  modSize)
{
    if (!funcStart || !modBase) return nullptr;

    __try {
        for (ULONG i = 0; i + 9 < scanLen; i++) {
            // ModRM with mod=00, rm=101 -> RIP+disp32 addressing.
            // Mask 0xC7 keeps mod and rm bits, ignores the reg field.
            UCHAR modrm = funcStart[i];
            if ((modrm & 0xC7) != 0x05) continue;

            INT32 disp = *(INT32*)&funcStart[i + 1];

            // Three instruction lengths to try, in order of likelihood
            // for HAL-style "test and clear" prologues:
            //   +1 imm: mov byte/cmp byte/and-or/etc with imm8
            //   +0:     plain mov/lea/xchg
            //   +4 imm: mov dword imm32
            ULONG ends[] = { i + 6, i + 5, i + 9 };
            for (ULONG t = 0; t < 3; t++) {
                PUCHAR resolved = funcStart + ends[t] + disp;
                if (HvciIsAddressInWritableSection(modBase, modSize, resolved)) {
                    return resolved;
                }
            }
        }
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {}

    return nullptr;
}

// Follow a single tail-call (E9 disp32 or FF 25 disp32) at the start of
// a function. HAL ships many "Stop" / "Start" / "Initialize" entries as
// 1-line thunks that jmp to a shared helper - the actual data reference
// to HalpTimerWatchdog lives in the helper, not the dispatch-table
// thunk. Returns the resolved target on success, or `funcStart`
// unchanged if the prologue does not start with a tail-call.
PUCHAR HvciFollowTailCall(_In_ PUCHAR funcStart, _In_ ULONG scanBytes)
{
    if (!funcStart || !MmIsAddressValid(funcStart)) return funcStart;

    __try {
        for (ULONG i = 0; i + 6 < scanBytes; i++) {
            UCHAR b = funcStart[i];

            // E9 disp32 - jmp rel32 (direct tail-jump)
            if (b == 0xE9) {
                INT32 disp = *(INT32*)&funcStart[i + 1];
                PUCHAR target = funcStart + i + 5 + disp;
                if (MmIsAddressValid(target)) return target;
                return funcStart;
            }

            // FF 25 disp32 - jmp [rip+disp32] (indirect tail-jump
            // through an import table; rare for HAL internals but
            // harmless to handle).
            if (b == 0xFF && funcStart[i + 1] == 0x25) {
                INT32 disp = *(INT32*)&funcStart[i + 2];
                PVOID* slot = (PVOID*)(funcStart + i + 6 + disp);
                if (MmIsAddressValid(slot)) {
                    PUCHAR target = (PUCHAR)*slot;
                    if (target && MmIsAddressValid(target)) return target;
                }
                return funcStart;
            }

            // Stop at a clear function-body boundary.
            if (b == 0xC3 || b == 0xC2 || b == 0xCC) break;
            // Single-byte prefixes / NOP-equivalents that may precede
            // the tail-jump (compiler-emitted ENDBR64 = F3 0F 1E FA).
            if (b == 0x90) continue;
            if (b == 0xF3 && i + 3 < scanBytes &&
                funcStart[i + 1] == 0x0F &&
                funcStart[i + 2] == 0x1E &&
                funcStart[i + 3] == 0xFA) {
                i += 3;     // skip ENDBR64 (4 bytes total, loop adds 1)
                continue;
            }
            break;
        }
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {}

    return funcStart;
}

BOOLEAN HvciResolveAndSnapshotHalTimerWatchdog()
{
    if (!MmIsAddressValid(HalPrivateDispatchTable)) {
        DbgPrint("[Suppressor] HalpTimerWatchdog: HalPrivateDispatchTable invalid\n");
        return FALSE;
    }

    PULONG_PTR dispatch = (PULONG_PTR)HalPrivateDispatchTable;
    ULONG idx = HAL_PDT_TIMER_WATCHDOG_STOP_OFFSET / sizeof(ULONG_PTR);
    PUCHAR halStop = (PUCHAR)dispatch[idx];

    if (!halStop || !MmIsAddressValid(halStop)) {
        DbgPrint("[Suppressor] HalpTimerWatchdog: HalTimerWatchdogStop entry invalid\n");
        return FALSE;
    }

    DbgPrint("[Suppressor] HalTimerWatchdogStop @ %p\n", halStop);

    // Detect a no-op stub. On dispatch-table v61 (build 26200) the slot
    // that *was* HalTimerWatchdogStop on older builds points to a 3-byte
    // function that just RETs (`C2 00 00` or `C3`). KeBugCheck2 calls it,
    // it returns immediately, no HalpTimerWatchdog is touched, and no
    // restoration is needed. Mark resolved-but-null so the restore path
    // becomes a no-op and we don't print the misleading "snapshot failed"
    // warning that suggested degraded recovery.
    __try {
        UCHAR b0 = halStop[0];
        UCHAR b1 = halStop[1];
        if (b0 == 0xC3 ||                                    // ret
            (b0 == 0xC2 && halStop[2] == 0x00) ||            // ret imm16
            (b0 == 0xF3 && b1 == 0xC3) ||                    // rep ret (legacy)
            (b0 == 0xCC)) {                                  // int3 (placeholder)
            DbgPrint("[Suppressor] HalTimerWatchdogStop is a no-op stub "
                "(first byte 0x%02X) - no watchdog state to snapshot\n", b0);
            HvciLog(HVCI_EV_HAL_WD_STUB, b0, 0, 0, 0, 0, 0);
            g_HalpTimerWatchdog = nullptr;
            InterlockedExchange(&g_HalpTimerWatchdogResolved, 1);
            return TRUE;
        }
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {
        DbgPrint("[Suppressor] HalpTimerWatchdog: stub-check faulted\n");
        return FALSE;
    }

    // Find HAL's base/size from the function address so we can validate
    // candidate RIP-relative targets fall inside HAL's writable .data,
    // not in some unrelated module or in .text.
    PVOID halBase = nullptr;
    ULONG halSize = 0;
    if (!HvciFindModuleRangeFromAddress(halStop, &halBase, &halSize)) {
        DbgPrint("[Suppressor] HalpTimerWatchdog: could not locate HAL base from %p\n",
            halStop);
        return FALSE;
    }
    DbgPrint("[Suppressor] HAL base: %p size: 0x%lx\n", halBase, halSize);

    // Two-pass scan with tail-call follow:
    //   pass 1: scan halStop directly (works for non-thunk functions)
    //   pass 2: if pass 1 fails, follow a tail-jump and re-scan
    // We follow at most one level - HAL helpers don't typically chain.
    PVOID resolved = HvciScanFirstRipRelInModule((PUCHAR)halStop, 0x100,
        halBase, halSize);

    if (!resolved) {
        PUCHAR realFn = HvciFollowTailCall((PUCHAR)halStop, 16);
        if (realFn != (PUCHAR)halStop) {
            DbgPrint("[Suppressor] HalTimerWatchdogStop is a thunk -> %p\n",
                realFn);
            // Make sure the tail-call target is still inside HAL.
            if ((PUCHAR)realFn >= (PUCHAR)halBase &&
                (PUCHAR)realFn < (PUCHAR)halBase + halSize) {
                resolved = HvciScanFirstRipRelInModule(realFn, 0x200,
                    halBase, halSize);
            }
            else {
                DbgPrint("[Suppressor] HalpTimerWatchdog: tail-call "
                    "target %p escapes HAL image, skipping\n", realFn);
            }
        }
    }

    if (!resolved) {
        DbgPrint("[Suppressor] HalpTimerWatchdog: no writable-section "
            "RIP-relative ref found in HalTimerWatchdogStop or its "
            "tail-call helper\n");
        // Dump first 0x40 bytes of halStop to the debugger so the user
        // can paste them and we can pattern-match exactly.
        __try {
            DbgPrint("[Suppressor] HalTimerWatchdogStop bytes:\n");
            for (ULONG row = 0; row < 4; row++) {
                DbgPrint("[Suppressor]   %p: "
                    "%02X %02X %02X %02X %02X %02X %02X %02X "
                    "%02X %02X %02X %02X %02X %02X %02X %02X\n",
                    halStop + row * 16,
                    halStop[row * 16 + 0], halStop[row * 16 + 1],
                    halStop[row * 16 + 2], halStop[row * 16 + 3],
                    halStop[row * 16 + 4], halStop[row * 16 + 5],
                    halStop[row * 16 + 6], halStop[row * 16 + 7],
                    halStop[row * 16 + 8], halStop[row * 16 + 9],
                    halStop[row * 16 + 10], halStop[row * 16 + 11],
                    halStop[row * 16 + 12], halStop[row * 16 + 13],
                    halStop[row * 16 + 14], halStop[row * 16 + 15]);
            }
        }
        __except (EXCEPTION_EXECUTE_HANDLER) {}
        return FALSE;
    }

    g_HalpTimerWatchdog = resolved;

    __try {
        RtlCopyMemory(g_HalpTimerWatchdogSnap, g_HalpTimerWatchdog,
            HVCI_HAL_WATCHDOG_SNAP_SIZE);
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {
        DbgPrint("[Suppressor] HalpTimerWatchdog: snapshot copy faulted\n");
        g_HalpTimerWatchdog = nullptr;
        return FALSE;
    }

    InterlockedExchange(&g_HalpTimerWatchdogResolved, 1);
    DbgPrint("[Suppressor] HalpTimerWatchdog @ %p, %lu-byte snapshot taken\n",
        g_HalpTimerWatchdog, (ULONG)HVCI_HAL_WATCHDOG_SNAP_SIZE);
    return TRUE;
}


// ============================================================================
// Release the bugcheck state so slave CPUs exit KeBugCheck2's spin.
// ============================================================================

__forceinline VOID HvciReleaseBugcheckState()
{
    if (g_KiHardwareTrigger) {
        InterlockedDecrement(g_KiHardwareTrigger);
    }
    if (g_KiFreezeExecutionLock) {
        InterlockedExchange(g_KiFreezeExecutionLock, 0);
    }
    if (g_KiFreezeLockBackup) {
        InterlockedExchange(g_KiFreezeLockBackup, 0);
    }
    _mm_mfence();

    // KiBugCheckActive cleared LAST - this releases slaves from
    // KeBugCheck2's outer loop.
    if (g_KiBugCheckActive) {
        InterlockedExchange(g_KiBugCheckActive, 0);
    }

    // IpiFrozen is an enum, not a flag:
    //   1 = freeze requested,  2 = thawed/active (steady state),
    //   4 = frozen & acknowledged
    // We must transition slaves from 1 or 4 to 2; writing 0 would leave
    // the IPI state machine desynchronized and a later KeGenericCallDpc
    // would spin forever. Master CPU never went through the state
    // machine - skip it.
    {
        ULONG masterCpu = KeGetCurrentProcessorIndex();
        for (ULONG i = 0; i < g_CompleteState.ValidProcessorCount; i++) {
            PPROCESSOR_CONTROL_STATE state = &g_CompleteState.ProcessorStates[i];
            if (!state->IsValid || !state->IpiFrozen) continue;
            if (i == masterCpu) continue;

            __try {
                LONG prev = InterlockedCompareExchange(
                    (volatile LONG*)state->IpiFrozen, 2, 4);
                if (prev != 4) {
                    InterlockedCompareExchange(
                        (volatile LONG*)state->IpiFrozen, 2, 1);
                }
            }
            __except (EXCEPTION_EXECUTE_HANDLER) {}
        }
    }

    _mm_mfence();
    KeMemoryBarrier();
}

// ============================================================================
// Bit-blit the init snapshot back over HalpTimerWatchdog. KeBugCheck2
// disarms the timer watchdog; without restoring it, clock-tick servicing
// stays degraded and the system trips 0x133 within 10-20s of resume.
// Called at HIGH_LEVEL with peers pinned - no DbgPrint, no scheduling.
// ============================================================================

__forceinline VOID HvciRestoreHalTimerWatchdog()
{
    if (!g_HalpTimerWatchdogResolved || !g_HalpTimerWatchdog) return;

    __try {
        RtlCopyMemory(g_HalpTimerWatchdog, g_HalpTimerWatchdogSnap,
            HVCI_HAL_WATCHDOG_SNAP_SIZE);
        _mm_mfence();
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {}
}

// ============================================================================
// DPC WATCHDOG RESET
//
// During the bugcheck preamble (which can hold a CPU off the clock-tick
// servicing path for tens of ms), every CPU's Prcb->DpcWatchdogCount
// keeps decrementing on each clock tick. After a successful suppression
// the accumulated decrement can push the counter past the 0x133
// (DPC_WATCHDOG_VIOLATION) threshold within seconds.
//
// Reload Count from Period on each CPU's own PRCB. Must run from a
// per-CPU context (gs is implicit), so it is broadcast via
// KeIpiGenericCall from the suppression-clear DPC at DISPATCH_LEVEL,
// well after the freeze window has closed.
// ============================================================================

ULONG_PTR HvciResetDpcWatchdogIpi(ULONG_PTR Ctx)
{
    UNREFERENCED_PARAMETER(Ctx);

    PKPRCB prcb = (PKPRCB)__readgsqword(0x20);
    if (!prcb || !MmIsAddressValid(prcb)) return 0;

    ULONG cpuIdx = KeGetCurrentProcessorIndex();

    __try {
        volatile PULONG period =
            (volatile PULONG)((PUCHAR)prcb + KPRCB_DPC_WATCHDOG_PERIOD_OFFSET);
        volatile PULONG count =
            (volatile PULONG)((PUCHAR)prcb + KPRCB_DPC_WATCHDOG_COUNT_OFFSET);

        ULONG p = *period;
        ULONG oldCount = *count;

        // Sanity-check Period before writing Count. Plausible kernel values
        // are 0x100 .. 0x100000 (build-dependent default ~0x12000 at 100Hz).
        // If Period reads as zero or wildly out of range we are almost
        // certainly looking at the wrong PRCB offset for this build, in
        // which case writing Count would corrupt some unrelated PRCB
        // field. Bail out instead - a single missed reset is recoverable
        // (timeout is minutes), but PRCB corruption is not.
        if (p < 0x100 || p > 0x100000) {
            HvciLog(HVCI_EV_DPC_WD_RESET, cpuIdx, p, oldCount,
                1 /*skipped*/, 0, 0);
            return 0;
        }

        InterlockedExchange((volatile LONG*)count, (LONG)p);
        HvciLog(HVCI_EV_DPC_WD_RESET, cpuIdx, p, oldCount,
            0 /*ok*/, 0, 0);
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {
        HvciLog(HVCI_EV_DPC_WD_RESET, cpuIdx, 0, 0, 1 /*skipped*/, 0, 0);
    }

    return 0;
}

// Detect a kCET supervisor shadow-stack token. The token occupies three
// consecutive slots {SAVED_SSP, LIP, CS} that the CPU writes on a
// transition to a higher-privilege/IST frame. SAVED_SSP is recognizable
// because its value equals (&slot + 24) - a self-referential pointer to
// where the previous shadow stack continues.
static BOOLEAN HvciIsSupervisorToken(_In_ PULONG64 base, _In_ ULONG idx)
{
    ULONG64 slotAddr = (ULONG64)&base[idx];
    ULONG64 slotVal = base[idx];

    if (slotVal != slotAddr + 24) return FALSE;
    if (!MmIsAddressValid(&base[idx + 1])) return FALSE;
    if (!MmIsAddressValid(&base[idx + 2])) return FALSE;

    ULONG64 cs = base[idx + 2];
    if (cs != 0x10 && cs != 0x33) return FALSE;       // kernel or user CS

    return TRUE;
}

// Walk shadow stack from current SSP, locate the suspended thread's
// pending return address by following the supervisor token if present.
// Returns TRUE on success and writes:
//   *outTargetRip - the bad driver's pending return address.
//   *outPopCount  - INCSSPQ count needed so that, after JMP to TargetRip,
//                   SSP[0] is the *next* entry above (caller-of-caller's
//                   pending RA, which the resumed function's RET will pop).
BOOLEAN HvciResolveTargetFromSsp(
    _Out_ PULONG64 outTargetRip,
    _Out_ PULONG   outPopCount,
    _In_  ULONG    maxDepth)
{
    *outTargetRip = 0;
    *outPopCount = 0;

    __try {
        ULONG64 ssp = HvciReadSsp();
        if (ssp == 0) return FALSE;

        PULONG64 base = (PULONG64)ssp;

        for (ULONG i = 0; i + 2 < maxDepth; i++) {
            if (!MmIsAddressValid(&base[i])) return FALSE;

            if (HvciIsSupervisorToken(base, i)) {
                ULONG64 savedSsp = base[i];
                if (!MmIsAddressValid((PVOID)savedSsp)) {
                    // bad token - keep walking, don't bail
                    continue;
                }

                ULONG64 ra = *(PULONG64)savedSsp;
                if (!MmIsAddressValid((PVOID)ra) ||
                    ra < 0xFFFF000000000000ULL) {
                    continue;
                }

                // Reject any candidate that points into our own module.
                // The token may belong to an unrelated older transition
                // whose savedSSP lands on a stale frame inside our hook
                // code; redirecting there causes #CP on the next RET.
                if (HvciIsOurAddress(ra)) {
                    HvciLog(HVCI_EV_TOKEN_TARGET_NOT_ON_RSP, ra,
                        i, 0, 0, 0, 0);
                    continue;
                }

                *outTargetRip = ra;
                // Pop count is intended for the asm stub's INCSSPQ, which
                // executes one CALL frame SHALLOWER than this rdsspq does
                // (rdsspq is 2 CALLs deep from the C hook: hook ->
                // HvciResolveTargetFromSsp -> HvciReadSsp; INCSSPQ is 1
                // CALL deep: hook -> HvciKcetJmpRestoreFixed). The asm-
                // context index of the supervisor token is therefore
                // (i - 1), not i. To pop the dispatcher chain, the 3-slot
                // token, AND the bad driver's RA at base[i+3] (consuming
                // it for JMP-to-caller semantics), the asm INCSSPQ count
                // is (i - 1) + 3 + 1 = i + 3.
                //
                // Empirically validated against build 26200 by computing
                // the exact savedSsp address: target post-pop SSP must
                // land at &caller_caller_RA on the kernel shadow stack.
                *outPopCount = i + 3;

                HvciLog(HVCI_EV_TOKEN_FOUND, i, savedSsp, ra,
                    *outPopCount, 0, 0);
                return TRUE;
            }
        }

        HvciLog(HVCI_EV_NO_TOKEN, maxDepth, 0, 0, 0, 0, 0);
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {
        HvciLog(HVCI_EV_RESOLVE_EXC, 0, 0, 0, 0, 0, 0);
    }

    return FALSE;
}

// Locate targetRip on the regular stack starting at rspStart. Returns the
// 8-byte index where it is stored, or -1 if not found within maxSlots.
LONG HvciFindRegularStackOffset(
    _In_ ULONG64 rspStart,
    _In_ ULONG64 targetRip,
    _In_ ULONG   maxSlots)
{
    __try {
        for (ULONG i = 0; i < maxSlots; i++) {
            PULONG64 slot = (PULONG64)(rspStart + (i * 8));
            if (!MmIsAddressValid(slot)) return -1;
            if (*slot == targetRip) return (LONG)i;
        }
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {}
    return -1;
}

// ============================================================================
// FIND DISPATCHER-CHAIN DEPTH ON SHADOW STACK
//
// Walks the current shadow stack from SSP[0] downward looking for the
// first supervisor-token-shaped slot. The number of entries above the
// token is the dispatcher chain depth - i.e., the number of CALL pushes
// from the trap entry up to our hook frame.
//
// For non-IST traps we use this as the INCSSPQ count: after popping
// dispatcher frames, SSP[0] becomes the bad driver's pending RA, which
// is what the resumed function's eventual RET will validate against.
//
// Returns -1 if no token found within maxDepth.
// ============================================================================
LONG HvciFindDispatcherDepth(_In_ ULONG maxDepth)
{
    __try {
        ULONG64 ssp = HvciReadSsp();
        if (ssp == 0) return -1;
        PULONG64 base = (PULONG64)ssp;

        for (ULONG i = 0; i + 2 < maxDepth; i++) {
            if (!MmIsAddressValid(&base[i + 2])) return -1;
            if (HvciIsSupervisorToken(base, i)) return (LONG)i;
        }
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {}
    return -1;
}

// ============================================================================
// MINIMAL X86-64 LENGTH DISASSEMBLER
//
// Returns the byte length of the instruction at `rip`, or 0 if the
// instruction is not in our supported set (caller bails to fallback).
//
// Coverage chosen to handle the instructions we observe at synthetic
// trap sites: legacy + REX prefixes, MOV variants, INT N, INT3, common
// 0x0F escapes, near CALL/JMP, and FF /2-/5 indirect control transfers.
// Anything outside this set returns 0 - we'd rather fall back than skip
// past an instruction we mis-sized.
// ============================================================================
ULONG HvciInstructionLength(_In_ ULONG64 rip)
{
    if (!MmIsAddressValid((PVOID)rip)) return 0;
    if (!MmIsAddressValid((PVOID)(rip + 15))) return 0;

    PUCHAR  p = (PUCHAR)rip;
    ULONG   len = 0;
    BOOLEAN sizeOverride = FALSE;
    BOOLEAN rexW = FALSE;
    BOOLEAN hasModRM = FALSE;
    ULONG   immSize = 0;

    // --- Legacy prefixes (max 4) ---
    for (ULONG i = 0; i < 4; i++) {
        UCHAR b = p[len];
        if (b == 0x66) { sizeOverride = TRUE; len++; continue; }
        if (b == 0x67 || b == 0x2E || b == 0x36 || b == 0x3E ||
            b == 0x26 || b == 0x64 || b == 0x65 ||
            b == 0xF0 || b == 0xF2 || b == 0xF3) {
            len++; continue;
        }
        break;
    }

    // --- REX prefix (one byte, must precede opcode) ---
    if ((p[len] & 0xF0) == 0x40) {
        if (p[len] & 0x08) rexW = TRUE;
        len++;
    }

    // --- Opcode ---
    UCHAR op = p[len++];

    switch (op) {
    case 0x90: case 0xC3: case 0xCB: case 0xCC:        // NOP, RET, RETF, INT3
    case 0xF4:                                          // HLT
        return len;

    case 0xCD:                                          // INT N
        return len + 1;

        // MOV r/m, r ; MOV r, r/m
    case 0x88: case 0x89: case 0x8A: case 0x8B:
        hasModRM = TRUE;
        immSize = 0;
        break;

        // MOV r/m8, imm8
    case 0xC6:
        hasModRM = TRUE;
        immSize = 1;
        break;

        // MOV r/m32/64, imm32 (sign-extended in 64-bit mode)
    case 0xC7:
        hasModRM = TRUE;
        immSize = sizeOverride ? 2 : 4;
        break;

        // MOV AL/AX/EAX/RAX, moffs (long mode: 8-byte address)
    case 0xA0: case 0xA1: case 0xA2: case 0xA3:
        return len + 8;

        // MOV r8, imm8  (B0+r)
    case 0xB0: case 0xB1: case 0xB2: case 0xB3:
    case 0xB4: case 0xB5: case 0xB6: case 0xB7:
        return len + 1;

        // MOV r16/32/64, imm  (B8+r)
    case 0xB8: case 0xB9: case 0xBA: case 0xBB:
    case 0xBC: case 0xBD: case 0xBE: case 0xBF:
        return len + (rexW ? 8 : (sizeOverride ? 2 : 4));

    case 0xE8: case 0xE9:                               // CALL/JMP near rel32
        return len + 4;

    case 0xEB:                                          // JMP rel8
        return len + 1;

    case 0x0F: {                                        // 2-byte opcode escape
        UCHAR op2 = p[len++];
        switch (op2) {
        case 0x05: case 0x07:                           // SYSCALL, SYSRET
        case 0x0B:                                      // UD2
        case 0x34: case 0x35:                           // SYSENTER, SYSEXIT
            return len;
        default:
            return 0;
        }
    }

    case 0xFF:                                          // /2 call, /4 jmp, etc
        hasModRM = TRUE;
        immSize = 0;
        break;

    default:
        return 0;
    }

    if (hasModRM) {
        UCHAR modrm = p[len++];
        UCHAR mod = (UCHAR)((modrm >> 6) & 0x3);
        UCHAR rm = (UCHAR)(modrm & 0x7);

        if (mod != 0x3) {
            // Memory operand
            BOOLEAN hasSib = (rm == 0x4);
            if (hasSib) {
                UCHAR sib = p[len++];
                UCHAR base = (UCHAR)(sib & 0x7);
                if (mod == 0x0 && base == 0x5) len += 4;        // disp32
            }
            else if (mod == 0x0 && rm == 0x5) {
                len += 4;                                       // RIP-rel disp32
            }

            if (mod == 0x1) len += 1;                            // disp8
            else if (mod == 0x2) len += 4;                       // disp32
        }
    }

    return len + immSize;
}

// Find an x64 IRETQ trap frame on the regular stack starting at rspStart.
//
// On a trap delivery (HVCI #VE, kernel #PF, __fastfail #SX, etc.) the CPU
// pushes a 5-quad frame at the new RSP:
//
//   [+ 0]  saved RIP        (kernel-canonical address)
//   [+ 8]  saved CS         (low 16 = selector, upper 48 = 0)
//   [+16]  saved RFLAGS     (low 32 = flags, upper 32 = 0; bit 1 always 1)
//   [+24]  saved RSP        (kernel-canonical address)
//   [+32]  saved SS         (low 16 = selector, upper 48 = 0)
//
// For IST-routed traps the new RSP is on the IST stack while the saved
// RSP at +24 still points to the bad driver's per-thread kernel stack.
// That recovered RSP is the only reliable handle on the bad driver's
// real frame from inside our hook (which runs many call frames removed
// from the trap point on a possibly-different stack).
//
// Returns the slot offset of the frame (where the saved RIP sits), or
// -1 if no plausible trap frame is found within maxSlots. Also writes
// the saved RIP and saved RSP through outBadRip / outBadRsp.
LONG HvciFindTrapFrame(
    _In_  ULONG64  rspStart,
    _In_  ULONG    maxSlots,
    _Out_ PULONG64 outBadRip,
    _Out_ PULONG64 outBadRsp)
{
    *outBadRip = 0;
    *outBadRsp = 0;

    __try {
        for (ULONG i = 0; i + 4 < maxSlots; i++) {
            PULONG64 base = (PULONG64)(rspStart + (i * 8));
            if (!MmIsAddressValid(&base[4])) return -1;

            ULONG64 rip = base[0];
            ULONG64 cs = base[1];
            ULONG64 rflags = base[2];
            ULONG64 rsp = base[3];
            ULONG64 ss = base[4];

            if (rip < 0xFFFF000000000000ULL) continue;
            if (!MmIsAddressValid((PVOID)rip)) continue;

            if (cs & 0xFFFFFFFFFFFF0000ULL) continue;
            USHORT csSel = (USHORT)cs;
            if (csSel != 0x10 && csSel != 0x33) continue;

            if (rflags & 0xFFFFFFFF00000000ULL) continue;
            if ((rflags & 0x2) == 0) continue;          // reserved-1

            if (rsp < 0xFFFF000000000000ULL) continue;
            if (!MmIsAddressValid((PVOID)rsp)) continue;

            if (ss & 0xFFFFFFFFFFFF0000ULL) continue;
            USHORT ssSel = (USHORT)ss;
            if (ssSel != 0x18 && ssSel != 0x2B && ssSel != 0) continue;

            *outBadRip = rip;
            *outBadRsp = rsp;
            return (LONG)i;
        }
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {}

    return -1;
}

// Cross-stack matcher. Walks the regular stack from rspStart looking for
// 8-byte values that ALSO appear on the shadow stack starting at SSP.
// The first such match is, by construction, the most recent pending
// return address - i.e. the bad driver's pending RA. Works whether or
// not a supervisor shadow-stack token is present (so it covers
// synthetic kernel-mode __fastfail, which does not push a token).
//
// On success writes:
//   *outTargetRip   - matching pending return address
//   *outRspOffset   - 8-byte offset on regular stack where it lives
//   *outSspIndex    - 8-byte index on shadow stack where it lives
BOOLEAN HvciResolveTargetByCrossStack(
    _In_  ULONG64  rspStart,
    _Out_ PULONG64 outTargetRip,
    _Out_ PLONG    outRspOffset,
    _Out_ PLONG    outSspIndex,
    _In_  ULONG    rspMaxSlots,
    _In_  ULONG    sspMaxDepth)
{
    *outTargetRip = 0;
    *outRspOffset = -1;
    *outSspIndex = -1;

    __try {
        ULONG64 ssp = HvciReadSsp();
        if (ssp == 0) return FALSE;

        PULONG64 sspBase = (PULONG64)ssp;

        for (ULONG i = 0; i < rspMaxSlots; i++) {
            PULONG64 rspSlot = (PULONG64)(rspStart + (i * 8));
            if (!MmIsAddressValid(rspSlot)) return FALSE;

            ULONG64 candidate = *rspSlot;
            if (!MmIsAddressValid((PVOID)candidate) ||
                candidate < 0xFFFF000000000000ULL) continue;

            // Skip RAs in our own module - never a valid bad-driver RA
            // and would cause #CP when the resumed function RETs.
            if (HvciIsOurAddress(candidate)) continue;

            for (ULONG j = 0; j < sspMaxDepth; j++) {
                if (!MmIsAddressValid(&sspBase[j])) break;
                if (sspBase[j] == candidate) {
                    *outTargetRip = candidate;
                    *outRspOffset = (LONG)i;
                    *outSspIndex = (LONG)j;
                    HvciLog(HVCI_EV_CROSS_MATCH, candidate, i, j,
                        0, 0, 0);
                    return TRUE;
                }
            }
        }
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {
        HvciLog(HVCI_EV_CROSS_EXC, 0, 0, 0, 0, 0, 0);
    }

    return FALSE;
}

// ============================================================================
// FREEZE HOOK - intercepts slave CPU freeze notifications
// ===========================================================================

NTSTATUS __stdcall HvciHookedNotifyProcessorFreeze(BOOLEAN Freeze)
{
    InterlockedIncrement(&g_FreezeHookEntries);

    // Deferred state-var resolution: if state vars aren't resolved yet,
    // and this is a freeze call (so caller is KiFreezeTargetExecution),
    // capture the return address and resolve from there.
    if (Freeze && !g_StateVarsResolved && !g_SuppressionInProgress) {
        PVOID retAddr = _ReturnAddress();
        if (retAddr) {
            PVOID funcBase = HvciFindFunctionStart(retAddr);
            if (funcBase) {
                if (HvciResolveBugCheckStateFromFreezeFunc(funcBase)) {
                    InterlockedExchange(&g_StateVarsResolved, 1);
                }
            }
        }
    }

    ULONG cpuIdx = KeGetCurrentProcessorIndex();
    if (cpuIdx >= MAX_PROCESSORS) cpuIdx = 0;

    if (!Freeze) {
        // Unfreeze path. If we previously suppressed the matching Freeze=TRUE
        // for this CPU, also suppress this Freeze=FALSE so HAL never sees the
        // unbalanced bookkeeping. Otherwise pass through to original.
        LONG count = InterlockedExchangeAdd(&g_FreezeSuppressedCount[cpuIdx], 0);
        if (count > 0) {
            InterlockedDecrement(&g_FreezeSuppressedCount[cpuIdx]);
            InterlockedIncrement(&g_FreezeHookSuppressFalse);
            HvciLog(HVCI_EV_FREEZE_HOOK, 0, cpuIdx, (ULONG64)(count - 1),
                1 /*eat*/, 0, 0);
            return STATUS_SUCCESS;
        }

        InterlockedIncrement(&g_FreezeHookPassThrough);
        HvciLog(HVCI_EV_FREEZE_HOOK, 0, cpuIdx, 0, 0 /*pass*/, 0, 0);
        if (HalNotifyProcessorFreezeOrig) {
            return HalNotifyProcessorFreezeOrig(Freeze);
        }
        return STATUS_SUCCESS;
    }

    // Freeze == TRUE
    if (!g_SuppressionInProgress) {
        // Normal pass-through (no bugcheck in flight on master).
        InterlockedIncrement(&g_FreezeHookPassThrough);
        HvciLog(HVCI_EV_FREEZE_HOOK, 1, cpuIdx, 0, 0 /*pass*/, 0, 0);
        if (HalNotifyProcessorFreezeOrig) {
            return HalNotifyProcessorFreezeOrig(Freeze);
        }
        return STATUS_SUCCESS;
    }

    // Suppression in progress: eat this Freeze=TRUE (do NOT call original,
    // which would spin in KiFreezeTargetExecution), and remember that we owe
    // the slave a matching Freeze=FALSE eat so its later unfreeze stays
    // balanced even if g_SuppressionInProgress is cleared by the cleanup DPC
    // before the slave gets that far.
    InterlockedIncrement(&g_FreezeSuppressedCount[cpuIdx]);
    InterlockedIncrement(&g_FreezeHookSuppressTrue);

    // Set this slave's IpiFrozen to THAWED(2), not 0. Writing 0 leaves
    // the IPI state machine desynchronized; subsequent IPI broadcasts
    // hang waiting for an ack from this CPU. See HvciReleaseBugcheckState
    // for the full rationale.
    PKPRCB prcb = (PKPRCB)__readgsqword(0x20);
    if (prcb && MmIsAddressValid(prcb)) {
        volatile PULONG ipiFrozen =
            (volatile PULONG)((PUCHAR)prcb + KPRCB_IPI_FROZEN_OFFSET);
        InterlockedExchange((volatile LONG*)ipiFrozen, 2);
        _mm_mfence();
    }

    HvciLog(HVCI_EV_FREEZE_HOOK, 1, cpuIdx,
        (ULONG64)g_FreezeSuppressedCount[cpuIdx], 1, 0, 0);
    return STATUS_SUCCESS;
}

// ============================================================================
// SYNCHRONIZED FREEZE HOOK INSTALL
// ============================================================================

ULONG_PTR HvciFreezeHookInstallIpi(ULONG_PTR Arg)
{
    PFREEZE_HOOK_INSTALL_CTX ctx = (PFREEZE_HOOK_INSTALL_CTX)Arg;

    if (InterlockedCompareExchange(&ctx->Completed, 1, 0) == 0) {
        ULONG_PTR oldVal = (ULONG_PTR)InterlockedExchangePointer(
            (PVOID*)ctx->DispatchEntry,
            (PVOID)ctx->NewValue);
        ctx->OldValue = oldVal;
    }

    _mm_mfence();
    return 0;
}

BOOLEAN HvciInstallFreezeHookSynchronized()
{
    if (InterlockedCompareExchange(&g_FreezeHookActive, 1, 0) != 0) {
        return TRUE;
    }

    if (!MmIsAddressValid(HalPrivateDispatchTable)) {
        InterlockedExchange(&g_FreezeHookActive, 0);
        return FALSE;
    }

    PULONG_PTR dispatchArray = (PULONG_PTR)HalPrivateDispatchTable;
    ULONG index = HAL_PDT_NOTIFY_PROCESSOR_FREEZE_OFFSET / sizeof(ULONG_PTR);
    PULONG_PTR entry = &dispatchArray[index];

    if (!MmIsAddressValid(entry)) {
        InterlockedExchange(&g_FreezeHookActive, 0);
        return FALSE;
    }

    ULONG_PTR currentFunc = *entry;
    if (!currentFunc || !MmIsAddressValid((PVOID)currentFunc)) {
        InterlockedExchange(&g_FreezeHookActive, 0);
        return FALSE;
    }

    HalNotifyProcessorFreezeOrig = (PHAL_NOTIFY_PROCESSOR_FREEZE)currentFunc;

    DbgPrint("[Suppressor] Installing freeze hook via synchronized IPI\n");
    DbgPrint("[Suppressor]   Original: %p\n", (PVOID)currentFunc);
    DbgPrint("[Suppressor]   New:      %p\n", HvciHookedNotifyProcessorFreeze);

    FREEZE_HOOK_INSTALL_CTX ctx = { 0 };
    ctx.DispatchEntry = entry;
    ctx.NewValue = (ULONG_PTR)HvciHookedNotifyProcessorFreeze;
    ctx.OldValue = 0;
    ctx.Completed = 0;

    KeIpiGenericCall(HvciFreezeHookInstallIpi, (ULONG_PTR)&ctx);

    DbgPrint("[Suppressor] Freeze hook installed (synchronized)\n");
    return TRUE;
}

VOID HvciUninstallFreezeHookSynchronized()
{
    if (InterlockedCompareExchange(&g_FreezeHookActive, 0, 1) != 1) {
        return;
    }

    if (!HalNotifyProcessorFreezeOrig || !HalPrivateDispatchTable) return;

    PULONG_PTR dispatchArray = (PULONG_PTR)HalPrivateDispatchTable;
    ULONG index = HAL_PDT_NOTIFY_PROCESSOR_FREEZE_OFFSET / sizeof(ULONG_PTR);

    FREEZE_HOOK_INSTALL_CTX ctx = { 0 };
    ctx.DispatchEntry = &dispatchArray[index];
    ctx.NewValue = (ULONG_PTR)HalNotifyProcessorFreezeOrig;
    ctx.Completed = 0;

    KeIpiGenericCall(HvciFreezeHookInstallIpi, (ULONG_PTR)&ctx);

    HalNotifyProcessorFreezeOrig = nullptr;
    DbgPrint("[Suppressor] Freeze hook uninstalled\n");
}

VOID HvciFreezeHookInstallWorker(PVOID Context)
{
    UNREFERENCED_PARAMETER(Context);

    LARGE_INTEGER delay;
    delay.QuadPart = -10000000LL;  // 1s
    KeDelayExecutionThread(KernelMode, FALSE, &delay);

    DbgPrint("[Suppressor] Deferred freeze hook install starting\n");
    HvciInstallFreezeHookSynchronized();
}

// ============================================================================
// SUPPRESSION FLAG CLEAR DPC
// ============================================================================

VOID HvciClearSuppressionFlag(PKDPC Dpc, PVOID Ctx, PVOID A1, PVOID A2)
{
    UNREFERENCED_PARAMETER(Dpc);
    UNREFERENCED_PARAMETER(Ctx);
    UNREFERENCED_PARAMETER(A1);
    UNREFERENCED_PARAMETER(A2);

    // Reload Prcb->DpcWatchdogCount on every CPU. The bugcheck preamble
    // can stall a CPU off the clock-tick path long enough to push us
    // across the 0x133 threshold within seconds of resume. KeIpiGenericCall
    // requires DISPATCH_LEVEL (would deadlock at HIGH_LEVEL inside the hook).
    KeIpiGenericCall(HvciResetDpcWatchdogIpi, 0);

    InterlockedExchange(&g_SuppressionInProgress, 0);
    DbgPrint("[Suppressor] Bypass!\n");
}

// ============================================================================
// MAIN MASTER HOOK - kCET-AWARE BUGCHECK SUPPRESSION
// ============================================================================
VOID __stdcall HvciHalPrepareForBugcheckHook(BOOLEAN NmiFlag)
{
    UNREFERENCED_PARAMETER(NmiFlag);

    InterlockedExchange(&g_SuppressionInProgress, 1);

    // Reentrancy: a second bugcheck while we're mid-suppression must
    // forward to the original handler so the system gets a normal BSOD
    // instead of looping back through us with corrupted state.
    if (InterlockedCompareExchange(&g_InHookGuard, 1, 0) != 0) {
        if (HalPrepareForBugcheckOrig) {
            HalPrepareForBugcheckOrig(NmiFlag);
        }
        return;
    }

    PCONTEXT savedContext = nullptr;
    UCHAR savedIrql = DISPATCH_LEVEL;
    ULONG64 badDriverReturnAddr = 0;
    BOOLEAN kcetEnabledLocal = (g_KcetEnabled != 0);

    __try {
        // No DbgPrint in this path - we are at HIGH_LEVEL with peers
        // pinned by the bugcheck freeze IPI; DbgPrint would deadlock
        // against them via KdEnterDebugger.
        HvciLog(HVCI_EV_HOOK_ENTRY, NmiFlag, 0, 0, 0, 0, 0);

        ULONG64 currentSsp = HvciReadSsp();
        if (currentSsp != 0 && !kcetEnabledLocal) {
            kcetEnabledLocal = TRUE;
        }

        PKPRCB prcb = (PKPRCB)__readgsqword(0x20);
        if (!prcb || !MmIsAddressValid(prcb)) {
            HvciLog(HVCI_EV_BAD_PRCB, 0, 0, 0, 0, 0, 0);
            goto fallback;
        }

        PCONTEXT* ctxPtrLoc = (PCONTEXT*)((PUCHAR)prcb + KPRCB_CONTEXT_OFFSET);
        savedContext = *ctxPtrLoc;
        if (!savedContext || !MmIsAddressValid(savedContext)) {
            HvciLog(HVCI_EV_BAD_CONTEXT, 0, 0, 0, 0, 0, 0);
            goto fallback;
        }

        if ((savedContext->ContextFlags & CONTEXT_AMD64_BIT) != CONTEXT_AMD64_BIT) {
            HvciLog(HVCI_EV_BAD_CTX_FLAGS, savedContext->ContextFlags,
                0, 0, 0, 0, 0);
            goto fallback;
        }

        savedIrql = *(PUCHAR)((PUCHAR)prcb + KPRCB_DEBUGGER_SAVED_IRQL_OFFSET);
        if (savedIrql > 15) savedIrql = DISPATCH_LEVEL;


        InterlockedIncrement((volatile LONG*)&g_SuppressedBugchecks);

        if (!MmIsAddressValid((PVOID)savedContext->Rsp)) {
            HvciLog(HVCI_EV_BAD_RSP, savedContext->Rsp, 0, 0, 0, 0, 0);
            goto fallback;
        }

        // Kernel-origin detection - HIGH_LEVEL safe. We only do memory
        // reads via MmIsAddressValid (HvciFindTrapFrame is implemented
        // that way); we do NOT call any Rtl* unwind helper here. Rtl*
        // unwind APIs may take RtlpDynamicFunctionTableLock, and pinned
        // peers can hold that lock indefinitely - calling them at HIGH_LEVEL
        // with peers in KiFreezeTargetExecution deadlocks the master CPU
        // before any output, "Bypass!" never fires, system freezes solid.
        //
        // Detection rule: parse the IRETQ trap frame the CPU pushed when
        // delivering the trap. The saved RIP in that frame is the
        // faulting instruction's address. If it lies inside ntoskrnl,
        // the bugcheck originated in the kernel itself - typical sources:
        //   * PspProcessDelete `int 29h` (FAST_FAIL_INCORRECT_STACK,
        //     list-entry corruption check)
        //   * KASAN-style invariant checks
        //   * PatchGuard self-checks
        //
        // For kernel-origin we MUST skip the SEH-unwind recovery path:
        // with no bad-driver frame, the walker would either exhaust
        // UNWIND_MAX_DEPTH (waste) or, with stale image ranges, dispatch
        // to a misclassified frame and freeze the system post-"Bypass!"
        // (locks held, kCET SSP/RSP mismatched, garbage in RAX).
        //
        // Recovery still runs - via the asm skip-instruction path. For
        // int 29h (2 bytes: cd 29) the resolver computes
        //   targetRip = badRip + 2
        // which lands on the instruction following the fastfail. For
        // PspProcessDelete (`je +0x97 / mov ecx,3 / int 29h / <cont>`)
        // that is the same continuation as if the je had been taken.
        BOOLEAN kernelOrigin = FALSE;
        {
            ULONG64 koBadRip = 0;
            ULONG64 koBadRsp = 0;
            LONG koTrapOff = HvciFindTrapFrame(savedContext->Rsp, 512,
                &koBadRip, &koBadRsp);
            if (koTrapOff >= 0 && koBadRip != 0 &&
                g_NtosBase && g_NtosSize) {
                ULONG64 ntBase = (ULONG64)g_NtosBase;
                if (koBadRip >= ntBase && koBadRip < ntBase + g_NtosSize) {
                    kernelOrigin = TRUE;
                    HvciLog(HVCI_EV_FALLBACK, 0xC0DE0001ULL, koBadRip, 0,
                        0, 0, 0);
                }
            }
        }

        // Resolve the bad-driver context. savedContext was captured deep
        // inside the dispatcher so its RSP is the dispatcher's frame, not
        // the bad driver's. We use the shadow stack (kCET path) or a
        // trap-frame parse to recover the bad driver's actual RIP/RSP.
        // Asm-fallback recovery only - the SEH unwind in
        // HvciSehUnwindRecovery uses image-zone classification instead and
        // does not consume any of these locals.
        ULONG64 targetRip = 0;
        ULONG64 targetRsp = 0;
        ULONG64 targetRax = 0xC0000001;     // STATUS_UNSUCCESSFUL
        ULONG   sspPopCount = 0;
        LONG    rspOffset = -1;
        LONG    sspIndex = -1;
        BOOLEAN skipInstruction = FALSE;
        ULONG64 rspBase = savedContext->Rsp;

        if (kcetEnabledLocal) {
            BOOLEAN haveToken = HvciResolveTargetFromSsp(&targetRip,
                &sspPopCount, 64);

            if (haveToken) {
                // For IST-routed traps savedContext->Rsp points at the
                // IST stack; the bad driver's frame lives on a different
                // per-thread kernel stack reachable only via the IRETQ
                // trap frame the CPU pushed on the IST stack.
                ULONG64 badRip = 0;
                ULONG64 badRsp = 0;
                LONG trapOff = HvciFindTrapFrame(savedContext->Rsp, 512,
                    &badRip, &badRsp);

                if (trapOff >= 0 &&
                    badRsp != 0 &&
                    !HvciIsOurAddress(badRip)) {

                    HvciLog(HVCI_EV_TRAP_FRAME, trapOff, badRip, badRsp,
                        0, 0, 0);

                    // Stack-delta classifier. IST-routed traps (e.g. NMI,
                    // __fastfail #SX) save RSP on a different stack from
                    // our current one (delta on the order of MB). Non-IST
                    // traps (HVCI write -> kernel-mode #PF) keep the same
                    // stack with a few-KB delta - those use skip-instruction.
                    ULONG64 stackDelta = (savedContext->Rsp > badRsp)
                        ? (savedContext->Rsp - badRsp)
                        : (badRsp - savedContext->Rsp);
                    BOOLEAN isIst = (stackDelta >= 0x10000);

                    if (!isIst) {
                        // Non-IST: skip the faulting instruction. The bad
                        // driver continues at the next instruction; its
                        // RET will pop the regular-stack RA (untouched).
                        // INCSSPQ count = dispatcherDepth + 2 lands SSP[0]
                        // on the bad driver's caller RA - empirically
                        // verified for build 26200.
                        ULONG instLen = HvciInstructionLength(badRip);
                        if (instLen == 0) goto fallback;

                        LONG dispatcherDepth = HvciFindDispatcherDepth(64);
                        if (dispatcherDepth < 0) goto fallback;

                        targetRip = badRip + instLen;
                        targetRsp = badRsp;
                        targetRax = savedContext->Rax;
                        sspPopCount = (ULONG)dispatcherDepth + 2;
                        skipInstruction = TRUE;
                        rspOffset = 0;
                    }
                    else {
                        // IST: use the bad driver's per-thread RSP.
                        rspBase = badRsp;
                        rspOffset = HvciFindRegularStackOffset(rspBase,
                            targetRip, 256);
                        if (rspOffset < 0) goto fallback;
                    }
                }
                else {
                    // No trap frame parse: search the current stack for
                    // the resolver's targetRip directly.
                    rspOffset = HvciFindRegularStackOffset(rspBase,
                        targetRip, 512);
                    if (rspOffset < 0) goto fallback;
                }
            }
            else {
                // No supervisor token (CPL=0->CPL=0, no IST switch).
                // Cross-stack matcher is best-effort but the only signal
                // available without a real unwinder.
                if (!HvciResolveTargetByCrossStack(savedContext->Rsp,
                    &targetRip, &rspOffset, &sspIndex, 256, 128)) {
                    goto fallback;
                }
                // Off-by-one correction: sspIndex was measured at
                // HvciReadSsp's rdsspq (2 CALLs deep from C hook). The
                // asm INCSSPQ runs 1 CALL deep. JMP-to-caller wants
                // post-pop SSP[0] = pre-pop SSP[sspIndex+1] (= the entry
                // above the matched RA, which becomes caller's eventual
                // RET target). In rdsspq context that's pop sspIndex+1;
                // in asm context one-less = sspIndex.
                sspPopCount = (ULONG)sspIndex;
            }
        }
        else {
            // Non-kCET path. The original *savedContext->Rsp heuristic
            // is unreliable - savedRsp captured deep inside KeBugCheck2,
            // so the value at the top of that stack is some KeBugCheck2
            // local, NOT the bad driver's pending RA. Empirically this
            // resolves to garbage (e.g. inter-function 0xCC padding),
            // and resuming there hits int 3 or worse.
            //
            // Use the same trap-frame parser the kCET non-IST path uses:
            // walk up from savedRsp looking for the IRETQ frame the CPU
            // pushed when delivering the trap, recover the bad driver's
            // RIP/RSP, and skip-instruction. No shadow stack to update.
            ULONG64 badRip = 0;
            ULONG64 badRsp = 0;
            LONG trapOff = HvciFindTrapFrame(savedContext->Rsp, 512,
                &badRip, &badRsp);

            if (trapOff < 0 || badRsp == 0 || HvciIsOurAddress(badRip)) {
                HvciLog(HVCI_EV_TOKEN_TARGET_NOT_ON_RSP,
                    badRip, savedContext->Rsp, 1, 0, 0, 0);
                goto fallback;
            }

            HvciLog(HVCI_EV_TRAP_FRAME, trapOff, badRip, badRsp,
                0, 0, 0);

            ULONG instLen = HvciInstructionLength(badRip);
            if (instLen == 0) {
                HvciLog(HVCI_EV_INST_DECODE_FAIL, badRip, 0, 0, 0, 0, 0);
                goto fallback;
            }

            targetRip = badRip + instLen;
            targetRsp = badRsp;
            targetRax = savedContext->Rax;
            sspPopCount = 0;             // no shadow stack on this path
            skipInstruction = TRUE;
            rspOffset = 0;             // satisfy validation below

            HvciLog(HVCI_EV_SKIP_INSTRUCTION,
                badRip, instLen, targetRip, 0, 0, 0);
        }

        // Kernel-origin recovery override. The kCET path above resolves
        // skip-instruction targets that resume the faulting function
        // mid-stride. That works for bad-driver scenarios where the
        // fault is incidental and post-fault state is otherwise sound.
        // For kernel-origin (e.g. PspProcessDelete `int 29h` list-entry
        // check) the fastfail is the kernel reporting its OWN state is
        // corrupt - the corruption is still there at instruction+2, so
        // the function faults again immediately and the system hangs.
        //
        // Replace skip-instruction with early-return-to-caller: walk
        // up the regular kernel stack from `badRsp` (the IRETQ-saved
        // RSP, i.e. the faulting function's RSP) cross-matched against
        // the shadow stack. The first match is the function's pending
        // return address. We resume there with RAX=STATUS_UNSUCCESSFUL,
        // abandoning the function's body. The caller treats the call
        // as having failed - which for cleanup paths is benign (the
        // process is leaked, system continues).
        if (kernelOrigin && skipInstruction) {
            ULONG64 koBadRip2 = 0;
            ULONG64 koBadRsp2 = 0;
            HvciFindTrapFrame(savedContext->Rsp, 512,
                &koBadRip2, &koBadRsp2);

            LONG    koRspOff = -1;
            LONG    koSspIdx = -1;
            ULONG64 koCallerRip = 0;
            if (koBadRsp2 != 0 &&
                HvciResolveTargetByCrossStack(koBadRsp2,
                    &koCallerRip, &koRspOff, &koSspIdx, 256, 128)) {

                HvciLog(HVCI_EV_FALLBACK, 0xC0DE0002ULL, koCallerRip,
                    (ULONG64)koRspOff, (ULONG64)koSspIdx, 0, 0);
                targetRip = koCallerRip;
                rspBase = koBadRsp2;
                rspOffset = koRspOff;
                sspIndex = koSspIdx;
                sspPopCount = (ULONG)koSspIdx;
                skipInstruction = FALSE;   // recompute targetRsp/Rax below
            }
            else {
                HvciLog(HVCI_EV_FALLBACK, 0xC0DE0003ULL, 0, 0, 0, 0, 0);
            }
        }

        if (!MmIsAddressValid((PVOID)targetRip) ||
            targetRip < 0xFFFF000000000000ULL) {
            HvciLog(HVCI_EV_INVALID_TARGET, targetRip, 0, 0, 0, 0, 0);
            goto fallback;
        }

        if (rspOffset < 0) {
            HvciLog(HVCI_EV_NO_RSP_OFF, 0, 0, 0, 0, 0, 0);
            goto fallback;
        }

        badDriverReturnAddr = targetRip;

        HvciLog(HVCI_EV_TARGET_RESOLVED, targetRip, savedContext->Rip,
            rspBase, savedIrql, rspOffset, sspPopCount);

        if (!skipInstruction) {
            targetRsp = rspBase + ((ULONG64)rspOffset + 1) * 8;
            targetRax = 0xC0000001;
        }

        if (g_StateVarsResolved) {
            HvciReleaseBugcheckState();
            HvciRestoreHalTimerWatchdog();
        }

        // 500ms is the maximum the freeze hook may eat Freeze=TRUE calls
        // before swallowing legitimate freezes (clock maintenance, I/O completion DPCs) and stalling the IPI sender forever.
        LARGE_INTEGER clearDelay;
        clearDelay.QuadPart = -5000000LL;
        KeSetTimer(&g_SuppressionClearTimer, clearDelay, &g_SuppressionClearDpc);

        // Lower CR8 to the kernel-original IRQL before the unwind so
        // RtlUnwindEx and the __finally handlers run at the IRQL the
        // original frames expect (not at HIGH_LEVEL). RtlUnwindEx itself
        // requires interrupts on.
        _disable();
        __writecr8(savedIrql);
        _enable();
        InterlockedExchange(&g_InHookGuard, 0);

        // Diagnostic: state release and IRQL lower complete. Peers are
        // unfrozen, IRQL is at savedIrql, DbgPrint is safe again.
        DbgPrint("[Suppressor] Recover: kOrigin=%d tRip=%p tRsp=%p "
            "tRax=%p sspPop=%lu skipI=%d\n",
            (int)kernelOrigin, (PVOID)targetRip, (PVOID)targetRsp,
            (PVOID)targetRax, sspPopCount, (int)skipInstruction);

        // Preferred path: SEH unwind. Walks frames from this hook to the
        // resume target, runs every __finally on the way, then resumes
        // there with RAX = targetRax. For the write-to-RX case the
        // caller is IopLoadDriver, which then does its standard
        // driver-failed-to-load cleanup. For kernel-origin (no
        // bad-driver frame) we use the kernel-origin variant that takes
        // the cross-stack-resolved target directly - this runs the
        // intermediate ntoskrnl frames' __finally handlers, releasing
        // locks/critical-regions that the asm-restore-only path leaks.
        // Both variants return only on failure; on success never return.
        if (kernelOrigin) {
            NTSTATUS koStatus = HvciKernelOriginUnwindRecovery(
                savedContext, targetRip, targetRsp, targetRax);
            HvciLog(HVCI_EV_FALLBACK, 0xC0DE0004ULL, (ULONG64)koStatus,
                0, 0, 0, 0);
            // If we get here the kernel-origin unwind couldn't deliver -
            // fall through to the asm restore as a last resort.
        }
        else {
            NTSTATUS unwindStatus = HvciSehUnwindRecovery(savedContext,
                targetRax);
            HvciLog(HVCI_EV_FALLBACK, (ULONG64)unwindStatus, 0, 0, 0, 0, 0);
        }

        DbgPrint("[Suppressor] Asm-restore: jumping\n");

        // Asm restore. For kernel-origin this is the primary recovery
        // path. For bad-driver scenarios it is a fallback reached only
        // when SEH unwind couldn't find suitable .pdata.
        _disable();
        if (kcetEnabledLocal) {
            HvciKcetJmpRestoreFixed(savedContext, targetRip, targetRsp,
                targetRax, sspPopCount);
        }
        else {
            HvciJmpRestore(savedContext, targetRip, targetRsp, targetRax);
        }

        __assume(0);    // DECLSPEC_NORETURN
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {
        HvciLog(HVCI_EV_HOOK_EXCEPTION, GetExceptionCode(), 0, 0, 0, 0, 0);
    }

fallback:
    // We are still inside the bugcheck/IPI window - peers are pinned in
    // KiIpiWaitForRequestBarrier. No DbgPrint/DbgBreakPoint here (both
    // route through KdEnterDebugger and would deadlock against the
    // pinned peers). Defer to the original handler so Windows produces
    // a normal BSOD/dump. Bugcheck state flags are intentionally left
    // untouched - the original path needs them.
    HvciLog(HVCI_EV_FALLBACK, 0, 0, 0, 0, 0, 0);
    InterlockedIncrement((volatile LONG*)&g_FailedSuppressions);
    InterlockedExchange(&g_SuppressionInProgress, 0);
    InterlockedExchange(&g_InHookGuard, 0);

    if (HalPrepareForBugcheckOrig) {
        HalPrepareForBugcheckOrig(NmiFlag);
    }
}

// ============================================================================
// HOOK INSTALLATION
// ============================================================================

NTSTATUS HvciInstallHooks()
{
    if (g_HalHookInstalled) return STATUS_ALREADY_INITIALIZED;

    DbgPrint("[Suppressor] Installing hooks...\n");

    if (!HvciBuildProcessorControlStates()) {
        DbgPrint("[Suppressor] Failed to build processor states\n");
        return STATUS_UNSUCCESSFUL;
    }

    // Resolve bugcheck state variables (KiBugCheckActive, etc.)
    if (!HvciResolveBugCheckStateVariables()) {
        DbgPrint("[Suppressor] WARNING: state resolution failed - slaves may hang\n");
    }

    // Locate HalpTimerWatchdog and snapshot its bytes while
    // still in known-good state. Without this snapshot, the post-recovery
    // restore is a no-op and the system trips 0x133 ~10-20s after the
    // first suppressed bugcheck.
    if (!HvciResolveAndSnapshotHalTimerWatchdog()) {
        DbgPrint("[Suppressor] WARNING: HalpTimerWatchdog snapshot failed - "
            "non-IST recovery will degrade to soft-freeze\n");
    }

    if (!MmIsAddressValid(HalPrivateDispatchTable)) {
        return STATUS_ACCESS_VIOLATION;
    }

    __try {
        PULONG versionPtr = (PULONG)HalPrivateDispatchTable;
        ULONG halVersion;

        if (!HvciSafeReadMemory(versionPtr, &halVersion, sizeof(ULONG))) {
            return STATUS_UNSUCCESSFUL;
        }

        DbgPrint("[Suppressor] HAL Version: %lu\n", halVersion);

        PULONG_PTR dispatchArray = (PULONG_PTR)HalPrivateDispatchTable;

        KIRQL oldIrql = KeRaiseIrqlToDpcLevel();

        __try {
            ULONG index = HAL_PDT_PREPARE_FOR_BUGCHECK_OFFSET / sizeof(ULONG_PTR);
            PULONG_PTR entryPtr = &dispatchArray[index];

            if (!MmIsAddressValid(entryPtr)) {
                KeLowerIrql(oldIrql);
                return STATUS_UNSUCCESSFUL;
            }

            ULONG_PTR originalFunc = *entryPtr;
            if (!originalFunc || !MmIsAddressValid((PVOID)originalFunc)) {
                KeLowerIrql(oldIrql);
                return STATUS_UNSUCCESSFUL;
            }

            HalPrepareForBugcheckOrig = (PHAL_PREPARE_FOR_BUGCHECK)originalFunc;

            InterlockedExchangePointer((PVOID*)entryPtr,
                (PVOID)HvciHalPrepareForBugcheckHook);

            DbgPrint("[Suppressor] Installed HalPrepareForBugcheck: %p -> %p\n",
                HalPrepareForBugcheckOrig, HvciHalPrepareForBugcheckHook);
        }
        __finally {
            KeLowerIrql(oldIrql);
        }

        _mm_mfence();
        KeMemoryBarrier();

        g_HalHookInstalled = TRUE;
        InterlockedExchange((volatile LONG*)&g_SuppressorActive, TRUE);

        // NOTE: HvciPrestampIpiFrozenAllCpus() removed - on build 26200 it
        // hangs the kernel at init. Setting IpiFrozen != 0 on every CPU
        // when no bugcheck is in progress appears to interfere with normal
        // scheduler/IPI-dispatch flow. We set the marker only on the
        // slave-freeze path (when a slave actually enters freeze) and on
        // the release path - never at init.

        DbgPrint("[Suppressor] ========================================\n");
        DbgPrint("[Suppressor] HOOKS INSTALLED\n");
        DbgPrint("[Suppressor] g_KcetEnabled raw value: %ld\n", g_KcetEnabled);
        DbgPrint("[Suppressor] kCET mode: %s\n", g_KcetEnabled ? "ON" : "OFF");
        DbgPrint("[Suppressor] ========================================\n");

        return STATUS_SUCCESS;
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {
        DbgPrint("[Suppressor] Hook install exception: 0x%08X\n", GetExceptionCode());
        return STATUS_UNSUCCESSFUL;
    }
}

// ============================================================================
// HOOK REMOVAL
// ============================================================================

VOID HvciUninstallHooks()
{
    if (!g_HalHookInstalled) return;

    DbgPrint("[Suppressor] Uninstalling hooks...\n");

    __try {
        InterlockedExchange((volatile LONG*)&g_SuppressorActive, FALSE);
        KeStallExecutionProcessor(20000);

        if (HalPrepareForBugcheckOrig && MmIsAddressValid(HalPrivateDispatchTable)) {
            PULONG_PTR dispatchArray = (PULONG_PTR)HalPrivateDispatchTable;
            ULONG index = HAL_PDT_PREPARE_FOR_BUGCHECK_OFFSET / sizeof(ULONG_PTR);

            ULONG_PTR originalFunc = (ULONG_PTR)HalPrepareForBugcheckOrig;
            InterlockedExchangePointer((PVOID*)&dispatchArray[index],
                (PVOID)originalFunc);
            DbgPrint("[Suppressor] HalPrepareForBugcheck restored\n");
        }

        g_HalHookInstalled = FALSE;
    }
    __except (EXCEPTION_EXECUTE_HANDLER) {
        DbgPrint("[Suppressor] Uninstall exception: 0x%08X\n", GetExceptionCode());
    }
}

// DRIVER ENTRY / UNLOAD
// ============================================================================

VOID DriverUnload(PDRIVER_OBJECT DriverObject)
{
    UNREFERENCED_PARAMETER(DriverObject);

    DbgPrint("[Suppressor] ========================================\n");
    DbgPrint("[Suppressor] Driver unloading\n");
    DbgPrint("[Suppressor] STATISTICS:\n");
    DbgPrint("[Suppressor]   Bugchecks suppressed:  %lu\n", g_SuppressedBugchecks);
    DbgPrint("[Suppressor]   Failed suppressions:   %lu\n", g_FailedSuppressions);
    DbgPrint("[Suppressor] ========================================\n");

    HvciUninstallFreezeHookSynchronized();
    HvciUninstallHooks();

    KeCancelTimer(&g_SuppressionClearTimer);

    if (g_FreezeHookInstallWorkItem) {
        ExFreePoolWithTag(g_FreezeHookInstallWorkItem, 'FhkI');
        g_FreezeHookInstallWorkItem = nullptr;
    }
}

extern "C" NTSTATUS DriverEntry(PDRIVER_OBJECT DriverObject, PUNICODE_STRING RegistryPath)
{
    UNREFERENCED_PARAMETER(RegistryPath);

    DbgPrint("[Suppressor] ========================================\n");
    DbgPrint("[Suppressor] HVCI/kCET-aware bugcheck suppressor loading\n");
    DbgPrint("[Suppressor] Build: %lu\n", GetWindowsBuildNumber());
    DbgPrint("[Suppressor] ========================================\n");

    RtlZeroMemory(&g_CompleteState, sizeof(g_CompleteState));

    // Resolve our own image range from DriverObject so we can reject
    // resolver candidates that point into our hook code (those would
    // cause a kCET shadow-stack mismatch on the resumed RET).
    if (DriverObject && DriverObject->DriverStart && DriverObject->DriverSize) {
        g_OurBase = DriverObject->DriverStart;
        g_OurSize = DriverObject->DriverSize;
        DbgPrint("[Suppressor] Our image: %p size=0x%lx\n",
            g_OurBase, g_OurSize);
    }

    // Resolve ntoskrnl base+size up front. The kernel-origin guard in the
    // bugcheck hook compares each frame's RIP against [g_NtosBase,
    // g_NtosBase+g_NtosSize); if the size is wrong the guard cannot
    // distinguish kernel-internal bugchecks (e.g. PspProcessDelete
    // int 29h list-corruption check) from real bad-driver faults.
    if (!HvciFindNtosBase()) {
        DbgPrint("[Suppressor] WARNING: ntoskrnl resolution failed - "
            "kernel-origin guard will be conservative\n");
    }

    // Step 1: Detect kCET state
    BOOLEAN kcetDetected = HvciDetectKcet();
    InterlockedExchange(&g_KcetEnabled, kcetDetected ? 1 : 0);
    DbgPrint("[Suppressor] kCET state: %s\n",
        g_KcetEnabled ? "ENABLED" : "DISABLED");

    // Step 2: Verify offsets
    HvciVerifyKprcbOffsets();

    // Brief pause for debugger
    LARGE_INTEGER delay;
    delay.QuadPart = -10000000LL;  // 1 second
    KeDelayExecutionThread(KernelMode, FALSE, &delay);

    // Step 3: Initialize deferred infrastructure
    KeInitializeTimer(&g_SuppressionClearTimer);
    KeInitializeDpc(&g_SuppressionClearDpc, HvciClearSuppressionFlag, NULL);

    g_FreezeHookInstallWorkItem = (PWORK_QUEUE_ITEM)ExAllocatePool2(
        POOL_FLAG_NON_PAGED, sizeof(WORK_QUEUE_ITEM), 'FhkI');
    if (g_FreezeHookInstallWorkItem) {
        ExInitializeWorkItem(g_FreezeHookInstallWorkItem,
            HvciFreezeHookInstallWorker, NULL);
        ExQueueWorkItem(g_FreezeHookInstallWorkItem, DelayedWorkQueue);
    }

    // Step 4: Install master hook
    NTSTATUS status = HvciInstallHooks();
    if (!NT_SUCCESS(status)) {
        DbgPrint("[Suppressor] Hook installation failed: 0x%08X\n", status);

        if (g_FreezeHookInstallWorkItem) {
            ExFreePoolWithTag(g_FreezeHookInstallWorkItem, 'FhkI');
            g_FreezeHookInstallWorkItem = nullptr;
        }
        return status;
    }

    DriverObject->DriverUnload = DriverUnload;

    DbgPrint("[Suppressor] ========================================\n");
    DbgPrint("[Suppressor] DRIVER LOADED SUCCESSFULLY\n");
    DbgPrint("[Suppressor] Mode: %s\n",
        g_KcetEnabled ? "kCET unwind" : "direct restore");
    DbgPrint("[Suppressor] ========================================\n");

    return STATUS_SUCCESS;
}
```

`README.md`:

```md
# BugcheckSuppressor

**HVCI/kCET-aware bugcheck suppressor PoC**

Catches Windows kernel bugchecks before they BSOD the system. Uses SEH
unwind through `__finally` handlers instead of patching kernel code.
Tested on **Windows 11 24H2 build 26200** with HVCI + kCET enabled.


---

## Notice
Treat this PoC as a thought experiment: 
"how far can we bend kernel to suppress a bugcheck without writing any code into the kernel at runtime under HVCI?" 
The longer-term motivation is exploring whether a true PatchGuard suppressor is feasible on HVCI+kCET systems.

This implementation is a foothold for that question, not the answer to it.
It shows that bugcheck dispatch can be intercepted with data-only hooks,
that recovery can be driven entirely through `RtlUnwindEx`, and that the resulting flow is shadow-stack clean. 
That's the easy half of the PatchGuard problem: catching the bugcheck. 
The hard half is surviving after the catch when the bugcheck originates inside ntoskrnl itself
(e.g. PG's own `KeBugCheckEx 0x109` from a DPC, or the kernel's own
list-entry validation `int 29h` from `PspProcessDelete`).

---


## What it does

When a kernel bugcheck would normally take the system down,
BugcheckSuppressor:

1. **Hooks the dispatch table.**
   `HalPrivateDispatchTable[HalpPrepareForBugcheck]` and
   `[HalNotifyProcessorFreeze]` are data pointers, so we can hook them
   without triggering HVCI.
2. **Catches the bugcheck on entry.** When a bugcheck fires, the hook
   runs *before* the BSOD presentation.
3. **Releases the bugcheck globals** so the slave CPUs that were spinning
   in `KiFreezeTargetExecution` exit their wait loop. The system is no
   longer frozen.
4. **Unwinds the kernel stack via `RtlUnwindEx`** from the hook frame
   back to the bad driver's caller (typically `IopLoadDriver`), running
   every `__finally` handler on the way. That cleans up the
   `IopLoadDriver` side resources.
5. **Resumes execution** at the caller with `RAX = STATUS_ACCESS_VIOLATION`.
   From the caller's perspective, the offending function returned with an
   error, the same way it would have if SEH had naturally caught the
   violation.

The bad driver fails to load, the user gets an `ERROR_NOACCESS`-class
status, and the system stays running.
```

```
## How it works (technical detail)

### The hook is data-only

`HalPrivateDispatchTable` is a writable data page that exports function
pointers. HVCI does not protect data pages, only .text. We swap two
slots (`HalpPrepareForBugcheck` at offset `0x108`,
`HalNotifyProcessorFreeze` at offset `0x1A8`) atomically via
`InterlockedExchangePointer`.

### Bugcheck path entry

When `KeBugCheckEx` is called:

```
KeBugCheckEx
  KeBugCheck2
    HalpPrepareForBugcheck     <- our hook fires here
      ... freezes other CPUs via IPI ...
      ... captures CONTEXT into PRCB+0x8FC0 ...
```

The hook reads the saved `CONTEXT` from KPRCB, identifies the trap
RIP/RSP, and decides whether the fault originated in a third-party
driver (recoverable via SEH unwind) or inside ntoskrnl (best-effort
kernel-origin recovery).

### Bad-driver-origin recovery (`HvciSehUnwindRecovery`)

Walks frames from the hook upward via `RtlVirtualUnwind`. Identifies
the "bad-driver frame" by image classification: the first transition
from known-image (ntoskrnl plus the suppressor) to unknown is the bad
driver's faulting RIP. Walks one more frame to find the bad driver's
caller (typically `IopLoadDriver`), then dispatches `RtlUnwindEx` to
that frame with `RAX = STATUS_ACCESS_VIOLATION`. Every `__finally`
between the hook frame and the target runs, releasing locks, dropping
references, restoring per-thread state.

This is the path the suppressor handles cleanly end-to-end.

### Kernel-origin recovery (`HvciKernelOriginUnwindRecovery`)

Used when the bugcheck originates inside ntoskrnl (e.g a fault inside
`PspProcessDelete`). No bad driver image to anchor on, so the
cross-stack resolver picks a target frame from the saved RSP and we
`RtlUnwindEx` towards it, walking as deep as `RtlVirtualUnwind` reaches.
That maximizes the number of `__finally` handlers run. Limited by
ntoskrnl's lock acquisition style.

### Asm fallback (`cet_stubs.asm`)

If `RtlUnwindEx` cannot deliver (`.pdata` gap, RSP mismatch, etc.) we
fall back to a kCET-aware asm stub (`HvciKcetJmpRestoreFixed`) that
pops the shadow stack by a precomputed count, restores registers from
`CONTEXT`, switches RSP, and indirect-JMPs to the target. It works,
but it does not run intermediate `__finally` blocks, so locks held by
walked-over frames leak.

### State releases on the way out

Before unwinding, the hook:

- Releases `KiBugCheckActive` / `KiHardwareTrigger` / `KiFreezeExecutionLock`
  so peer CPUs exit their freeze loops.
- Resets per-PRCB DPC watchdog `Count = Period` so the post-recovery
  system doesn't trip bugcheck `0x133` 10 to 20 seconds later from
  accumulated decrement.
- Restores `nt!HalpTimerWatchdog` from a known-good snapshot taken at
  driver init, so clock-tick servicing returns to normal.
- Pre-stamps every `PRCB->IpiFrozen = 5` so the freeze IPI's slave loop
  sees "already frozen with our marker" and exits immediately.

These cleanups are what allow the system to remain usable after the
suppression. Without them, you'd see a brief survival followed by a
0x133 or another freeze a few seconds later.

## Achievements

1. **End-to-end bugcheck suppression on HVCI+kCET (Win11 24H2 26200).**
   Bad driver origin faults (write-to-RX, kernel exception, fastfail
   from third-party driver code) are caught and the system continues
   running. The bad driver fails to load with `STATUS_ACCESS_VIOLATION`,
   the same way it would have if SEH had naturally caught the fault.

2. **All hooks are data-only.** No write to ntoskrnl `.text`, no patch
   of any `.text` byte, nothing for HVCI to reject.

3. **Recovery uses `RtlUnwindEx`**, the kernel's own SEH dispatcher,
   which is aware of shadow stack. The 2022 ROP-via-suspended-thread
   approach is killed by kCET and this approach is not.

---

## Limitations

### Production-relevant limits

- **Build-specific KPRCB offsets.** The suppressor uses hardcoded
  offsets for KPRCB context save area, debugger-saved IRQL, IpiFrozen, DPC watchdog, etc. 
  These are stable within a Windows servicing branch
  but drift roughly 0x10 to 0x40 between LCUs. Verify against `ntoskrnl.pdb` when porting.

### Recovery boundary

- **Kernel-origin bugchecks at arbitrary points are "best effort", not
  guaranteed.** Many ntoskrnl functions acquire locks via explicit
  `KeAcquireSpinLock` / `KeReleaseSpinLock` pairs without surrounding
  `__try/__finally`. When `RtlUnwindEx` walks past those frames, the
  `__finally` handlers that would release them simply don't exist. The
  locks leak. Whatever next thread tries to acquire them deadlocks.
  This affects contrived test cases like injecting a null deref into
  `PspProcessDelete` via WinDbg's `r rdi=0`. Real-world kernel-origin
  scenarios at well-defined points (PG bugchecks via
  `KeBugCheckEx 0x109`, kASAN-style invariant checks) tend to be more
  recoverable because their callers are designed with "recovery friendly
  state".
- **Bugchecks during very early or very late system state** (boot
  before HAL is ready, late shutdown after the driver is unloaded) are
  not in scope.
- **Bugchecks during extremely deep stack contexts** may exceed
  `RtlVirtualUnwind`'s ability to walk. The asm fallback handles these
  but leaks intermediate `__finally`s.
- The system will freeze if BugcheckSuppressor cannot recover the
  exception.


### Operational

- **Unloading the driver is not fully clean.** Unhooking restores the
  dispatch pointers atomically, but if a bugcheck is in flight at the
  moment of unload, behaviour is undefined. Don't unload while running
  stress tests.
- **One suppressor instance per machine.** Multiple instances would
  race on the dispatch-table swap.
- **`/CETCOMPAT` build is required.** Without it, the asm fallback's
  `incsspq` instructions trap.
```
